import { Component, createSignal } from 'solid-js';
import { Button } from '@repo/ui';
import { Login } from './Login';
import { TodoList } from './components/TodoList';
import { MaterialsView } from './components/MaterialsView';
import { TaxonomyView } from './components/TaxonomyView';
import { ProjectDefinitionsView } from './components/ProjectDefinitionsView';
import { UnitOverridesView } from './components/UnitOverridesView';
import { logger } from '@repo/logger';

const App: Component = () => {
    const [token, setToken] = createSignal<string | null>(null);
    const [view, setView] = createSignal<'materials' | 'taxonomy' | 'projects' | 'units'>('materials');

    return (
        <div>
            {!token() ? (
                <Login onLogin={(t) => setToken(t)} />
            ) : (
                <div style={{ padding: '20px', "max-width": "800px", margin: "0 auto" }}>
                    <div style={{ display: "flex", "justify-content": "space-between", "align-items": "center" }}>
                        <h1>Material Admin</h1>
                        <Button onClick={() => {
                            logger.info('Logged out');
                            setToken(null);
                        }}>Logout</Button>
                    </div>

                    <div style={{ display: "flex", gap: "10px", margin: "20px 0" }}>
                        <button onClick={() => setView('materials')} class={view() === 'materials' ? 'font-bold underline' : ''}>Materials</button>
                        <button onClick={() => setView('taxonomy')} class={view() === 'taxonomy' ? 'font-bold underline' : ''}>Taxonomy</button>
                        <button onClick={() => setView('projects')} class={view() === 'projects' ? 'font-bold underline' : ''}>Projects</button>
                        <button onClick={() => setView('units')} class={view() === 'units' ? 'font-bold underline' : ''}>Units</button>
                    </div>

                    <hr />

                    {view() === 'materials' && <MaterialsView token={token()!} />}
                    {view() === 'taxonomy' && <TaxonomyView />}
                    {view() === 'projects' && <ProjectDefinitionsView />}
                    {view() === 'units' && <UnitOverridesView />}
                </div>
            )}
        </div>
    );
};

export default App;
