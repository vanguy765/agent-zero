// CREATE SCOPED CLIENT HELPER
const getScopedClient = (c: any) => {
    const token = c.req.header('Authorization')?.replace('Bearer ', '');
    if (token) {
        return createClient(SUPABASE_URL, SUPABASE_ANON_KEY, {
            global: { headers: { Authorization: `Bearer ${token}` } }
        });
    }
    // For testing without auth - use service role to bypass RLS
    return createClient(SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY);
};

// CREATE WORK ORDER (from template)
workOrdersRoute.post('/work-orders', async (c) => {
    const supabase = getScopedClient(c);
    const body = await c.req.json();

    const result = CreateWorkOrderRequestSchema.safeParse(body);
    if (!result.success) {
        return c.json({ error: result.error }, 400);
    }

    const { unit_id, template_id, assigned_contractor_id } = result.data;

    // 1. Get the template
    const { data: template, error: templateError } = await supabase
        .from('work_order_templates')
        .select('*')
        .eq('id', template_id)
        .single();

    if (templateError) {
        return c.json({ error: 'Template not found' }, 404);
    }

    // 2. Get organization_id from unit
    const { data: unit, error: unitError } = await supabase
        .from('units')
        .select('organization_id')
        .eq('id', unit_id)
        .single();

    if (unitError) {
        return c.json({ error: 'Unit not found' }, 404);
    }

    // 3. Create work order
    const { data: workOrder, error: woError } = await supabase
        .from('work_orders')
        .insert({
            organization_id: unit.organization_id,
            unit_id,
            template_id,
            trade_name: template.trade_name,
            assigned_contractor_id,
            status: 'pending'
        })
        .select()
        .single();

    if (woError) {
        return c.json({ error: woError.message }, 500);
    }

    // 4. Get rooms for this unit (prefer unit_rooms mappings, fall back to all org rooms)
    const { data: unitRooms, error: unitRoomsError } = await supabase
        .from('unit_rooms')
        .select('room_id, rooms(id, name, display_order)')
        .eq('unit_id', unit_id);

    if (unitRoomsError) {
        return c.json({ error: unitRoomsError.message }, 500);
    }

    let rooms: any[];
    if (unitRooms && unitRooms.length > 0) {
        // Use only rooms mapped to this unit, sorted by display_order
        rooms = unitRooms
            .map((ur: any) => ur.rooms)
            .sort((a: any, b: any) => (a.display_order ?? 0) - (b.display_order ?? 0));
    } else {
        // Fallback: all rooms for the organization
        const { data: orgRooms, error: orgRoomsError } = await supabase
            .from('rooms')
            .select('*')
            .eq('organization_id', unit.organization_id)
            .order('display_order', { ascending: true });

        if (orgRoomsError) {
            return c.json({ error: orgRoomsError.message }, 500);
        }
        rooms = orgRooms || [];
    }

    // 5. Create tasks for each room (snapshot template definition)
    const tasks = rooms.map((room, index) => ({
        work_order_id: workOrder.id,
        room_id: room.id,
        row_order: index,
        task_data: {
            steps: Object.values(template.definition.sections).flatMap((section: any) =>
                section.columns.map((col: any) => ({
                    key: col.key,
                    label: col.label,
                    status: 'pending',
                    not_applicable: col.default_na || false
                }))
            )
        }
    }));

    const { error: tasksError } = await supabase
        .from('work_order_tasks')
        .insert(tasks);

    if (tasksError) {
        return c.json({ error: tasksError.message }, 500);
    }

    return c.json(workOrder, 201);
});

// UPDATE TASK STEP
workOrdersRoute.patch('/work-orders/:workOrderId/tasks/:taskId', async (c) => {
    const supabase = getScopedClient(c);
    const user = (c.get('user') as any) || { id: 'test-user' }; // Fallback for testing
    const taskId = c.req.param('taskId');
    const body = await c.req.json();

    const result = UpdateTaskStepRequestSchema.safeParse(body);
    if (!result.success) {
        return c.json({ error: result.error }, 400);
    }

    const { step_key, status, not_applicable } = result.data;

    // Get current task
    const { data: task, error: fetchError } = await supabase
        .from('work_order_tasks')
        .select('task_data')
        .eq('id', taskId)
        .single();

    if (fetchError) {
        return c.json({ error: fetchError.message }, 404);
    }

    // Update the specific step
    const updatedSteps = task.task_data.steps.map((step: any) => {
        if (step.key === step_key) {
            return {
                ...step,
                status,
                not_applicable: not_applicable ?? step.not_applicable,
                completed_by: status === 'complete' ? user.id : step.completed_by,
                completed_at: status === 'complete' ? Date.now() : step.completed_at
            };
        }
        return step;
    });

    // Save updated task
    const { data, error } = await supabase
        .from('work_order_tasks')
        .update({ task_data: { steps: updatedSteps } })
        .eq('id', taskId)
        .select()
        .single();

    if (error) {
        return c.json({ error: error.message }, 500);
    }
    return c.json(data);
});
