CREATE VIEW public.v_budget_tracker AS
 WITH trade_budget AS (
         SELECT c.trade_key,
            u.unit_number,
            u.floor,
            r.name AS room_name,
            sum(COALESCE(c.price_estimated, c.value_numeric)) FILTER (WHERE ((c.category = ANY (ARRAY['cost'::text, 'cost_actual'::text])) AND (c.unit_of_measure = '$'::text))) AS estimated,
            sum(c.price_actual) FILTER (WHERE ((c.category = ANY (ARRAY['cost'::text, 'cost_actual'::text])) AND (c.unit_of_measure = '$'::text))) AS actual,
            string_agg(DISTINCT c.product_estimated, ', '::text) FILTER (WHERE (c.product_estimated IS NOT NULL)) AS products_planned,
            string_agg(DISTINCT c.product_actual, ', '::text) FILTER (WHERE (c.product_actual IS NOT NULL)) AS products_installed,
            sum(c.hours_estimated) FILTER (WHERE (c.hours_estimated IS NOT NULL)) AS hours_planned,
            sum(c.hours_actual) FILTER (WHERE (c.hours_actual IS NOT NULL)) AS hours_worked,
            avg(c.percent_complete) FILTER (WHERE ((c.percent_complete IS NOT NULL) AND (c.percent_complete > (0)::numeric))) AS pct_complete
           FROM (((public.unit_room_calculations c
             JOIN public.unit_rooms ur ON ((ur.id = c.unit_room_id)))
             JOIN public.units u ON ((u.id = ur.unit_id)))
             JOIN public.rooms r ON ((r.id = ur.room_id)))
          GROUP BY c.trade_key, u.unit_number, u.floor, r.name
        )
 SELECT trade_key,
    unit_number,
    floor,
    room_name,
    estimated,
    actual,
    (actual - estimated) AS variance,
        CASE
            WHEN (estimated > (0)::numeric) THEN round((((actual - estimated) / estimated) * (100)::numeric), 1)
            ELSE NULL::numeric
        END AS variance_pct,
    products_planned,
    products_installed,
        CASE
            WHEN ((products_planned IS DISTINCT FROM products_installed) AND (products_installed IS NOT NULL)) THEN true
            ELSE false
        END AS product_changed,
    hours_planned,
    hours_worked,
    (hours_worked - hours_planned) AS hours_variance,
    pct_complete
   FROM trade_budget
  WHERE ((estimated IS NOT NULL) OR (actual IS NOT NULL))
  ORDER BY trade_key, floor, unit_number, room_name;
