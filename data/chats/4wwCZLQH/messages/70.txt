import { OpenAPIHono, createRoute, z } from '@hono/zod-openapi';
import { createClient } from '@supabase/supabase-js';

const projectDefinitionsRouter = new OpenAPIHono();

const supabaseUrl = process.env.SUPABASE_URL || 'http://localhost:54321';
const supabaseKey = process.env.SUPABASE_SERVICE_ROLE_KEY || 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZS1kZW1vIiwicm9sZSI6InNlcnZpY2Vfcm9sZSIsImV4cCI6MTk4MzgxMjk5Nn0.EGIM96RAZx35lJzdJsyH-qQwv8Hdp7fsn3W0YpN81IU';
const supabase = createClient(supabaseUrl, supabaseKey);

const ProjectDefinitionSchema = z.object({
  project_id: z.string().uuid(),
  taxonomy_version: z.string().nullable(),
  floorplan_definitions: z.record(z.any()), // JSONB
});

const UpdateDefinitionSchema = z.object({
  floorplan_definitions: z.record(z.any()),
});

// GET /projects (List)
projectDefinitionsRouter.openapi(
  createRoute({
    method: 'get',
    path: '/',
    responses: {
      200: {
        description: 'List Projects',
        content: {
          'application/json': {
            schema: z.array(z.object({
              id: z.string().uuid(),
              name: z.string(),
              code: z.string(),
            })),
          },
        },
      },
    },
  }),
  async (c) => {
    const { data, error } = await supabase
      .from('projects')
      .select('id, name, code');

    if (error) {
      return c.json({ error: error.message }, 500);
    }
    return c.json(data);
  }
);

// GET /projects/:id/definitions
projectDefinitionsRouter.openapi(
  createRoute({
    method: 'get',
    path: '/{id}/definitions',
    responses: {
      200: {
        description: 'Get Project Definitions',
        content: {
          'application/json': {
            schema: ProjectDefinitionSchema.nullable(),
          },
        },
      },
      404: { description: 'Project not found' }
    },
  }),
  async (c) => {
    const id = c.req.param('id');
    const { data, error } = await supabase
      .from('project_definitions')
      .select('*')
      .eq('project_id', id)
      .single();

    if (error && error.code !== 'PGRST116') { // PGRST116 is "The result contains 0 rows"
       return c.json({ error: error.message }, 500);
    }
    
    // Return null if not found (frontend should handle it) or create default?
    return c.json(data || null);
  }
);

// PUT /projects/:id/definitions
projectDefinitionsRouter.openapi(
  createRoute({
    method: 'put',
    path: '/{id}/definitions',
    request: {
      body: {
        content: {
          'application/json': {
            schema: UpdateDefinitionSchema,
          },
        },
      },
    },
    responses: {
      200: {
        description: 'Updated Project Definition',
        content: {
            'application/json': {
              schema: ProjectDefinitionSchema,
            },
          },
      },
      500: { description: 'Error updating definitions' }
    },
  }),
  async (c) => {
    const id = c.req.param('id');
    const { floorplan_definitions } = c.req.valid('json');

    const { data, error } = await supabase
      .from('project_definitions')
      .upsert({
        project_id: id,
        floorplan_definitions
      }, { onConflict: 'project_id' })
      .select()
      .single();

    if (error) {
        console.error('Error updating project definitions:', error);
      return c.json({ error: error.message }, 500);
    }

    return c.json(data);
  }
);

export default projectDefinitionsRouter;
