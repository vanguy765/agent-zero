-- Organizations (Multi-tenancy)
CREATE TABLE IF NOT EXISTS public.organizations (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name TEXT NOT NULL,
    slug TEXT UNIQUE NOT NULL,
    settings JSONB DEFAULT '{}'::jsonb,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- Units (Physical spaces being renovated)
CREATE TABLE IF NOT EXISTS public.units (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id UUID NOT NULL REFERENCES public.organizations(id) ON DELETE CASCADE,
    unit_number TEXT NOT NULL,
    unit_type TEXT, -- e.g., "A", "B", "C" (from architectural drawings)
    floor INTEGER,
    area_sf NUMERIC,
    status TEXT DEFAULT 'not_started', -- not_started, in_progress, complete
    metadata JSONB DEFAULT '{}'::jsonb, -- For unit-specific notes, special instructions
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,
    UNIQUE(organization_id, unit_number)
);

-- Rooms (Spaces within units)
CREATE TABLE IF NOT EXISTS public.rooms (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id UUID NOT NULL REFERENCES public.organizations(id) ON DELETE CASCADE,
    name TEXT NOT NULL, -- e.g., "Kitchen", "Bedroom", "Bathroom"
    display_order INTEGER DEFAULT 0,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- Work Order Templates (Definitions of work for each trade)
CREATE TABLE IF NOT EXISTS public.work_order_templates (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id UUID NOT NULL REFERENCES public.organizations(id) ON DELETE CASCADE,
    trade_name TEXT NOT NULL, -- e.g., "Painting", "Carpentry", "Plumbing"
    version INTEGER DEFAULT 1,
    is_active BOOLEAN DEFAULT true,
    definition JSONB NOT NULL, -- The template structure (columns, logic rules, etc.)
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,
    UNIQUE(organization_id, trade_name, version)
);

-- Work Orders (Instance of a template for a specific unit)
CREATE TABLE IF NOT EXISTS public.work_orders (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id UUID NOT NULL REFERENCES public.organizations(id) ON DELETE CASCADE,
    unit_id UUID NOT NULL REFERENCES public.units(id) ON DELETE CASCADE,
    template_id UUID NOT NULL REFERENCES public.work_order_templates(id) ON DELETE RESTRICT,
    trade_name TEXT NOT NULL, -- Denormalized for easy filtering
    status TEXT DEFAULT 'pending', -- pending, in_progress, complete, on_hold
    assigned_contractor_id UUID, -- References auth.users (contractor)
    inspector_id UUID, -- References auth.users (inspector)
    start_date TIMESTAMP WITH TIME ZONE,
    completion_date TIMESTAMP WITH TIME ZONE,
    inspector_signoff_date TIMESTAMP WITH TIME ZONE,
    metadata JSONB DEFAULT '{}'::jsonb, -- Contractor info, special notes
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,
    UNIQUE(organization_id, unit_id, trade_name)
);

-- TEMPLATE HIERARCHY SYSTEM
-- ============================================================================
-- This migration implements a three-tier template hierarchy:
-- 1. App-level (system defaults)
-- 2. Company-level (tenant defaults)
-- 3. Project-level (project-specific)

BEGIN;

-- ============================================================================
-- 1. CREATE PROJECTS TABLE (Work Orders)
-- ============================================================================

CREATE TABLE IF NOT EXISTS wo_projects (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id UUID NOT NULL REFERENCES organizations(id) ON DELETE CASCADE,
    name TEXT NOT NULL,
    description TEXT,
    status TEXT NOT NULL DEFAULT 'active' CHECK (status IN ('active', 'completed', 'archived')),
    metadata JSONB DEFAULT '{}',
    created_at TIMESTAMPTZ DEFAULT now(),
    updated_at TIMESTAMPTZ DEFAULT now(),
    
    CONSTRAINT unique_wo_project_name_per_org UNIQUE (organization_id, name)
);

CREATE INDEX idx_wo_projects_org ON wo_projects(organization_id);
CREATE INDEX idx_wo_projects_status ON wo_projects(status);

COMMENT ON TABLE wo_projects IS 'Work Order Projects/properties within an organization';

-- ============================================================================
-- 2. UPDATE WORK_ORDER_TEMPLATES TABLE
-- ============================================================================

-- Add new columns
ALTER TABLE work_order_templates
    ADD COLUMN IF NOT EXISTS scope TEXT NOT NULL DEFAULT 'company'
        CHECK (scope IN ('app', 'company', 'project')),
    ADD COLUMN IF NOT EXISTS project_id UUID REFERENCES wo_projects(id) ON DELETE CASCADE,
    ADD COLUMN IF NOT EXISTS parent_template_id UUID REFERENCES work_order_templates(id) ON DELETE SET NULL,
    ADD COLUMN IF NOT EXISTS copied_from_template_id UUID REFERENCES work_order_templates(id) ON DELETE SET NULL,
    ADD COLUMN IF NOT EXISTS created_by UUID REFERENCES auth.users(id),
    ADD COLUMN IF NOT EXISTS updated_by UUID REFERENCES auth.users(id);

-- Make organization_id nullable (for app-level templates)
ALTER TABLE work_order_templates
    ALTER COLUMN organization_id DROP NOT NULL;

-- Add constraints
ALTER TABLE work_order_templates
    ADD CONSTRAINT valid_app_scope CHECK (
        (scope = 'app' AND organization_id IS NULL AND project_id IS NULL)
        OR scope != 'app'
    ),
    ADD CONSTRAINT valid_company_scope CHECK (
        (scope = 'company' AND organization_id IS NOT NULL AND project_id IS NULL)
        OR scope != 'company'
    ),
    ADD CONSTRAINT valid_project_scope CHECK (
        (scope = 'project' AND organization_id IS NOT NULL AND project_id IS NOT NULL)
        OR scope != 'project'
    );

-- Create indexes
CREATE INDEX idx_templates_scope ON work_order_templates(scope);
CREATE INDEX idx_templates_project ON work_order_templates(project_id);
CREATE INDEX idx_templates_parent ON work_order_templates(parent_template_id);
CREATE INDEX idx_templates_copied_from ON work_order_templates(copied_from_template_id);

-- ============================================================================
-- 3. UPDATE UNITS TABLE
-- ============================================================================

ALTER TABLE units
    ADD COLUMN IF NOT EXISTS project_id UUID REFERENCES wo_projects(id) ON DELETE SET NULL;

CREATE INDEX idx_units_project ON units(project_id);

-- 1. CREATE TEMPLATE LIBRARIES TABLE
-- ============================================================================

CREATE TABLE IF NOT EXISTS template_libraries (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name TEXT NOT NULL,
    description TEXT,
    project_type TEXT NOT NULL, -- 'residential_house', 'commercial_apartment', 'office', 'retail', etc.
    scope TEXT NOT NULL DEFAULT 'app' CHECK (scope IN ('app', 'company')),
    organization_id UUID REFERENCES organizations(id) ON DELETE CASCADE,
    
    -- Visual/metadata
    icon TEXT, -- Icon identifier (e.g., 'home', 'building', 'office')
    color TEXT, -- Hex color for UI display
    metadata JSONB DEFAULT '{}',
    
    -- Status
    is_active BOOLEAN NOT NULL DEFAULT true,
    is_default BOOLEAN NOT NULL DEFAULT false, -- One default library per project_type at app level
    
    -- Audit
    created_at TIMESTAMPTZ DEFAULT now(),
    updated_at TIMESTAMPTZ DEFAULT now(),
    
    -- Constraints
    CONSTRAINT valid_app_library CHECK (
        (scope = 'app' AND organization_id IS NULL)
        OR scope != 'app'
    ),
    CONSTRAINT valid_company_library CHECK (
        (scope = 'company' AND organization_id IS NOT NULL)
        OR scope != 'company'
    ),
    CONSTRAINT unique_app_library_name UNIQUE (name, scope, organization_id)
);

CREATE INDEX idx_template_libraries_scope ON template_libraries(scope);
CREATE INDEX idx_template_libraries_org ON template_libraries(organization_id);
CREATE INDEX idx_template_libraries_type ON template_libraries(project_type);
