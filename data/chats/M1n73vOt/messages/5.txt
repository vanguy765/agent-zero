import { createSignal, onCleanup } from 'solid-js';

export interface Route {
  path: string;
  params: Record<string, string>;
}

function parseHash(): Route {
  const hash = window.location.hash.slice(1) || '/dashboard';
  const parts = hash.split('/');

  // /table/:tableName
  if (parts[1] === 'table' && parts[2]) {
    return { path: '/table/:tableName', params: { tableName: decodeURIComponent(parts[2]) } };
  }

  // /tenant/:id
  if (parts[1] === 'tenant' && parts[2]) {
    return { path: '/tenant/:id', params: { id: decodeURIComponent(parts[2]) } };
  }

  return { path: hash, params: {} };
}

export function createRouter() {
  const [route, setRoute] = createSignal<Route>(parseHash());

  const handler = () => setRoute(parseHash());
  window.addEventListener('hashchange', handler);
  onCleanup(() => window.removeEventListener('hashchange', handler));

  const navigate = (path: string) => {
    window.location.hash = path;
  };

  return { route, navigate };
}
