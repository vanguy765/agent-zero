# reset-and-start.ps1
# Stops servers, resets DB, seeds users, starts dev servers
# ============================================================

Write-Host "`n=== Step 1: Killing dev servers ===" -ForegroundColor Cyan
& "$PSScriptRoot\kill-dev-servers.ps1"

Write-Host "`n=== Step 2: Resetting database ===" -ForegroundColor Cyan
Push-Location "$PSScriptRoot\.."
supabase db reset
Pop-Location

Write-Host "`n=== Step 3: Seeding users (alice + eric) ===" -ForegroundColor Cyan
Push-Location "$PSScriptRoot\..\apps\api"
npx tsx scripts/create_user.ts
Pop-Location

Write-Host "`n=== Step 4: Creating sysadmin user ===" -ForegroundColor Cyan
$sysadminBody = @{
    email         = "sysadmin@example.com"
    password      = "sysadmin123"
    email_confirm = $true
    user_metadata = @{
        role      = "sysadmin"
        full_name = "System Administrator"
    }
} | ConvertTo-Json -Depth 3

$serviceRoleKey = "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZS1kZW1vIiwicm9sZSI6InNlcnZpY2Vfcm9sZSIsImV4cCI6MTk4MzgxMjk5Nn0.EGIM96RAZx35lJzdJsyH-qQwv8Hdp7fsn3W0YpN81IU"
try {
    $response = Invoke-RestMethod -Uri "http://127.0.0.1:54321/auth/v1/admin/users" `
        -Method Post `
        -Headers @{
        "apikey"        = $serviceRoleKey
        "Authorization" = "Bearer $serviceRoleKey"
        "Content-Type"  = "application/json"
    } `
        -Body $sysadminBody
    Write-Host "  Sysadmin created: $($response.email)" -ForegroundColor Green
}
catch {
    Write-Host "  Sysadmin creation skipped (may already exist): $($_.Exception.Message)" -ForegroundColor Yellow
}

Write-Host "`n=== Step 5: Starting dev servers ===" -ForegroundColor Cyan
Push-Location "$PSScriptRoot\.."
pnpm dev
Pop-Location
