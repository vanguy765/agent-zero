(venv) root@d2d162d983f0:/a0/usr/projects/timesheet_app# # Kill All Dev Servers
# Kills any Node.js processes listening on monorepo app ports (3000-3005)
# Safe to run anytime before `pnpm dev`

$ports = @(3000, 3001, 3002, 3003, 3004, 3005)
$killed = 0

foreach ($port in $ports) {
    $connections = Get-NetTCPConnection -LocalPort $port -State Listen -ErrorAction SilentlyContinue
    foreach ($conn in $connections) {
        $proc = Get-Process -Id $conn.OwningProcess -ErrorAction SilentlyContinue
        if ($proc) {
            Write-Host "Killing $($proc.ProcessName) (PID $($proc.Id)) on port $port"
            Stop-Process -Id $proc.Id -Force
            $killed++
        }
    }
}

if ($killed -eq 0) {
    Write-Host "No processes found on ports 3000-3005. All clear!"
} else {
    Write-Host "`nKilled $killed process(es). Ready to start fresh."
}
