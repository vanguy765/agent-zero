// ─── Sorting helpers ────────────────────────────────────────────────────────
type SortDir = 'asc' | 'desc';

function useSortable<T>(initial: keyof T) {
    const [sortKey, setSortKey] = createSignal<keyof T>(initial);
    const [sortDir, setSortDir] = createSignal<SortDir>('asc');

    const toggle = (key: keyof T) => {
        if (sortKey() === key) {
            setSortDir(d => d === 'asc' ? 'desc' : 'asc');
        } else {
            setSortKey(() => key);
            setSortDir('asc');
        }
    };

    const sorted = (items: T[]) => {
        const k = sortKey();
        const dir = sortDir();
        return [...items].sort((a, b) => {
            const av = a[k] as any;
            const bv = b[k] as any;
            if (av == null && bv == null) return 0;
            if (av == null) return 1;
            if (bv == null) return -1;
            const cmp = typeof av === 'string'
                ? av.localeCompare(bv)
                : av < bv ? -1 : av > bv ? 1 : 0;
            return dir === 'asc' ? cmp : -cmp;
        });
    };

    const icon = (key: keyof T) => {
        if (sortKey() !== key) return ' ↕';
        return sortDir() === 'asc' ? ' ↑' : ' ↓';
    };

    return { sortKey, sortDir, toggle, sorted, icon };
}

// ─── Status badge helper ─────────────────────────────────────────────────────
const STATUS_CLASSES: Record<string, string> = {
    active:      'badge badge-success',
    completed:   'badge badge-primary',
    staged:      'badge badge-secondary',
    paused:      'badge badge-warning',
    archived:    'badge badge-secondary',
    draft:       'badge badge-secondary',
    inactive:    'badge badge-secondary',
    frozen:      'badge badge-primary',
};

const statusBadge = (status: string) => (
    <span class={STATUS_CLASSES[status] ?? 'badge badge-secondary'}>
        {status.charAt(0).toUpperCase() + status.slice(1)}
    </span>
);

const SortTh: Component<{ label: string; icon: string; onClick: () => void }> = (props) => (
    <th
        class="px-4 py-3 text-left text-xs font-semibold text-gray-500 uppercase tracking-wider cursor-pointer select-none hover:bg-gray-100 whitespace-nowrap"
        onClick={props.onClick}
    >
        {props.label}{props.icon}
    </th>
);

// ─── Collapse toggle icon ────────────────────────────────────────────────────
const CollapseIcon: Component<{ open: boolean }> = (props) => (
    <svg
        class="w-5 h-5 text-gray-400 transition-transform duration-200"
        style={{ transform: props.open ? 'rotate(0deg)' : 'rotate(-90deg)' }}
        fill="none" stroke="currentColor" stroke-width="2"
        viewBox="0 0 24 24"
    >
        <path stroke-linecap="round" stroke-linejoin="round" d="M19 9l-7 7-7-7" />
    </svg>
);

// ─── Section wrapper with collapse ──────────────────────────────────────────
const Section: Component<{
    title: string;
    count?: number;
    linkHref: string;
    linkLabel: string;
    children: any;
}> = (props) => {
    const [open, setOpen] = createSignal(true);
    return (
        <section>
            <div class="flex items-center justify-between mb-3">
                <button
                    class="flex items-center gap-2 group focus:outline-none"
                    onClick={() => setOpen(o => !o)}
                    aria-expanded={open()}
                >
                    <CollapseIcon open={open()} />
                    <h2 class="text-xl font-semibold text-gray-800 group-hover:text-primary-700 transition-colors">
                        {props.title}
                    </h2>
                    <Show when={props.count !== undefined}>
                        <span class="text-sm text-gray-400 font-normal">({props.count})</span>
                    </Show>
                </button>
                <A href={props.linkHref} class="text-sm text-primary-600 hover:underline">{props.linkLabel}</A>
            </div>
            <Show when={open()}>
                {props.children}
            </Show>
        </section>
    );
};

    const [stats]      = createResource<DashboardStats>(() => api.getDashboardStats());

    // Highlighted archetype row (for scroll-to-highlight)
    const [highlightedArchetype, setHighlightedArchetype] = createSignal<string | null>(null);

    const scrollToArchetype = (id: string) => {
        setHighlightedArchetype(id);
        const el = document.getElementById(`archetype-row-${id}`);
        if (el) {
            el.scrollIntoView({ behavior: 'smooth', block: 'center' });
        }
        // Clear highlight after 2.5 s
        setTimeout(() => setHighlightedArchetype(null), 2500);
    };

    const templateRows = (): TemplateRow[] => {
        if (!templates() || !stats()) return [];
        return (templates() ?? []).map(t => ({
            id:         t.id,
            trade_name: t.trade_name,
            version:    t.version,
            status:     t.is_frozen ? 'frozen' : (t.is_active ? 'active' : 'inactive'),
            scope:      t.scope,
            wo_count:   stats()?.template_wo_counts?.[t.id] ?? 0,
            is_frozen:  t.is_frozen ?? false,
        }));
    };

    const projectRows = (): ProjectRow[] => {
        if (!projects()) return [];
        return (projects() ?? []).map(p => {
            const us        = stats()?.project_unit_stats?.[p.id];
            const total     = us?.total ?? 0;
            const completed = us?.completed ?? 0;
            const inProg    = us?.in_progress ?? 0;
            return {
                id:                p.id,
                name:              p.name,
                project_type:      ptLabel(p.project_type),
                status:            p.status,
                units_total:       total,
                units_completed:   completed,
                units_in_progress: inProg,
                pct_complete:      total > 0 ? Math.round((completed / total) * 100) : 0,
            };
        });
    };

    const archetypeRows = (): ArchetypeRow[] => {
        if (!typeOrders()) return [];
        return (typeOrders() ?? []).map(to => {
            const comps = stats()?.type_order_compositions?.[to.id] ?? [];
            const composition = comps.length > 0
                ? comps.map(c => c.quantity > 1 ? `${c.quantity}× ${c.name}` : c.name).join(', ')
                : (to.is_composite ? '(no components)' : '—');
            // Use static description map first, fall back to DB description
            const desc = ARCHETYPE_DESCRIPTIONS[to.code] ?? to.description ?? '—';
            return {
                id:           to.id,
                name:         to.name,
                code:         to.code,
                category:     CAT_LABELS[to.category] ?? to.category,
                description:  desc,
                composition,
                is_composite: to.is_composite,
            };
        });
    };

    const tSort = useSortable<TemplateRow>('trade_name');
    const pSort = useSortable<ProjectRow>('name');
    const aSort = useSortable<ArchetypeRow>('name');

        <div class="space-y-10">
            <div>
                <h1 class="text-3xl font-bold text-gray-900">Overview</h1>
                <p class="mt-1 text-gray-500">Projects, company WO templates, and space archetypes at a glance.</p>
            </div>

            <Show when={anyLoading()}>
                <div class="flex items-center gap-2 text-gray-500">
                    <div class="animate-spin rounded-full h-5 w-5 border-b-2 border-primary-600" />
                    Loading dashboard data…
                </div>
            </Show>

            <Show when={anyError()}>
                <ErrorAlert message={`Failed to load dashboard data: ${(anyError() as any)?.message ?? 'API server may be offline.'}`} />
            </Show>

            {/* ── 1. Projects ─────────────────────────────────────────────── */}
            <Section title="Projects" count={projectRows().length || undefined} linkHref="/projects" linkLabel="View all →">
                <div class="card overflow-hidden p-0">
                    <div class="overflow-x-auto">
                        <table class="min-w-full divide-y divide-gray-200">
                            <thead class="bg-gray-50">
                                <tr>
                                    <SortTh label="Project" icon={pSort.icon('name')} onClick={() => pSort.toggle('name')} />
                                    <SortTh label="Type" icon={pSort.icon('project_type')} onClick={() => pSort.toggle('project_type')} />
                                    <SortTh label="Status" icon={pSort.icon('status')} onClick={() => pSort.toggle('status')} />
                                    <SortTh label="Units" icon={pSort.icon('units_total')} onClick={() => pSort.toggle('units_total')} />
                                    <SortTh label="% Complete" icon={pSort.icon('pct_complete')} onClick={() => pSort.toggle('pct_complete')} />
                                    <th class="px-4 py-3" />
                                </tr>
                            </thead>
                            <tbody class="bg-white divide-y divide-gray-100">
                                <Show when={!projects.loading} fallback={<tr><td colspan="6" class="px-4 py-6 text-center text-gray-400">Loading…</td></tr>}>
                                    <Show when={!projects.error} fallback={<tr><td colspan="6" class="px-4 py-6"><ErrorAlert message="Could not load projects." /></td></tr>}>
                                        <Show when={projectRows().length > 0} fallback={<tr><td colspan="6" class="px-4 py-6 text-center text-gray-400">No projects found.</td></tr>}>
                                            <For each={pSort.sorted(projectRows())}>{(row) => (

                                                            title={`Highlight ${row.name}`}
                                                        >
                                                            Details ↓
                                                        </button>
                                                    </td>
                                                </tr>
                                            )}</For>
                                        </Show>
                                    </Show>
                                </Show>
                            </tbody>
                        </table>
                    </div>
                </div>
            </Section>
        </div>
    );
};
