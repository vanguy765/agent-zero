workOrdersRoute.get('/dashboard-stats', async (c) => {
    const supabase = getScopedClient(c);

    // 1. Work order counts per template
    const { data: woData, error: woError } = await supabase
        .from('work_orders')
        .select('template_id, status');

    if (woError) return c.json({ error: woError.message }, 500);

    const templateWoCounts: Record<string, number> = {};
    const templateWoStatusCounts: Record<string, Record<string, number>> = {};
    for (const wo of (woData || [])) {
        if (!wo.template_id) continue;
        templateWoCounts[wo.template_id] = (templateWoCounts[wo.template_id] || 0) + 1;
        if (!templateWoStatusCounts[wo.template_id]) templateWoStatusCounts[wo.template_id] = {};
        templateWoStatusCounts[wo.template_id][wo.status] = (templateWoStatusCounts[wo.template_id][wo.status] || 0) + 1;
    }

    // 2. Unit stats per project
    const { data: unitData, error: unitError } = await supabase
        .from('units')
        .select('project_id, status');

    if (unitError) return c.json({ error: unitError.message }, 500);

    const projectUnitStats: Record<string, { total: number; not_started: number; in_progress: number; completed: number }> = {};
    for (const unit of (unitData || [])) {
        if (!unit.project_id) continue;
        if (!projectUnitStats[unit.project_id]) {
            projectUnitStats[unit.project_id] = { total: 0, not_started: 0, in_progress: 0, completed: 0 };
        }
        projectUnitStats[unit.project_id].total++;
        const s = unit.status as string;
        if (s === 'not_started') projectUnitStats[unit.project_id].not_started++;
        else if (s === 'in_progress') projectUnitStats[unit.project_id].in_progress++;
        else if (s === 'completed') projectUnitStats[unit.project_id].completed++;
    }

    // 3. Type order compositions (bulk)
    const { data: compData, error: compError } = await supabase
        .from('type_order_compositions')
        .select('composite_type_order_id, quantity, display_order, component:type_orders!component_type_order_id(id, name, code)')
        .order('display_order', { ascending: true });

    if (compError) return c.json({ error: compError.message }, 500);

    const typeOrderCompositions: Record<string, Array<{ id: string; name: string; code: string; quantity: number }>> = {};
    for (const comp of (compData || [])) {
        const parentId = comp.composite_type_order_id;
        if (!typeOrderCompositions[parentId]) typeOrderCompositions[parentId] = [];
        const component = comp.component as any;
        if (component) {
            typeOrderCompositions[parentId].push({
                id: component.id,
                name: component.name,
                code: component.code,
                quantity: comp.quantity,
            });
        }
    }

    return c.json({
        template_wo_counts: templateWoCounts,
        template_wo_status_counts: templateWoStatusCounts,
        project_unit_stats: projectUnitStats,
        type_order_compositions: typeOrderCompositions,
    });
});

// ─── Sorting helpers ────────────────────────────────────────────────────────
type SortDir = 'asc' | 'desc';

function useSortable<T>(initial: keyof T) {
    const [sortKey, setSortKey] = createSignal<keyof T>(initial);
    const [sortDir, setSortDir] = createSignal<SortDir>('asc');

    const toggle = (key: keyof T) => {
        if (sortKey() === key) {
            setSortDir(d => d === 'asc' ? 'desc' : 'asc');
        } else {
            setSortKey(() => key);
            setSortDir('asc');
        }
    };

    const sorted = (items: T[]) => {
        const k = sortKey();
        const dir = sortDir();
        return [...items].sort((a, b) => {
            const av = a[k] as any;
            const bv = b[k] as any;
            if (av == null && bv == null) return 0;
            if (av == null) return 1;
            if (bv == null) return -1;
            const cmp = typeof av === 'string'
                ? av.localeCompare(bv)
                : av < bv ? -1 : av > bv ? 1 : 0;
            return dir === 'asc' ? cmp : -cmp;
        });
    };

    const icon = (key: keyof T) => {
        if (sortKey() !== key) return ' \u21d5';
        return sortDir() === 'asc' ? ' \u2191' : ' \u2193';
    };

    return { sortKey, sortDir, toggle, sorted, icon };
}

                                                        <A href={`/projects/${row.id}`} class="hover:text-primary-600 hover:underline">{row.name}</A>
                                                    </td>
                                                    <td class="px-4 py-3 text-gray-600 text-sm">{row.project_type}</td>
                                                    <td class="px-4 py-3">{statusBadge(row.status)}</td>
                                                    <td class="px-4 py-3 text-gray-700 text-sm">
                                                        <Show when={row.units_total > 0} fallback={<span class="text-gray-400">\u2014</span>}>
                                                            <span class="font-semibold text-green-600">{row.units_completed}</span>
                                                            <span class="text-gray-400"> / {row.units_total}</span>
                                                            <Show when={row.units_in_progress > 0}>
                                                                <span class="text-yellow-500 text-xs ml-1">({row.units_in_progress} active)</span>
                                                            </Show>
                                                        </Show>
                                                    </td>
                                                    <td class="px-4 py-3">
                                                        <Show when={row.units_total > 0} fallback={<span class="text-gray-400 text-sm">\u2014</span>}>
                                                            <div class="flex items-center gap-2">
                                                                <div class="flex-1 bg-gray-200 rounded-full h-2 min-w-16">
                                                                    <div class="bg-primary-600 h-2 rounded-full transition-all" style={{ width: `${row.pct_complete}%` }} />
                                                                </div>
                                                                <span class="text-xs font-medium text-gray-700 w-9 text-right">{row.pct_complete}%</span>
                                                            </div>
                                                        </Show>
                                                    </td>
                                                    <td class="px-4 py-3 text-right">
                                                        <A href={`/projects/${row.id}`} class="text-xs text-primary-600 hover:underline">Details \u2192</A>
                                                    </td>
                                                </tr>
                                            )}</For>
                                        </Show>
                                    </Show>
                                </Show>
                            </tbody>
                        </table>
                    </div>
                </div>
            </Section>

            {/* ── 2. Company WO Templates ──────────────────────────────────── */}
            <Section title="Company WO Templates" count={templateRows().length || undefined} linkHref="/templates/company" linkLabel="View all \u2192">
                <div class="card overflow-hidden p-0">
                    <div class="overflow-x-auto">
                        <table class="min-w-full divide-y divide-gray-200">
                            <thead class="bg-gray-50">
                                <tr>
                                    <SortTh label="Trade" icon={tSort.icon('trade_name')} onClick={() => tSort.toggle('trade_name')} />
                                    <SortTh label="Version" icon={tSort.icon('version')} onClick={() => tSort.toggle('version')} />
                                    <SortTh label="Status" icon={tSort.icon('status')} onClick={() => tSort.toggle('status')} />
                                    <SortTh label="Work Orders" icon={tSort.icon('wo_count')} onClick={() => tSort.toggle('wo_count')} />
                                    <th class="px-4 py-3" />
                                </tr>
                            </thead>
                            <tbody class="bg-white divide-y divide-gray-100">
                                <Show when={!templates.loading} fallback={<tr><td colspan="5" class="px-4 py-6 text-center text-gray-400">Loading\u2026</td></tr>}>
                                    <Show when={!templates.error} fallback={<tr><td colspan="5" class="px-4 py-6"><ErrorAlert message="Could not load templates." /></td></tr>}>
                                        <Show when={templateRows().length > 0} fallback={<tr><td colspan="5" class="px-4 py-6 text-center text-gray-400">No company WO templates found.</td></tr>}>
                                            <For each={tSort.sorted(templateRows())}>{(row) => (
                                                <tr class="hover:bg-gray-50 transition-colors">
                                                    <td class="px-4 py-3 font-medium text-gray-900">
                                                        <A href={`/templates/${row.id}`} class="hover:text-primary-600 hover:underline">{row.trade_name}</A>
                                                    </td>
                                                    <td class="px-4 py-3 text-gray-600">v{row.version}</td>
                                                    <td class="px-4 py-3">{statusBadge(row.status)}</td>
                                                    <td class="px-4 py-3 text-gray-700">
                                                        <span class="font-semibold">{row.wo_count}</span>
                                                        <span class="text-gray-400 text-xs ml-1">WOs</span>
                                                    </td>
                                                    <td class="px-4 py-3 text-right">
                                                        <A href={`/templates/${row.id}`} class="text-xs text-primary-600 hover:underline">Details \u2192</A>
                                                    </td>
                                                </tr>
                                            )}</For>
                                        </Show>
                                    </Show>
                                </Show>
                            </tbody>
                        </table>
                    </div>
                </div>
            </Section>

            {/* ── 3. Space Archetypes ──────────────────────────────────────── */}
            <Section title="Space Archetypes" count={archetypeRows().length || undefined} linkHref="/settings/type-orders" linkLabel="Manage \u2192">
                <div class="card overflow-hidden p-0">
                    <div class="overflow-x-auto">
                        <table class="min-w-full divide-y divide-gray-200">
                            <thead class="bg-gray-50">
                                <tr>
                                    <SortTh label="Name" icon={aSort.icon('name')} onClick={() => aSort.toggle('name')} />
                                    <SortTh label="Code" icon={aSort.icon('code')} onClick={() => aSort.toggle('code')} />
                                    <SortTh label="Category" icon={aSort.icon('category')} onClick={() => aSort.toggle('category')} />
                                    <SortTh label="Size" icon={aSort.icon('size')} onClick={() => aSort.toggle('size')} />
                                    <th class="px-4 py-3 text-left text-xs font-semibold text-gray-500 uppercase tracking-wider">Description</th>
                                    <th class="px-4 py-3 text-left text-xs font-semibold text-gray-500 uppercase tracking-wider">Composition</th>
                                    <th class="px-4 py-3" />
                                </tr>
                            </thead>
                            <tbody class="bg-white divide-y divide-gray-100">
                                <Show when={!typeOrders.loading} fallback={<tr><td colspan="7" class="px-4 py-6 text-center text-gray-400">Loading\u2026</td></tr>}>
                                    <Show when={!typeOrders.error} fallback={<tr><td colspan="7" class="px-4 py-6"><ErrorAlert message="Could not load space archetypes." /></td></tr>}>
                                        <Show when={archetypeRows().length > 0} fallback={<tr><td colspan="7" class="px-4 py-6 text-center text-gray-400">No space archetypes found.</td></tr>}>
                                            <For each={aSort.sorted(archetypeRows())}>{(row) => (
                                                <>
                                                    <tr
                                                        id={`archetype-row-${row.id}`}
                                                        class="hover:bg-gray-50 transition-colors cursor-pointer"
                                                        onClick={() => toggleArchetype(row.id)}
                                                    >
                                                        <td class="px-4 py-3 font-medium text-gray-900">
                                                            <div class="flex items-center gap-1.5">
                                                                <svg
                                                                    class="w-3.5 h-3.5 text-gray-400 shrink-0 transition-transform duration-150"
                                                                    style={{ transform: expandedArchetype() === row.id ? 'rotate(90deg)' : 'rotate(0deg)' }}
                                                                    fill="none" stroke="currentColor" stroke-width="2.5" viewBox="0 0 24 24"
                                                                >
                                                                    <path stroke-linecap="round" stroke-linejoin="round" d="M9 5l7 7-7 7" />
                                                                </svg>
                                                                {row.name}
                                                            </div>
                                                        </td>
                                                        <td class="px-4 py-3">
                                                            <code class="text-xs bg-gray-100 px-1.5 py-0.5 rounded text-gray-700">{row.code}</code>
                                                        </td>
                                                        <td class="px-4 py-3">
                                                            <span class="badge badge-secondary text-xs">{row.category}</span>
                                                        </td>
                                                        <td class="px-4 py-3 text-gray-600 text-sm whitespace-nowrap">{row.size}</td>
                                                        <td class="px-4 py-3 text-gray-600 text-xs max-w-xs">
                                                            <span class="line-clamp-2 leading-relaxed">{row.brief}</span>
                                                        </td>
                                                        <td class="px-4 py-3 text-gray-500 text-sm max-w-xs">
                                                            <Show when={row.is_composite} fallback={<span class="text-gray-400 italic text-xs">Atomic</span>}>
                                                                <span class="text-xs">{row.composition}</span>
                                                            </Show>
                                                        </td>
                                                        <td class="px-4 py-3 text-right">
                                                            <span class="text-xs text-primary-600">
                                                                {expandedArchetype() === row.id ? 'Hide \u2191' : 'Details \u2193'}
                                                            </span>
                                                        </td>
                                                    </tr>
                                                    <Show when={expandedArchetype() === row.id}>
                                                        <tr class="bg-blue-50 border-l-4 border-primary-400">
                                                            <td colspan="7" class="px-6 py-4">
                                                                <div class="flex gap-6">
                                                                    <div class="flex-1">
                                                                        <p class="text-sm font-semibold text-gray-700 mb-1">{row.name}</p>
                                                                        <p class="text-sm text-gray-600 leading-relaxed">{row.detail}</p>
                                                                    </div>
                                                                    <div class="shrink-0 text-right space-y-1">
                                                                        <div class="text-xs text-gray-500">Size</div>
                                                                        <div class="text-sm font-medium text-gray-700">{row.size}</div>
                                                                        <Show when={row.is_composite}>
                                                                            <div class="text-xs text-gray-500 mt-2">Components</div>
                                                                            <div class="text-xs text-gray-600">{row.composition}</div>
                                                                        </Show>
                                                                    </div>
                                                                </div>
                                                            </td>
                                                        </tr>
                                                    </Show>
                                                </>
                                            )}</For>
                                        </Show>
                                    </Show>
                                </Show>
                            </tbody>
                        </table>
                    </div>
                </div>
            </Section>
        </div>
    );
};
