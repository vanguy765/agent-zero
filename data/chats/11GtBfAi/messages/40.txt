CREATE TABLE IF NOT EXISTS public.work_order_step_progress (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    work_order_id UUID NOT NULL REFERENCES public.work_orders(id) ON DELETE CASCADE,
    sop_step_id UUID NOT NULL REFERENCES public.sop_steps(id) ON DELETE RESTRICT,
    status TEXT NOT NULL DEFAULT 'pending' CHECK (status IN (
        'pending', 'in_progress', 'completed', 'skipped', 'blocked'
    )),
    completed_by UUID REFERENCES auth.users(id),
    started_at TIMESTAMPTZ,
    completed_at TIMESTAMPTZ,
    duration_minutes INTEGER,
    notes TEXT,
    photo_urls TEXT[] DEFAULT '{}',
    override_detail TEXT,
    is_override BOOLEAN DEFAULT false,
    organization_id UUID REFERENCES public.organizations(id) ON DELETE CASCADE,
    created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT now(),
    UNIQUE(work_order_id, sop_step_id)
);

COMMENT ON TABLE public.work_order_step_progress IS 'Step-level completion tracking for unit work orders. Duration feeds timesheet app.';

import { createSignal, createResource, For, Show, createMemo } from 'solid-js';
import { supabase } from '../lib/supabase';
import { useParams, A } from '@solidjs/router';
import type { Unit, UnitScopeItem } from '@repo/schemas';

const fetchUnit = async (id: string) => {
    const { data, error } = await supabase.from('units').select('*').eq('id', id).single();
    if (error) throw error;
    return data as Unit;
};

const fetchScope = async (id: string) => {
    const { data, error } = await supabase
        .from('unit_scope_items')
        .select('*')
        .eq('unit_id', id)
        .order('zone', { ascending: true });

    if (error) throw error;
    return data as UnitScopeItem[];
};

export default function UnitDetail() {
    const params = useParams();
    const [unit] = createResource(() => params.id, fetchUnit);
    const [scope, { mutate }] = createResource(() => params.id, fetchScope);
    const [selectedTrade, setSelectedTrade] = createSignal<string>('All');

    // Derived trades list
    const trades = createMemo(() => {
        const s = scope();
        if (!s) return ['All'];
        const t = new Set(s.map(i => i.trade || 'General'));
        return ['All', ...Array.from(t).sort()];
    });

    // Filtered items
    const filteredItems = createMemo(() => {
        const s = scope();
        const trade = selectedTrade();
        if (!s) return [];
        if (trade === 'All') return s;
        return s.filter(i => (i.trade || 'General') === trade);
    });

    const updateStatus = async (itemId: string, newStatus: string) => {
        // Optimistic update
        mutate((prev) => prev?.map(i => i.id === itemId ? { ...i, status: newStatus as any } : i));

        const { error } = await supabase
            .from('unit_scope_items')
            .update({ status: newStatus })
            .eq('id', itemId);

        if (error) {
            console.error('Failed to update status', error);
            // Revert on error could be implemented here
        }
    };

    const StatusBadge = (props: { status: string }) => {
        const colors = {
            pending: 'bg-gray-100 text-gray-800',
            in_progress: 'bg-blue-100 text-blue-800',
            completed: 'bg-green-100 text-green-800',
            verified: 'bg-purple-100 text-purple-800'
        };
        const color = colors[props.status as keyof typeof colors] || colors.pending;
        return <span class={`px-2 py-1 rounded text-xs font-medium uppercase ${color}`}>{props.status.replace('_', ' ')}</span>;
    };

    return (
        <div class="p-4 max-w-4xl mx-auto">
            <A href="/units" class="text-blue-600 mb-4 inline-block">← Back to Units</A>

            <Show when={unit()}>
                <div class="mb-6 border-b pb-4">
                    <h1 class="text-3xl font-bold">Unit {unit()?.unit_number}</h1>
                    <p class="text-gray-600">Type: {unit()?.unit_type}</p>
                    <Show when={unit()?.floor_plan_image}>
                        <div class="mt-4">
                            <h2 class="text-lg font-semibold mb-2">Floor Plan</h2>
                            <img
                                src={unit()?.floor_plan_image!}
                                alt={`Floor Plan for Unit ${unit()?.unit_number}`}
                                class="max-w-full h-auto rounded border shadow-sm max-h-[400px]"
                            />
                        </div>
                    </Show>
                </div>
            </Show>

            {/* Controls */}
            <div class="flex flex-wrap gap-4 mb-6 items-center bg-gray-50 p-3 rounded">
                <label class="font-semibold">Filter by Trade:</label>
                <select
                    class="border p-2 rounded"
                    value={selectedTrade()}
                    onInput={(e) => setSelectedTrade(e.currentTarget.value)}
                >
                    <For each={trades()}>
                        {(trade) => <option value={trade}>{trade}</option>}
                    </For>
                </select>
            </div>

            {/* Scope Table */}
            <div class="space-y-4">
                <For each={filteredItems()} fallback={<div class="text-gray-500 italic">No items found.</div>}>
                    {(item) => (
                        <div class="border rounded p-3 flex flex-col md:flex-row justify-between items-start md:items-center gap-4 bg-white shadow-sm">
                            <div class="flex-1">
                                <span class="text-xs font-bold text-gray-500 uppercase tracking-wide">{item.zone} • {item.trade}</span>
                                <h3 class="font-semibold text-lg">{item.item}</h3>
                                <p class="text-gray-700">{item.action}</p>
                            </div>

                            <div class="flex items-center gap-2">
                                <select
                                    class="border p-1 rounded text-sm bg-white"
                                    value={item.status}
                                    onChange={(e) => updateStatus(item.id, e.currentTarget.value)}
                                >
                                    <option value="pending">Pending</option>
                                    <option value="in_progress">In Progress</option>
                                    <option value="completed">Completed</option>
                                    <option value="verified">Verified</option>
                                </select>
                                <StatusBadge status={item.status} />
                            </div>
                        </div>
                    )}
                </For>
            </div>
        </div>
    );
}

## Executive Summary

This prompt implements the bridge between the existing SOP library system and the existing Work Order (WO) template inheritance hierarchy. The goal is to let each WO template reference an SOP, expose SOP steps through the WO API, and track step-level completion/duration at the unit work order level — feeding a separate timesheet application.

The design principle is **surgical integration**: we add to existing files rather than creating parallel structures. The existing WO inheritance chain (system → company → project → unit) stays untouched. SOP linkage rides along automatically through the existing copy/freeze mechanisms.
