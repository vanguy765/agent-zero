import { OpenAPIHono } from '@hono/zod-openapi';
import { createClient } from '@supabase/supabase-js';
import { SUPABASE_URL, SUPABASE_ANON_KEY, SUPABASE_SERVICE_ROLE_KEY } from '@repo/config';

// ----------------------------------------------------------------------------
// REFERENCE IMPLEMENTATION: VOICE PROCESSING
// ----------------------------------------------------------------------------
// This file serves as a reference for how to handle:
// 1. File Uploads (Blob)
// 2. Mocking AI Latency
// 3. Database Insertion
// ----------------------------------------------------------------------------

export const voiceRoute = new OpenAPIHono();

// We need SERVICE ROLE to bypass RLS if the mobile app is unauthenticated (for the test).
// In production, you would use the User's JWT.
const supabaseAdmin = createClient(SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY);

voiceRoute.post('/', async (c) => {
    console.log('[Voice] Received Request');

    // 1. Parse Body (Multipart)
    // --------------------------------------------------------
    const body = await c.req.parseBody();
    const audioFile = body['audio'];
    // We expect a user_id to assign the task to, otherwise we can't notify the dashboard
    const targetUserId = body['user_id'] as string;

    if (!audioFile || !(audioFile instanceof Blob)) {
        return c.json({ error: 'No audio file provided' }, 400);
    }
    
    console.log(`[Voice] Processing Audio: ${audioFile.size} bytes`);

    // 2. Simulate AI Processing (The "Brain")
    // --------------------------------------------------------
    await new Promise(r => setTimeout(r, 1500)); // 1.5s latency

    // Mock Result
    const extractedTask = `Voice Entry at ${new Date().toLocaleTimeString()}`;
    console.log('[Voice] AI Extracted:', extractedTask);

    // 3. Database Interaction
    // --------------------------------------------------------
    // BROADCAST STRATEGY (Test Mode):
    // Insert this task for EVERY user in the system so that whoever is looking at the dashboard sees it.
    
    const { data: users, error: userError } = await supabaseAdmin.auth.admin.listUsers();
    
    if (userError || !users.users.length) {
        console.error('[Voice] No users found to assign task to.');
        return c.json({ message: "Voice processed but no users found." });
    }

    const tasksToInsert = users.users.map(u => ({
        task: extractedTask,
        is_completed: false,
        user_id: u.id
    }));

    const { error: insertError } = await supabaseAdmin
        .from('testrepo_todos')
        .insert(tasksToInsert);

    if (insertError) {
        console.error('[Voice] DB Error:', insertError);
        return c.json({ error: insertError.message }, 500);
    }
    
    console.log(`[Voice] Broadcasted task to ${tasksToInsert.length} users.`);
    
    return c.json({
        success: true,
        message: `Voice processed and broadcasted to ${tasksToInsert.length} users`,
        transcription: extractedTask
    });
});
