"""process_document.py - CLI client for the doc-parser API.

Submits a PDF to the doc-parser service, polls for completion,
and fetches the full results. Optionally saves results to JSON.

Usage:
    python process_document.py --pdf drawing.pdf
    python process_document.py --pdf drawing.pdf --engine hybrid --output results.json
    python process_document.py --pdf drawing.pdf \\
        --tenant acme --project site_a \\
        --supabase-url http://host.docker.internal:54321 \\
        --supabase-key eyJhbGci...
"""
import argparse
import json
import sys
import time
from pathlib import Path

import requests

DEFAULT_API_URL = "http://localhost:5000"
DEFAULT_POLL_INTERVAL = 5
MAX_POLL_ATTEMPTS = 120  # 10 minutes max


def submit_job(api_url, pdf_path, engine, provider,
               supabase_url, supabase_key, tenant_name, project_name):
    """Submit PDF to the parser and return job_id."""
    endpoint = api_url + "/parse/" + engine
    print("Submitting " + Path(pdf_path).name + " to " + endpoint + " ...")

    with open(pdf_path, "rb") as f:
        files = {"file": (Path(pdf_path).name, f, "application/pdf")}
        data = {"provider": provider}
        if supabase_url:
            data["supabase_url"] = supabase_url
        if supabase_key:
            data["supabase_key"] = supabase_key
        if tenant_name:
            data["tenant_name"] = tenant_name
        if project_name:
            data["project_name"] = project_name

        resp = requests.post(endpoint, files=files, data=data, timeout=30)

    resp.raise_for_status()
    payload = resp.json()
    job_id = payload["job_id"]
    print("Job queued: " + job_id)
    return job_id
