/**
 * Doc-Parser Proxy Routes for Hono API
 *
 * Integrates doc-parser service into the existing work-orders API.
 * Provides engine selection, status polling, and result retrieval.
 *
 * Routes:
 *   POST /api/doc-parser/parse           - Parse document (engine selection)
 *   GET  /api/doc-parser/status/:jobId  - Lightweight status (poll-safe)
 *   GET  /api/doc-parser/results/:jobId - Full results (fetch-once)
 *   GET  /api/doc-parser/health         - Service health check
 */

import { Hono } from 'hono';
import { getScopedClient } from '../middleware/supabase';
import {
  submitParseJob,
  getJobStatus,
  getJobResults,
  checkHealth,
  mapToParseDocument,
  type DocParserStatus,
  type DocParserResult,
} from '../services/doc-parser-client';

const docParserRoutes = new Hono();

// ============================================================
// POST /parse
// Accepts JSON { document_id, engine }
// Fetches PDF from Supabase storage, forwards to doc-parser
// ============================================================

docParserRoutes.post('/parse', async (c) => {
  const body = await c.req.json().catch(() => ({}));
  const documentId = body?.document_id;
  const engine = (body?.engine || 'hybrid') as string;

  if (!documentId) {
    return c.json({ error: 'document_id is required' }, 400);
  }

  // ---- Claude Direct: existing flow (unchanged) ----
  if (engine === 'claude-direct') {
    return c.json({
      error: 'claude-direct engine should be handled by existing route handler',
      hint: 'This route only handles docling and hybrid engines',
    }, 400);
  }

  // ---- Docling or Hybrid: proxy to doc-parser service ----
  if (engine !== 'docling' && engine !== 'hybrid') {
    return c.json(
      { error: `Unknown engine: ${engine}. Use 'claude-direct', 'docling', or 'hybrid'.` },
      400,
    );
  }

  try {
    // Fetch document metadata from database
    const supabase = getScopedClient(c);
    const { data: document, error: fetchError } = await supabase
      .from('project_documents')
      .select('*')
      .eq('id', documentId)
      .single();

    if (fetchError || !document) {
      return c.json({ error: 'Document not found' }, 404);
    }

    if (!document.file_type?.includes('pdf')) {
      return c.json({ error: 'Document is not a PDF' }, 400);
    }

    // Download PDF from Supabase storage
    const { data: fileData, error: downloadError } = await supabase.storage
      .from('project-documents')
      .download(document.file_url);

    if (downloadError || !fileData) {
      console.error('[doc-parser] Storage download failed:', downloadError);
      return c.json({ error: 'Failed to download document from storage' }, 500);
    }

    const pdfBuffer = await fileData.arrayBuffer();

    // Submit to doc-parser service
    const jobResponse = await submitParseJob(
      pdfBuffer,
      document.title || `document-${documentId}.pdf`,
      engine as 'docling' | 'hybrid',
      {
        provider: body?.provider || process.env.DEFAULT_LLM_PROVIDER,
        tenantName: body?.tenant_name,
        projectName: body?.project_name || documentId,
        supabaseUrl: process.env.SUPABASE_URL,
        supabaseKey: process.env.SUPABASE_SERVICE_KEY,
      },
    );

    // Return job info with polling URLs
    return c.json({
      engine,
      job_id: jobResponse.job_id,
      status: jobResponse.status,
      poll_url: `/api/doc-parser/status/${jobResponse.job_id}`,
      results_url: `/api/doc-parser/results/${jobResponse.job_id}`,
    });
  } catch (error) {
    const message = error instanceof Error ? error.message : 'Unknown error';
    console.error(`[doc-parser] Parse submission failed: ${message}`);
    return c.json({ error: `Doc-parser submission failed: ${message}` }, 502);
  }
});

// ============================================================
// GET /api/doc-parser/status/:jobId
// Lightweight status for polling (Priority 4a)
// Safe to call every 2-5 seconds. Small payload.
// ============================================================

docParserRoutes.get('/status/:jobId', async (c) => {
  const jobId = c.req.param('jobId');

  try {
    const status: DocParserStatus = await getJobStatus(jobId);

    // Ensure result data is NEVER included in status response
    // (defense-in-depth - server should already exclude it)
    const safeStatus = { ...status };
    delete (safeStatus as any).result;

    return c.json(safeStatus);
  } catch (error) {
    const message = error instanceof Error ? error.message : 'Unknown error';
    if (message.includes('not found')) {
      return c.json({ error: `Job ${jobId} not found` }, 404);
    }
    return c.json({ error: `Status check failed: ${message}` }, 502);
  }
});

// ============================================================
// GET /api/doc-parser/results/:jobId
// Full results - fetch once after status confirms "completed"
// Heavy payload. Includes ParsedDocument-compatible mapping.
// ============================================================

docParserRoutes.get('/results/:jobId', async (c) => {
  const jobId = c.req.param('jobId');
  const summaryOnly = c.req.query('summary') === 'true';

  try {
    const data = await getJobResults(jobId, summaryOnly);

    // If summary requested, return as-is
    if (summaryOnly) {
      return c.json(data);
    }

    // If hybrid engine with units array, map to ParsedDocument format
    const result = data.result as DocParserResult | undefined;
    if (result?.engine === 'hybrid' && result?.units) {
      return c.json({
        ...data,
        parsed: mapToParseDocument(result, jobId),
      });
    }

    // Docling engine - also map units if available
    if (result?.units) {
      return c.json({
        ...data,
        parsed: mapToParseDocument(result, jobId),
      });
    }

    // Fallback: return raw result
    return c.json(data);
  } catch (error) {
    const message = error instanceof Error ? error.message : 'Unknown error';
    if (message.includes('not found')) {
      return c.json({ error: `Results for job ${jobId} not found` }, 404);
    }
    return c.json({ error: `Results fetch failed: ${message}` }, 502);
  }
});

// ============================================================
// GET /api/doc-parser/health
// Service health check with dependency status
// ============================================================

docParserRoutes.get('/health', async (c) => {
  const health = await checkHealth();
  const statusCode = health.status === 'healthy' ? 200 : 503;
  return c.json(health, statusCode);
});

// Maximum dimension for Claude multi-image API
const MAX_IMAGE_DIMENSION = 2000;

/**
 * Resize an image to fit within MAX_IMAGE_DIMENSION while maintaining aspect ratio
 */
async function resizeImageIfNeeded(
    base64Data: string,
    mediaType: 'image/png' | 'image/jpeg'
): Promise<{ data: string; mediaType: 'image/png' | 'image/jpeg'; resized: boolean }> {
    // Decode base64 to buffer
    const buffer = Buffer.from(base64Data, 'base64');

    // Load image
    const image = await canvas.loadImage(buffer);
    const width = image.width;
    const height = image.height;

    // Check if resizing is needed
    if (width <= MAX_IMAGE_DIMENSION && height <= MAX_IMAGE_DIMENSION) {
        return { data: base64Data, mediaType, resized: false };
    }

    // Calculate new dimensions maintaining aspect ratio
    let newWidth: number;
    let newHeight: number;

    if (width > height) {
        newWidth = MAX_IMAGE_DIMENSION;
        newHeight = Math.round((height / width) * MAX_IMAGE_DIMENSION);
    } else {
        newHeight = MAX_IMAGE_DIMENSION;
        newWidth = Math.round((width / height) * MAX_IMAGE_DIMENSION);
    }

    // Create canvas and draw resized image
    const resizedCanvas = canvas.createCanvas(newWidth, newHeight);
    const ctx = resizedCanvas.getContext('2d');
    ctx.drawImage(image, 0, 0, newWidth, newHeight);

    // Encode back to base64
    let outputBuffer: Buffer;
    if (mediaType === 'image/png') {
        outputBuffer = resizedCanvas.toBuffer('image/png');
    } else {
        outputBuffer = resizedCanvas.toBuffer('image/jpeg');
    }

    const newBase64 = outputBuffer.toString('base64');

    console.log(`Resized image from ${width}x${height} to ${newWidth}x${newHeight}`);

    return { data: newBase64, mediaType, resized: true };
}

// UPLOAD DOCUMENT TO PROJECT
documentsRoute.post('/projects/:projectId/documents', async (c) => {
    const supabase = getScopedClient(c);
    const projectId = c.req.param('projectId');

    try {
        // Get form data
        const formData = await c.req.formData();
        const file = formData.get('file') as File;
        const title = formData.get('title') as string || file.name;
        const description = formData.get('description') as string || '';

        if (!file) {
            return c.json({ error: 'No file provided' }, 400);
        }

        // Convert File to Buffer
        const arrayBuffer = await file.arrayBuffer();
        const buffer = Buffer.from(arrayBuffer);

        // Generate unique filename
        const timestamp = Date.now();
        const filename = `${projectId}/${timestamp}_${file.name}`;

        // Upload to Supabase Storage
        const { data: uploadData, error: uploadError } = await supabase.storage
            .from('project-documents')
            .upload(filename, buffer, {
                contentType: file.type,
                upsert: false
            });

        if (uploadError) {
            console.error('Upload error:', uploadError);
            return c.json({ error: uploadError.message }, 500);
        }

        // Get file metadata
        const metadata: any = {
            original_name: file.name,
            mime_type: file.type
        };

        // For PDFs, try to get page count
        if (file.type === 'application/pdf') {
            try {
                const parser = new PDFParser(supabase);
                const textData = await parser.extractText(buffer);
                metadata.page_count = textData.pages.length;
            } catch (error) {
                console.error('Failed to extract PDF metadata:', error);
            }
        }

        // Create document record
        const { data: document, error: dbError } = await supabase
            .from('project_documents')
            .insert({
                project_id: projectId,
                title,
                description,
                file_url: uploadData.path,
                file_type: file.type,
                file_size_bytes: file.size,
                metadata
            })
            .select()
            .single();

        if (dbError) {
            console.error('Database error:', dbError);
            // Try to clean up uploaded file
            await supabase.storage.from('project-documents').remove([uploadData.path]);
            return c.json({ error: dbError.message }, 500);
        }

        return c.json(document, 201);
    } catch (error) {
        console.error('Error uploading document:', error);
        return c.json({
            error: error instanceof Error ? error.message : 'Failed to upload document'
        }, 500);
    }
});

        .from('project_documents')
        .delete()
        .eq('id', id);

        return c.json({ error: dbError.message }, 500);

    return c.json({ success: true });

// PARSE DOCUMENT WITH CLAUDE
documentsRoute.post('/documents/:id/parse', async (c) => {
    const supabase = getScopedClient(c);
    const id = c.req.param('id');

    try {
        // Check for API key
        const apiKey = process.env.ANTHROPIC_API_KEY;
        if (!apiKey) {
            return c.json({ error: 'ANTHROPIC_API_KEY not configured' }, 500);
        }

        // Get request body for optional custom prompt
        let customPrompt: string | undefined;
        try {
            const body = await c.req.json();
            customPrompt = body?.customPrompt;
        } catch {
            // No body provided, use default prompt
        }

        // Get document
        const { data: document, error: fetchError } = await supabase
            .from('project_documents')
            .select('*')
            .eq('id', id)
            .single();

        if (fetchError || !document) {
            return c.json({ error: 'Document not found' }, 404);
        }

        // Get extracted images for this document
        const { data: images, error: imagesError } = await supabase
            .from('document_images')
            .select('*')
            .eq('document_id', id)
            .order('page_number', { ascending: true });

        if (imagesError) {
            return c.json({ error: 'Failed to fetch document images' }, 500);
        }

        if (!images || images.length === 0) {
            return c.json({
                error: 'No images found. Please extract images first using /documents/:id/extract-images'
            }, 400);
        }

        console.log(`Parsing document ${id} with ${images.length} images using Claude...`);

        // Download images, resize if needed, and convert to base64
        // (Claude API requires HTTPS URLs but local Supabase uses HTTP, so we send base64 data)
        // Claude also limits images to 2000px max dimension for multi-image requests
        const base64Images: { data: string; mediaType: 'image/png' | 'image/jpeg' }[] = [];

        for (const image of images) {
            try {
                // Download image from storage
                const { data: imageData, error: downloadError } = await supabase.storage
                    .from('project-documents')
                    .download(image.image_url);

                if (downloadError || !imageData) {
                    console.error(`Failed to download image ${image.id}:`, downloadError);
                    continue;
                }

                // Convert to base64
                const arrayBuffer = await imageData.arrayBuffer();
                const base64 = Buffer.from(arrayBuffer).toString('base64');
                const mediaType = image.image_url.endsWith('.png') ? 'image/png' : 'image/jpeg';

                // Resize if needed (Claude limits images to 2000px for multi-image requests)
                const resizedImage = await resizeImageIfNeeded(base64, mediaType);

                base64Images.push({ data: resizedImage.data, mediaType: resizedImage.mediaType });
                const sizeKB = (Buffer.from(resizedImage.data, 'base64').length / 1024).toFixed(1);
                console.log(`Loaded image ${image.page_number || 'unknown'} (${sizeKB} KB)${resizedImage.resized ? ' [resized]' : ''}`);
            } catch (err) {
                console.error(`Failed to process image ${image.id}:`, err);
            }
        }

        if (base64Images.length === 0) {
            return c.json({ error: 'Failed to load any images' }, 500);
        }

        console.log(`Sending ${base64Images.length} images to Claude for parsing...`);

        // Parse with Claude using base64 images
        // Use batch processing for large documents (> 10 pages) to avoid token limits
        const claudeParser = new ClaudePdfParser(apiKey);
        const BATCH_SIZE = 10; // Process 10 pages at a time
        const parsedData = base64Images.length > BATCH_SIZE
            ? await claudeParser.parseInBatches(base64Images, id, BATCH_SIZE, customPrompt)
            : await claudeParser.parseBase64Images(base64Images, id, customPrompt);

        // Store parsed data in document metadata
        const { error: updateError } = await supabase
            .from('project_documents')
            .update({
                metadata: {
                    ...document.metadata,
                    parsed_data: parsedData,
                    parsed_at: new Date().toISOString()
                }
            })
            .eq('id', id);

        if (updateError) {
            console.error('Failed to save parsed data:', updateError);
        }

        return c.json({
            success: true,
            document_id: id,
            parsed: parsedData,
            summary: {
                units_count: parsedData.units?.length ?? 0,
                total_baseboard_ft: parsedData.total_baseboard_ft ?? 0,
                trades: parsedData.summary?.trades ?? []
            }
        });
    } catch (error) {
        console.error('Error parsing document with Claude:', error);
        return c.json({
            error: error instanceof Error ? error.message : 'Failed to parse document'
        }, 500);
    }
});
