import os
import requests
import json
import argparse
from pathlib import Path

# Configuration
API_URL = "http://localhost:5000/benchmark"

def run_benchmark(file_path: str, provider: str, extraction_type: str, page: int = 1):
    """
    Run a single benchmark request against the API.
    """
    print(f"--- Benchmarking {os.path.basename(file_path)} ---")
    print(f"Provider: {provider}")
    print(f"Type: {extraction_type}")
    
    try:
        with open(file_path, 'rb') as f:
            files = {'file': f}
            data = {
                'type': extraction_type,
                'page': str(page),
                'provider': provider
            }
            
            response = requests.post(API_URL, files=files, data=data)
            
            if response.status_code == 200:
                result = response.json()
                print("Success!")
                # print(json.dumps(result['result'], indent=2))
                return result['result']
            else:
                print(f"Error {response.status_code}: {response.text}")
                return None
    except Exception as e:
        print(f"Exception: {e}")
        return None

def main():
    parser = argparse.ArgumentParser(description="Run PDF Benchmark")
    parser.add_argument("file", help="Path to PDF file")
    parser.add_argument("--provider", default="mock", help="Provider (mock, docling, openai/gpt-4o, etc)")
    parser.add_argument("--type", default="table", help="Extraction type (table, floorplan, text)")
    parser.add_argument("--page", type=int, default=1, help="Page number")
    parser.add_argument("--output", help="Output JSON file")
    
    args = parser.parse_args()
    
    result = run_benchmark(args.file, args.provider, args.type, args.page)
    
    if args.output and result:
        with open(args.output, 'w') as f:
            json.dump(result, f, indent=2)
        print(f"Result saved to {args.output}")

if __name__ == "__main__":
    main()

import os
import requests
import argparse
import time

API_URL_DIRECT = "http://localhost:5000/parse/direct"
API_URL_PREPARSE = "http://localhost:5000/parse/preparse"

def run_method(api_url: str, method_name: str, file_path: str, output_path: str, page: int):
    print(f"\\n--- Running {method_name} Method ---")
    data = {
        'file_path': file_path,
        'output_path': output_path,
        'page': str(page),
        'provider': 'openai/gpt-4o'
    }
    
    start_time = time.time()
    try:
        response = requests.post(api_url, data=data)
        elapsed = time.time() - start_time
        
        if response.status_code == 200:
            result = response.json()
            print(f"Success! Time taken: {elapsed:.2f}s")
            print(f"Results saved internally to: {output_path}")
            return True
        else:
            print(f"Error {response.status_code}: {response.text}")
            return False
            
    except requests.exceptions.ConnectionError:
        print("Error: Could not connect to the API server at http://localhost:5000.")
        print("Make sure the server is running ('uvicorn server:app --port 5000').")
        return False
    except Exception as e:
        print(f"Exception: {e}")
        return False


def main():
    parser = argparse.ArgumentParser(description="Compare PDF Parsing Strategies")
    parser.add_argument("file", help="Path to PDF file (e.g., demo.pdf)")
    parser.add_argument("--page", type=int, default=1, help="Page number to parse")
    parser.add_argument("--outdir", default="output", help="Base output directory")
    
    args = parser.parse_args()
    
    if not os.path.exists(args.file):
        print(f"Error: Input file {args.file} not found.")
        return
        
    print(f"Benchmarking: {args.file} (Page {args.page})")
    
    # 1. Run Direct Method
    run_method(
        API_URL_DIRECT,
        "DIRECT LLM",
        args.file,
        os.path.join(args.outdir, "direct_run"),
        args.page
    )
    
    # 2. Run Preparse Method
    run_method(
        API_URL_PREPARSE,
        "PRE-PARSED + LLM",
        args.file,
        os.path.join(args.outdir, "preparse_run"),
        args.page
    )
    
    print("\\n--- Done ---")
    print(f"Check the '{args.outdir}/' folder to compare the trace artifacts and JSON outputs.")

if __name__ == "__main__":
    main()
