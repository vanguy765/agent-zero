/**
 * Doc-Parser Proxy Routes for Hono API
 *
 * Integrates doc-parser service into the existing work-orders API.
 * Provides engine selection, status polling, result retrieval,
 * and a full end-to-end pipeline with Supabase DB persistence.
 *
 * Routes:
 *   POST /api/doc-parser/parse           - Parse document (engine selection, returns job_id)
 *   POST /api/doc-parser/pipeline        - Full pipeline: submit + poll + persist to DB
 *   GET  /api/doc-parser/status/:jobId  - Lightweight status (poll-safe)
 *   GET  /api/doc-parser/results/:jobId - Full results (fetch-once)
 *   GET  /api/doc-parser/health         - Service health check
 */

import { Hono } from 'hono';
import { createClient } from '@supabase/supabase-js';
import { SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY } from '@repo/config';
import { getScopedClient } from '../middleware/supabase';
import {
  submitParseJob,
  getJobStatus,
  getJobResults,
  checkHealth,
  mapToParseDocument,
  type DocParserJobResponse,
  type DocParserStatus,
  type DocParserResult,
} from '../services/doc-parser-client';

const docParserRoutes = new Hono();

// ============================================================
// POST /parse
// Accepts JSON { document_id, engine }
// Fetches PDF from Supabase storage, forwards to doc-parser
// Returns job_id immediately — caller must poll /status/:jobId
// ============================================================

docParserRoutes.post('/parse', async (c) => {
  const body = await c.req.json().catch(() => ({}));
  const documentId = body?.document_id;
  const engine = (body?.engine || 'hybrid') as string;

  if (!documentId) {
    return c.json({ error: 'document_id is required' }, 400);
  }

  if (engine === 'claude-direct') {
    return c.json({
      error: 'claude-direct engine should be handled by existing route handler',
      hint: 'This route only handles docling and hybrid engines',
    }, 400);
  }

  if (engine !== 'docling' && engine !== 'hybrid') {
    return c.json(
      { error: `Unknown engine: ${engine}. Use 'claude-direct', 'docling', or 'hybrid'.` },
      400,
    );
  }

  try {
    const supabase = getScopedClient(c);
    const { data: document, error: fetchError } = await supabase
      .from('project_documents')
      .select('*')
      .eq('id', documentId)
      .single();

    if (fetchError || !document) {
      return c.json({ error: 'Document not found' }, 404);
    }
    if (!document.file_type?.includes('pdf')) {
      return c.json({ error: 'Document is not a PDF' }, 400);
    }

    const { data: fileData, error: downloadError } = await supabase.storage
      .from('project-documents')
      .download(document.file_url);

    if (downloadError || !fileData) {
      console.error('[doc-parser] Storage download failed:', downloadError);
      return c.json({ error: 'Failed to download document from storage' }, 500);
    }

    const pdfBuffer = await fileData.arrayBuffer();

    const jobResponse = await submitParseJob(
      pdfBuffer,
      document.title || `document-${documentId}.pdf`,
      engine as 'docling' | 'hybrid',
      {
        provider: body?.provider || process.env.DEFAULT_LLM_PROVIDER,
        tenantName: body?.tenant_name,
        projectName: body?.project_name || documentId,
        supabaseUrl: process.env.SUPABASE_URL,
        supabaseKey: process.env.SUPABASE_SERVICE_KEY,
      },
    );

    return c.json({
      engine,
      job_id: jobResponse.job_id,
      status: jobResponse.status,
      poll_url: `/api/doc-parser/status/${jobResponse.job_id}`,
      results_url: `/api/doc-parser/results/${jobResponse.job_id}`,
    });
  } catch (error) {
    const message = error instanceof Error ? error.message : 'Unknown error';
    console.error(`[doc-parser] Parse submission failed: ${message}`);
    return c.json({ error: `Doc-parser submission failed: ${message}` }, 502);
  }
});

// ============================================================
// POST /pipeline
// Full end-to-end pipeline — blocks until complete:
//   1. Download PDF from Supabase storage
//   2. Submit to doc-parser
//   3. Poll internally until completed / failed / timeout
//   4. Persist document_pages + ocr_results to DB
//   5. Return summary
//
// Body: {
//   document_id: string        (required)
//   engine?: 'docling'|'hybrid'  (default: 'hybrid')
//   poll_interval_ms?: number  (default: 3000)
//   timeout_ms?: number        (default: 600000 = 10 min)
//   tenant_name?: string
//   project_name?: string
// }
//
// Returns: {
//   document_id, job_id, engine, status,
//   pages_persisted, ocr_tables_persisted, summary,
//   poll_url, results_url
// }
//
// Note: Uses SUPABASE_SERVICE_ROLE_KEY for DB writes (bypasses RLS).
// ============================================================

docParserRoutes.post('/pipeline', async (c) => {
  const body = await c.req.json().catch(() => ({}));
  const documentId = body?.document_id as string | undefined;
  const engine = (body?.engine || 'hybrid') as 'docling' | 'hybrid';
  const pollIntervalMs = Number(body?.poll_interval_ms ?? 3000);
  const timeoutMs = Number(body?.timeout_ms ?? 600_000);

  if (!documentId) {
    return c.json({ error: 'document_id is required' }, 400);
  }
  if (engine !== 'docling' && engine !== 'hybrid') {
    return c.json(
      { error: `Unknown engine: ${engine}. Use 'docling' or 'hybrid'.` },
      400,
    );
  }

  // Service-role client for DB writes (bypasses RLS)
  const serviceClient = createClient(SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY);

  // ----------------------------------------------------------
  // 1. Fetch document metadata
  // ----------------------------------------------------------
  const { data: document, error: fetchError } = await serviceClient
    .from('project_documents')
    .select('*')
    .eq('id', documentId)
    .single();

  if (fetchError || !document) {
    return c.json({ error: 'Document not found' }, 404);
  }
  if (!document.file_type?.includes('pdf')) {
    return c.json({ error: 'Document is not a PDF' }, 400);
  }

  // ----------------------------------------------------------
  // 2. Download PDF from storage
  // ----------------------------------------------------------
  const { data: fileData, error: downloadError } = await serviceClient.storage
    .from('project-documents')
    .download(document.file_url);

  if (downloadError || !fileData) {
    console.error('[pipeline] Storage download failed:', downloadError);
    return c.json({ error: 'Failed to download document from storage' }, 500);
  }

  const pdfBuffer = await fileData.arrayBuffer();

  // ----------------------------------------------------------
  // 3. Submit to doc-parser
  // ----------------------------------------------------------
  let jobResponse: DocParserJobResponse;
  try {
    jobResponse = await submitParseJob(
      pdfBuffer,
      document.title || `document-${documentId}.pdf`,
      engine,
      {
        supabaseUrl: process.env.SUPABASE_URL,
        supabaseKey: process.env.SUPABASE_SERVICE_KEY,
        projectName: body?.project_name || documentId,
        tenantName: body?.tenant_name,
      },
    );
  } catch (err) {
    const msg = err instanceof Error ? err.message : 'Unknown error';
    console.error('[pipeline] Job submission failed:', msg);
    return c.json({ error: `Doc-parser submission failed: ${msg}` }, 502);
  }

  const jobId = jobResponse.job_id;
  console.log(`[pipeline] Job ${jobId} submitted (engine=${engine})`);

  // ----------------------------------------------------------
  // 4. Poll until complete or timeout
  // ----------------------------------------------------------
  const deadline = Date.now() + timeoutMs;
  let status: DocParserStatus;

  while (true) {
    if (Date.now() > deadline) {
      return c.json({
        error: `Pipeline timed out after ${timeoutMs / 1000}s`,
        job_id: jobId,
      }, 504);
    }

    try {
      status = await getJobStatus(jobId);
    } catch (err) {
      const msg = err instanceof Error ? err.message : 'Unknown error';
      return c.json({ error: `Status poll failed: ${msg}`, job_id: jobId }, 502);
    }

    console.log(
      `[pipeline] Job ${jobId} status=${status.status}` +
      ` (${status.progress?.percent_complete ?? 0}%)`,
    );

    if (status.status === 'completed') break;
    if (status.status === 'failed' || status.status === 'timeout') {
      return c.json({
        error: `Doc-parser job ${status.status}: ${status.error ?? 'no details'}`,
        job_id: jobId,
      }, 502);
    }

    await new Promise((resolve) => setTimeout(resolve, pollIntervalMs));
  }

  // ----------------------------------------------------------
  // 5. Fetch full results
  // ----------------------------------------------------------
  let rawResults: { job_id: string; status: string; result?: DocParserResult };
  try {
    rawResults = await getJobResults(jobId, false);
  } catch (err) {
    const msg = err instanceof Error ? err.message : 'Unknown error';
    return c.json({ error: `Results fetch failed: ${msg}`, job_id: jobId }, 502);
  }

  const result = rawResults.result;
  if (!result) {
    return c.json(
      { error: 'Job completed but result payload is empty', job_id: jobId },
      502,
    );
  }

  // ----------------------------------------------------------
  // 6. Persist document_pages
  // ----------------------------------------------------------
  const pageEntries = Object.entries(result.pages || {});
  const pageIdMap: Record<number, string> = {};

  for (const [pageNumStr, pageData] of pageEntries) {
    const pageNumber = parseInt(pageNumStr, 10);
    const pd = pageData as any;

    let category: string = 'text';
    if (pd?.has_floorplan || pd?.is_floorplan)        category = 'floorplan';
    else if (pd?.has_table || (pd?.table_count ?? 0) > 0) category = 'table';
    else if (pd?.is_site_photo)                        category = 'site_photo';
    else if (pd?.is_diagram)                           category = 'diagram';
    else if (pd?.has_text && (pd?.has_floorplan || pd?.has_table)) category = 'mixed';

    const { data: pageRow, error: pageErr } = await serviceClient
      .from('document_pages')
      .upsert(
        {
          document_id: documentId,
          page_number: pageNumber,
          storage_url: pd?.storage_url ?? pd?.image_url ?? null,
          thumbnail_url: pd?.thumbnail_url ?? null,
          category,
          raw_text: pd?.raw_text ?? pd?.text ?? null,
          metadata: pd ?? null,
        },
        { onConflict: 'document_id,page_number', ignoreDuplicates: false },
      )
      .select('id')
      .single();

    if (pageErr) {
      console.error(
        `[pipeline] document_pages upsert failed for page ${pageNumber}:`,
        pageErr.message,
      );
    } else if (pageRow) {
      pageIdMap[pageNumber] = pageRow.id;
    }
  }

  console.log(
    `[pipeline] Persisted ${Object.keys(pageIdMap).length}/${pageEntries.length} pages`,
  );

  // ----------------------------------------------------------
  // 7. Persist ocr_results (one row per extracted table)
  // ----------------------------------------------------------
  const tables: any[] = result.tables || [];
  let ocrCount = 0;

  for (let tableIndex = 0; tableIndex < tables.length; tableIndex++) {
    const tbl = tables[tableIndex] as any;
    const pageNumber: number | null = tbl?.page_number ?? tbl?.source_page ?? null;
    const pageId = pageNumber !== null ? (pageIdMap[pageNumber] ?? null) : null;

    const { error: ocrErr } = await serviceClient
      .from('ocr_results')
      .insert({
        document_id: documentId,
        page_id: pageId,
        table_index: tableIndex,
        markdown: tbl?.markdown ?? tbl?.content ?? null,
        structured_data: tbl?.structured_data ?? tbl?.data ?? tbl ?? null,
        confidence: tbl?.confidence ?? null,
      });

    if (ocrErr) {
      console.error(
        `[pipeline] ocr_results insert failed for table ${tableIndex}:`,
        ocrErr.message,
      );
    } else {
      ocrCount++;
    }
  }

  console.log(`[pipeline] Persisted ${ocrCount}/${tables.length} OCR table rows`);

  // ----------------------------------------------------------
  // 8. Return summary
  // ----------------------------------------------------------
  return c.json({
    document_id: documentId,
    job_id: jobId,
    engine,
    status: 'completed',
    pages_persisted: Object.keys(pageIdMap).length,
    ocr_tables_persisted: ocrCount,
    summary: result.summary ?? null,
    poll_url: `/api/doc-parser/status/${jobId}`,
    results_url: `/api/doc-parser/results/${jobId}`,
  });
});

// ============================================================
// GET /api/doc-parser/status/:jobId
// Lightweight status for polling (poll-safe, small payload)
// Safe to call every 2-5 seconds.
// ============================================================

docParserRoutes.get('/status/:jobId', async (c) => {
  const jobId = c.req.param('jobId');

  try {
    const status: DocParserStatus = await getJobStatus(jobId);
    const safeStatus = { ...status };
    delete (safeStatus as any).result;
    return c.json(safeStatus);
  } catch (error) {
    const message = error instanceof Error ? error.message : 'Unknown error';
    if (message.includes('not found')) {
      return c.json({ error: `Job ${jobId} not found` }, 404);
    }
    return c.json({ error: `Status check failed: ${message}` }, 502);
  }
});

// ============================================================
// GET /api/doc-parser/results/:jobId
// Full results — fetch once after status confirms "completed".
// Heavy payload. Includes ParsedDocument-compatible mapping.
// ============================================================

docParserRoutes.get('/results/:jobId', async (c) => {
  const jobId = c.req.param('jobId');
  const summaryOnly = c.req.query('summary') === 'true';

  try {
    const data = await getJobResults(jobId, summaryOnly);

    if (summaryOnly) {
      return c.json(data);
    }

    const result = data.result as DocParserResult | undefined;
    if (result?.units) {
      return c.json({
        ...data,
        parsed: mapToParseDocument(result, jobId),
      });
    }

    return c.json(data);
  } catch (error) {
    const message = error instanceof Error ? error.message : 'Unknown error';
    if (message.includes('not found')) {
      return c.json({ error: `Results for job ${jobId} not found` }, 404);
    }
    return c.json({ error: `Results fetch failed: ${message}` }, 502);
  }
});

// ============================================================
// GET /api/doc-parser/health
// Service health check with dependency status
// ============================================================

docParserRoutes.get('/health', async (c) => {
  const health = await checkHealth();
  const statusCode = health.status === 'healthy' ? 200 : 503;
  return c.json(health, statusCode);
});

export { docParserRoutes };
export default docParserRoutes;

 *
 * Shared service for communicating with the doc-parser backend.

const DOC_PARSER_URL = process.env.DOC_PARSER_URL || 'http://localhost:5000';

export interface DocParserJobResponse {
  job_id: string;
  status: 'queued' | 'processing' | 'completed' | 'failed' | 'timeout';
  engine: 'docling' | 'hybrid';
  file: string;
}

export interface DocParserProgress {
  current_phase: string;
  phases: string[];
  current_page: number;
  total_pages: number;
  current_step: string;
  floorplans_found: number;
  tables_found: number;
  percent_complete: number;
}

export interface DocParserStatus {
  job_id: string;
  status: 'queued' | 'processing' | 'completed' | 'failed' | 'timeout';
  engine: string;
  file_name: string;
  progress: DocParserProgress;
  error?: string;
  created_at: string;
  updated_at: string;
}

export interface DocParserResultSummary {
  pages_count: number;
  floorplans_count: number;
  tables_count: number;
  units_count: number;
  has_footer: boolean;
  total_doors: number;
  total_windows: number;
  total_baseboard_ft: number;
  trades: string[];
}

export interface DocParserConfidence {
  overall: number;
  room_dimensions: boolean;
  door_specifications: boolean;
  window_specifications: boolean;
  fixture_completeness: boolean;
  cross_page_synthesis: boolean;
  baseboard_data: boolean;
  unit_identification: boolean;
  per_page: Record<string, any>;
}

export interface DocParserUnit {
  unit_number: string;
  unit_type: string;
  floor: number;
  bedrooms: number;
  bathrooms: number;
  half_baths: number;
  area_sf: number;
  rooms: any[];
  doors: any[];
  windows: any[];
  fixtures: any[];
  upgrades: any[];
  general_notes: string[];
  source_page: number;
}

export interface DocParserResult {
  engine: string;
  pages: Record<number, any>;
  units: DocParserUnit[];
  tables: any[];
  footer_sample: any;
  general_notes: string[];
  synthesis: any;
  confidence: DocParserConfidence;
  summary: DocParserResultSummary;
}

/**
 * Submit a PDF for parsing with the specified engine.
 */
export async function submitParseJob(
  pdfBuffer: Buffer | ArrayBuffer,
  fileName: string,
  engine: 'docling' | 'hybrid',
  options: {
    provider?: string;
    tenantName?: string;
    projectName?: string;
    supabaseUrl?: string;
    supabaseKey?: string;
  } = {},
): Promise<DocParserJobResponse> {
  const endpoint = engine === 'hybrid' ? '/parse/hybrid' : '/parse/docling';

  const formData = new FormData();
  const blob = new Blob([new Uint8Array(pdfBuffer as any)], { type: 'application/pdf' });
  formData.append('file', blob, fileName);

  if (options.provider) {
    formData.append('provider', options.provider);
  }
  if (options.tenantName) {
    formData.append('tenant_name', options.tenantName);
  }
  if (options.projectName) {
    formData.append('project_name', options.projectName);
  }
  if (options.supabaseUrl) {
    formData.append('supabase_url', options.supabaseUrl);
  }
  if (options.supabaseKey) {
    formData.append('supabase_key', options.supabaseKey);
  }

  const response = await fetch(`${DOC_PARSER_URL}${endpoint}`, {
    method: 'POST',
    body: formData,
  });

  if (!response.ok) {
    const errorText = await response.text();
    throw new Error(
      `Doc-parser submission failed (${response.status}): ${errorText}`,
    );
  }

  return response.json();
}

/**
 * Get lightweight job status (poll-safe).
 * NEVER includes result data.
 */
export async function getJobStatus(
  jobId: string,
): Promise<DocParserStatus> {
  const response = await fetch(`${DOC_PARSER_URL}/status/${jobId}`);

  if (!response.ok) {
    if (response.status === 404) {
      throw new Error(`Job ${jobId} not found`);
    }
    throw new Error(`Status check failed (${response.status})`);
  }

  return response.json();
}

/**
 * Get full extraction results (fetch-once after completed).
 * Heavy payload - only call when status confirms "completed".
 */
export async function getJobResults(
  jobId: string,
  summaryOnly: boolean = false,
): Promise<{ job_id: string; status: string; result?: DocParserResult; summary?: DocParserResultSummary }> {
  const url = summaryOnly
    ? `${DOC_PARSER_URL}/results/${jobId}?summary=true`
    : `${DOC_PARSER_URL}/results/${jobId}`;

  const response = await fetch(url);

  if (!response.ok) {
    if (response.status === 404) {
      throw new Error(`Results for job ${jobId} not found`);
    }
    throw new Error(`Results fetch failed (${response.status})`);
  }

  return response.json();
}

/**
 * Check if doc-parser service is healthy and available.
 */
export async function checkHealth(): Promise<{
  status: string;
  engines: string[];
  dependencies: Record<string, boolean>;
}> {
  try {
    const response = await fetch(`${DOC_PARSER_URL}/health`, {
      signal: AbortSignal.timeout(5000),
    });
    if (!response.ok) {
      return {
        status: 'unhealthy',
        engines: [],
        dependencies: {},
      };
    }
    return response.json();
  } catch {
    return {
      status: 'unreachable',
      engines: [],
      dependencies: {},
    };
  }
}

/**
 * Map hybrid engine output to ParsedDocument-compatible format.
 *
 * When engine is "hybrid", the units array is already
 * ParsedDocument-compatible. This mapper normalizes the
 * response for the frontend.
 */
export function mapToParseDocument(
  result: DocParserResult,
  documentId: string,
): {
  document_id: string;
  units: DocParserUnit[];
  general_notes: string[];
  total_baseboard_ft: number;
  summary: DocParserResultSummary;
  confidence: DocParserConfidence;
} {
  return {
    document_id: documentId,
    units: result.units || [],
    general_notes: result.general_notes || [],
    total_baseboard_ft: result.summary?.total_baseboard_ft || 0,
    summary: result.summary,
    confidence: result.confidence,
  };
}
