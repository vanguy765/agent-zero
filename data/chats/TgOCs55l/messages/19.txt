import fs from 'fs';
import path from 'path';
import { createRequire } from 'module';
import dotenv from 'dotenv';
import postgres from 'postgres';

const require = createRequire(import.meta.url);
const pdf = require('pdf-parse');

// Load env vars
dotenv.config({ path: path.join(__dirname, '../../../.env') });

const connectionString = process.env.DATABASE_URL;
if (!connectionString) {
  console.error('DATABASE_URL is not set');
  process.exit(1);
}
const sql = postgres(connectionString);

const PDF_PATH = path.join(__dirname, '../../../temp_analysis.pdf');

// Regex patterns
const UNIT_PAGE_REGEX = /UNIT\s+(\d+)/i;
// Relaxed Regex (removed ^) and fixed Bathtub typo
const ROW_REGEX = /(Ceiling|Walls|Floors|Entry|Interior|Bifold|Sink|Faucet|Lavatory|Toilet|Bathtub|Bathub|Countertop|Cabinets|Shelving|Vanity|Range|Fridge|Hood Fan)/i;

async function parsePDF() {
    console.log('Starting PDF parsing...');
    if (!fs.existsSync(PDF_PATH)) {
        console.error(`PDF not found at ${PDF_PATH}`);
        process.exit(1);
    }

    const dataBuffer = fs.readFileSync(PDF_PATH);
    const data = await pdf(dataBuffer);
    const rawText = data.text;
    
    // Split by the header that appears on every page
    const unitBlocks = rawText.split(/SCH'EYK HOUSING SOCIETY UNIT TURNOVER PROJECT/i);

    let totalItemsInserted = 0;

    for (const block of unitBlocks) {
        // Look for "BH Unit XXX"
        const unitMatch = block.match(/BH Unit\s+(\d+)/i);
        if (!unitMatch) continue;

        const unitNum = unitMatch[1];
        
        // 1. Insert Unit
        const [unit] = await sql`
            INSERT INTO units (unit_number, project_id)
            VALUES (${unitNum}, '00000000-0000-0000-0000-000000000000')
            ON CONFLICT (unit_number) DO UPDATE SET unit_number = EXCLUDED.unit_number
            RETURNING id
        `;

        // 2. Parse Rows
        const lines = block.split('\n').map(l => l.trim()).filter(l => l);
        let currentSection = 'General';
        
        for (const line of lines) {
            // Detect Section Headers
            if (line.includes('1.0 FINISHES')) currentSection = 'Finishes';
            else if (line.includes('2.0 DOORS')) currentSection = 'Doors';
            else if (line.includes('3.0 LIGHT FIXTURES')) currentSection = 'Electrical';
            else if (line.includes('4.0 KITCHEN PLUMBING')) currentSection = 'Plumbing';
            else if (line.includes('5.0 BATHROOM PLUMBING')) currentSection = 'Plumbing';
            else if (line.includes('6.0 MILLWORK')) currentSection = 'Millwork';
            else if (line.includes('7.0 APPLIANCES')) currentSection = 'Appliances';
            else if (line.includes('8.0 HEATING')) currentSection = 'Heating';
            else if (line.includes('10.0 WINDOW')) currentSection = 'Windows';
            
            // Detect Item Row
            const rowMatch = line.match(ROW_REGEX);
            if (rowMatch) {
                const item = rowMatch[1];
                let action = line.replace(item, '').trim(); // Fallback action extraction
                
                // If the line *starts* with the item (fuzzy), assume the rest is action.
                // Since regex is loose, 'item' might be in the middle.
                // We'll try to split by the item match index.
                const matchIndex = line.toLowerCase().indexOf(item.toLowerCase());
                if (matchIndex >= 0) {
                     action = line.substring(matchIndex + item.length).trim();
                }

                if (action.length > 100) action = action.substring(0, 100);

                // Determine Trade
                let trade = 'General';
                const lowerItem = item.toLowerCase();
                const lowerAction = action.toLowerCase();
                const lowerSection = currentSection.toLowerCase();

                if (lowerSection === 'plumbing') trade = 'Plumber';
                else if (lowerSection === 'electrical' || lowerItem.includes('light')) trade = 'Electrician';
                else if (lowerAction.includes('paint')) trade = 'Painter';
                else if (lowerItem.includes('cabinet') || lowerItem.includes('countertop') || lowerItem.includes('shelf') || lowerItem.includes('vanity')) trade = 'Carpenter';
                else if (lowerItem.includes('floor')) trade = 'Flooring';
                else if (lowerItem.includes('door') || lowerItem.includes('entry') || lowerItem.includes('bifold')) trade = 'Carpenter';
                else if (lowerSection === 'appliances') trade = 'Appliances';
                else if (lowerSection === 'heating') trade = 'HVAC';

                if (action.length > 0) { // Only insert if there is an action
                    await sql`
                        INSERT INTO unit_scope_items (unit_id, zone, category, item, action, trade, status)
                        VALUES (${unit.id}, ${currentSection}, ${currentSection}, ${item}, ${action}, ${trade}, 'pending')
                         ON CONFLICT DO NOTHING
                    `;
                    totalItemsInserted++;
                }
            }
        }
    }
    
    fs.writeFileSync(path.join(process.cwd(), 'parsing_result.txt'), `Parsing complete. Inserted ${totalItemsInserted} items.`);
    console.log(`Parsing complete. Inserted ${totalItemsInserted} items.`);
    await sql.end();
}

parsePDF().catch(console.error);
