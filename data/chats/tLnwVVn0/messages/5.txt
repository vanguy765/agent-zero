def encode_image_base64_from_pix(pix):
    """Convert PyMuPDF pixmap to base64 string"""
    img_data = pix.tobytes("png")
    return base64.b64encode(img_data).decode("utf-8")

def encode_image_base64_from_file(filepath: str):
    """Convert local image file to base64 string"""
    with open(filepath, "rb") as image_file:
        return base64.b64encode(image_file.read()).decode('utf-8')

async def run_llm_extraction(provider: str, prompt_template: str, image_base64: str, text_context: str = None):
    """Run extraction using LiteLLM (OpenRouter)"""
    if not OPENROUTER_API_KEY:
        raise HTTPException(status_code=400, detail="OPENROUTER_API_KEY not set")

    model = f"openrouter/{provider}"
    
    content = [{"type": "text", "text": prompt_template}]
    
    if text_context:
        content.append({"type": "text", "text": f"\\nExtracted Raw Text/Data Context:\\n{text_context}"})
        
    content.append({
        "type": "image_url",
        "image_url": {"url": f"data:image/png;base64,{image_base64}"}
    })
    
    messages = [
        {
            "role": "user",
            "content": content
        }
    ]

    try:
        # Anthropic/Gemini might struggle with response_format in certain litellm versions
        kwargs = {
            "model": model,
            "messages": messages
        }
        if "openai" in provider.lower():
            kwargs["response_format"] = {"type": "json_object"}
            
        response = completion(**kwargs)
        return response.choices[0].message.content
    except Exception as e:
        print(f"LLM Error: {e}")
        return {{"error": str(e)}}


def is_black_and_white_pil(img_pil):
    img = np.array(img_pil)
    if len(img.shape) < 3:
        return True
    r, g, b = img[:,:,0], img[:,:,1], img[:,:,2]
    diff = np.mean(np.abs(r.astype(int) - g.astype(int))) + np.mean(np.abs(g.astype(int) - b.astype(int)))
    return diff < 2.0

def has_sufficient_text_pil(img_pil, min_words=3):
    text = pytesseract.image_to_string(img_pil)
    word_count = len(text.split())
    return word_count >= min_words

def calculate_areas(raw_json):
    CEILING_HEIGHT = 8.0
    WINDOW_AREA = 12.0
    INT_DOOR_AREA = 21.0
    SLIDING_DOOR_AREA = 42.0
    BIFOLD_DOOR_AREA = 35.0

    kitchen_area = float(raw_json.get("kitchen_area", 0) or 0)
    bathroom_area = float(raw_json.get("bathroom_area", 0) or 0)
    kitchen_perimeter = float(raw_json.get("kitchen_perimeter", 0) or 0)
    bathroom_perimeter = float(raw_json.get("bathroom_perimeter", 0) or 0)
    total_wall_perimeter = float(raw_json.get("total_wall_perimeter", 0) or 0)
    
    windows = float(raw_json.get("window_count", 0) or 0)
    int_doors = float(raw_json.get("interior_door_count", 0) or 0)
    slid_doors = float(raw_json.get("sliding_door_count", 0) or 0)
    bif_doors = float(raw_json.get("bifold_door_count", 0) or 0)

    calc = {
        "kitchen_ceiling_area": kitchen_area,
        "bathroom_ceiling_area": bathroom_area,
        "kitchen_wall_area": kitchen_perimeter * CEILING_HEIGHT,
        "bathroom_wall_area": bathroom_perimeter * CEILING_HEIGHT,
        "gross_unit_wall_area": total_wall_perimeter * CEILING_HEIGHT,
    }

    total_deductions_area = (windows * WINDOW_AREA) + (int_doors * INT_DOOR_AREA) + (slid_doors * SLIDING_DOOR_AREA) + (bif_doors * BIFOLD_DOOR_AREA)
    calc["total_deductions_area"] = total_deductions_area
    calc["net_remaining_wall_area"] = calc["gross_unit_wall_area"] - calc["kitchen_wall_area"] - calc["bathroom_wall_area"] - total_deductions_area

    return {**raw_json, "calculations": calc}

def compress_and_upload_to_supabase(sb_client, pil_img, bucket_path):
    """Compresses a PIL image to JPEG in memory and uploads directly to Supabase Storage."""
    from io import BytesIO
    import base64
    
    if not sb_client:
        # Fallback to local base64 if no supabase credentials
        buffered = BytesIO()
        pil_img.save(buffered, format="PNG")
        return base64.b64encode(buffered.getvalue()).decode("utf-8")
        
    try:
        # Compress immediately in-memory
        if pil_img.mode in ("RGBA", "P"):
            pil_img = pil_img.convert("RGB")
        
        img_byte_arr = BytesIO()
        pil_img.save(img_byte_arr, format='JPEG', optimize=True, quality=80)
        img_bytes = img_byte_arr.getvalue()
        
        # Upload directly to Supabase Storage
        res = sb_client.storage.from_("storage").upload(
            bucket_path,
            img_bytes,
            {"content-type": "image/jpeg", "upsert": "true"}
        )
        
        # Return the public URL or just the path itself
        return bucket_path
    except Exception as e:
        print(f"Failed to upload {bucket_path} to Supabase: {e}")
        return None
