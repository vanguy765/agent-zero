# Matches: 12\'-6", 12\'-6, 12\'6", 10\' x 12\', 3660mm, 3.5m, 10\'-0"
DIMENSION_PATTERNS = [
    r"\d+\'\s*-?\s*\d+\s*\"",          # 12\'-6"
    r"\d+\'\s*-?\s*\d+",               # 12\'-6 or 12-6
    r"\d+\'\s*x\s*\d+\'",              # 10\' x 12\'
    r"\d+\.\d+\s*m\b",                 # 3.5m
    r"\d{3,5}\s*mm\b",                 # 3660mm
    r"\d+\'-\d+\"\s*x\s*\d+\'-\d+\"", # 12\'-6" x 10\'-0"
    r"\d+\s*ft\.?\s*\d*\s*in\.?",      # 12 ft 6 in
]

DIMENSION_REGEX = re.compile("|".join(DIMENSION_PATTERNS), re.IGNORECASE)

# Architectural room/space labels
ARCH_LABELS = [
    "bedroom", "kitchen", "bathroom", "bath", "living", "dining",
    "closet", "hallway", "entry", "foyer", "laundry", "utility",
    "master", "garage", "patio", "balcony", "deck", "porch",
    "den", "study", "office", "pantry", "mudroom", "vestibule",
    "mechanical", "storage", "w.i.c", "wic", "w/d", "ref",
    "dw", "oven", "range", "shower", "tub", "lav", "wh",
]

ARCH_LABEL_REGEX = re.compile(
    r"\b(" + "|".join(ARCH_LABELS) + r")\b", re.IGNORECASE
)

# Scale notations
SCALE_PATTERNS = [
    r"scale\s*[:=]?\s*\d+",
    r"1/\d+\"\s*=\s*1\'-0\"",
    r"\d+\"\s*=\s*\d+\'-\d+\"",
    r"1:\d+",
    r"NTS",  # Not To Scale
]

SCALE_REGEX = re.compile("|".join(SCALE_PATTERNS), re.IGNORECASE)
