// Maximum dimension for Claude multi-image API
const MAX_IMAGE_DIMENSION = 2000;

/**
 * Resize an image to fit within MAX_IMAGE_DIMENSION while maintaining aspect ratio
 */
async function resizeImageIfNeeded(
    base64Data: string,
    mediaType: 'image/png' | 'image/jpeg'
): Promise<{ data: string; mediaType: 'image/png' | 'image/jpeg'; resized: boolean }> {
    // Decode base64 to buffer
    const buffer = Buffer.from(base64Data, 'base64');

    // Load image
    const image = await canvas.loadImage(buffer);
    const width = image.width;
    const height = image.height;

    // Check if resizing is needed
    if (width <= MAX_IMAGE_DIMENSION && height <= MAX_IMAGE_DIMENSION) {
        return { data: base64Data, mediaType, resized: false };
    }

    // Calculate new dimensions maintaining aspect ratio
    let newWidth: number;
    let newHeight: number;

    if (width > height) {
        newWidth = MAX_IMAGE_DIMENSION;
        newHeight = Math.round((height / width) * MAX_IMAGE_DIMENSION);
    } else {
        newHeight = MAX_IMAGE_DIMENSION;
        newWidth = Math.round((width / height) * MAX_IMAGE_DIMENSION);
    }

    // Create canvas and draw resized image
    const resizedCanvas = canvas.createCanvas(newWidth, newHeight);
    const ctx = resizedCanvas.getContext('2d');
    ctx.drawImage(image, 0, 0, newWidth, newHeight);

    // Encode back to base64
    let outputBuffer: Buffer;
    if (mediaType === 'image/png') {
        outputBuffer = resizedCanvas.toBuffer('image/png');
    } else {
        outputBuffer = resizedCanvas.toBuffer('image/jpeg');
    }

    const newBase64 = outputBuffer.toString('base64');

    console.log(`Resized image from ${width}x${height} to ${newWidth}x${newHeight}`);

    return { data: newBase64, mediaType, resized: true };
}

            original_name: file.name,

        };

        // For PDFs, try to get page count

                const parser = new PDFParser(supabase);

                console.error('Failed to extract PDF metadata:', error);

            .from('project_documents')

    const id = c.req.param('id');

    try {
        // Check for API key
        const apiKey = process.env.ANTHROPIC_API_KEY;
        if (!apiKey) {
            return c.json({ error: 'ANTHROPIC_API_KEY not configured' }, 500);
        }

        // Get request body for optional custom prompt
        let customPrompt: string | undefined;
        try {
            const body = await c.req.json();
            customPrompt = body?.customPrompt;
        } catch {
            // No body provided, use default prompt
        }

        // Get document
        const { data: document, error: fetchError } = await supabase
            .from('project_documents')
            .select('*')
            .eq('id', id)
            .single();

        if (fetchError || !document) {
            return c.json({ error: 'Document not found' }, 404);
        }

        // Get extracted images for this document
        const { data: images, error: imagesError } = await supabase
            .from('document_images')
            .select('*')
            .eq('document_id', id)
            .order('page_number', { ascending: true });

        if (imagesError) {
            return c.json({ error: 'Failed to fetch document images' }, 500);
        }

        if (!images || images.length === 0) {
            return c.json({
                error: 'No images found. Please extract images first using /documents/:id/extract-images'
            }, 400);
        }

        console.log(`Parsing document ${id} with ${images.length} images using Claude...`);

        // Download images, resize if needed, and convert to base64
        // (Claude API requires HTTPS URLs but local Supabase uses HTTP, so we send base64 data)
        // Claude also limits images to 2000px max dimension for multi-image requests
        const base64Images: { data: string; mediaType: 'image/png' | 'image/jpeg' }[] = [];

        for (const image of images) {
            try {
                // Download image from storage
                const { data: imageData, error: downloadError } = await supabase.storage
                    .from('project-documents')
                    .download(image.image_url);

                if (downloadError || !imageData) {
                    console.error(`Failed to download image ${image.id}:`, downloadError);
                    continue;
                }

                // Convert to base64
                const arrayBuffer = await imageData.arrayBuffer();
                const base64 = Buffer.from(arrayBuffer).toString('base64');
                const mediaType = image.image_url.endsWith('.png') ? 'image/png' : 'image/jpeg';

                // Resize if needed (Claude limits images to 2000px for multi-image requests)
                const resizedImage = await resizeImageIfNeeded(base64, mediaType);

                base64Images.push({ data: resizedImage.data, mediaType: resizedImage.mediaType });
                const sizeKB = (Buffer.from(resizedImage.data, 'base64').length / 1024).toFixed(1);
                console.log(`Loaded image ${image.page_number || 'unknown'} (${sizeKB} KB)${resizedImage.resized ? ' [resized]' : ''}`);
            } catch (err) {
                console.error(`Failed to process image ${image.id}:`, err);
            }
        }

        if (base64Images.length === 0) {
            return c.json({ error: 'Failed to load any images' }, 500);
        }

        console.log(`Sending ${base64Images.length} images to Claude for parsing...`);

        // Parse with Claude using base64 images
        // Use batch processing for large documents (> 10 pages) to avoid token limits
        const claudeParser = new ClaudePdfParser(apiKey);
        const BATCH_SIZE = 10; // Process 10 pages at a time
        const parsedData = base64Images.length > BATCH_SIZE
            ? await claudeParser.parseInBatches(base64Images, id, BATCH_SIZE, customPrompt)
            : await claudeParser.parseBase64Images(base64Images, id, customPrompt);

        // Store parsed data in document metadata
        const { error: updateError } = await supabase
            .from('project_documents')
            .update({
                metadata: {
                    ...document.metadata,
                    parsed_data: parsedData,
                    parsed_at: new Date().toISOString()
                }
            })
            .eq('id', id);

        if (updateError) {
            console.error('Failed to save parsed data:', updateError);
        }

        return c.json({
            success: true,
            document_id: id,
            parsed: parsedData,
            summary: {
                units_count: parsedData.units?.length ?? 0,
                total_baseboard_ft: parsedData.total_baseboard_ft ?? 0,
                trades: parsedData.summary?.trades ?? []
            }
        });
    } catch (error) {
        console.error('Error parsing document with Claude:', error);
        return c.json({
            error: error instanceof Error ? error.message : 'Failed to parse document'
        }, 500);
    }
});

    /**
     * Attempt to repair truncated JSON by closing open brackets/braces
     */
    private repairTruncatedJson(json: string): string {
        let repaired = json.trim();

        // Count open brackets and braces
        let openBraces = 0;
        let openBrackets = 0;
        let inString = false;
        let escapeNext = false;

        for (const char of repaired) {
            if (escapeNext) {
                escapeNext = false;
                continue;
            }
            if (char === '\\') {
                escapeNext = true;
                continue;
            }
            if (char === '"') {
                inString = !inString;
                continue;
            }
            if (inString) continue;

            if (char === '{') openBraces++;
            if (char === '}') openBraces--;
            if (char === '[') openBrackets++;
            if (char === ']') openBrackets--;
        }

        // If we ended inside a string, close it
        if (inString) {
            repaired += '"';
        }

        // Remove any trailing incomplete key-value pair
        repaired = repaired.replace(/,\s*"[^"]*"?\s*:?\s*[^,}\]]*$/, '');

        // Close open brackets and braces
        for (let i = 0; i < openBrackets; i++) {
            repaired += ']';
        }
        for (let i = 0; i < openBraces; i++) {
            repaired += '}';
        }

        return repaired;
    }

    /**
     * Try to extract partial units array from truncated JSON
     */
    private salvagePartialJson(text: string): any {
        // Try to find the units array
        const unitsMatch = text.match(/"units"\s*:\s*\[/);
        if (!unitsMatch) {
            return null;
        }

        // Find where units array starts
        const startIdx = text.indexOf(unitsMatch[0]) + unitsMatch[0].length - 1;
        let bracketCount = 1;
        let endIdx = startIdx + 1;
        let inString = false;
        let escapeNext = false;

        // Find complete unit objects
        const units: any[] = [];
        let currentUnitStart = startIdx + 1;

        while (endIdx < text.length && bracketCount > 0) {
            const char = text[endIdx];

            if (escapeNext) {
                escapeNext = false;
                endIdx++;
                continue;
            }
            if (char === '\\') {
                escapeNext = true;
                endIdx++;
                continue;
            }
            if (char === '"') {
                inString = !inString;
                endIdx++;
                continue;
            }
            if (inString) {
                endIdx++;
                continue;
            }

            if (char === '[') bracketCount++;
            if (char === ']') bracketCount--;
            if (char === '{' && bracketCount === 1) {
                currentUnitStart = endIdx;
            }
            if (char === '}' && bracketCount === 1) {
                // Found complete unit object
                try {
                    const unitJson = text.substring(currentUnitStart, endIdx + 1);
                    const unit = JSON.parse(unitJson);
                    units.push(unit);
                } catch (e) {
                    // Skip malformed unit
                }
            }

            endIdx++;
        }

        if (units.length === 0) {
            return null;
        }

        return {
            units,
            general_notes: [],
            total_baseboard_ft: 0,
            summary: {
                units_count: units.length,
                trades: [],
                total_doors: 0,
                total_windows: 0
            }
        };
    }

    /**
     * Parse a large document by splitting into batches
     * This avoids token limits by processing pages in smaller groups
     */
    async parseInBatches(
        images: { data: string; mediaType: 'image/png' | 'image/jpeg' | 'image/gif' | 'image/webp' }[],
        documentId: string,
        batchSize: number = 10,
        customPrompt?: string
    ): Promise<ParsedDocument> {
        // If small enough, just parse normally
        if (images.length <= batchSize) {
            return this.parseBase64Images(images, documentId, customPrompt);
        }

        console.log(`Parsing ${images.length} images in batches of ${batchSize}...`);

        // Split into batches
        const batches: typeof images[] = [];
        for (let i = 0; i < images.length; i += batchSize) {
            batches.push(images.slice(i, i + batchSize));
        }

        // Process each batch
        const batchResults: ParsedDocument[] = [];
        for (let i = 0; i < batches.length; i++) {
            console.log(`Processing batch ${i + 1}/${batches.length} (${batches[i].length} pages)...`);
            try {
                const result = await this.parseBase64Images(
                    batches[i],
                    `${documentId}_batch_${i}`,
                    customPrompt
                );
                batchResults.push(result);
                console.log(`Batch ${i + 1} complete: ${result.units.length} units found`);
            } catch (error) {
                console.error(`Batch ${i + 1} failed:`, error);
                // Continue with other batches
            }
        }

        // Merge results
        return this.mergeBatchResults(documentId, batchResults);
    }

    /**
     * Merge results from multiple batch parses into a single document
     */
    private mergeBatchResults(documentId: string, results: ParsedDocument[]): ParsedDocument {
        console.log(`Merging ${results.length} batch results...`);

        const allUnits: any[] = [];
        const allNotes: string[] = [];
        const tradesSet = new Set<string>();
        let totalBaseboard = 0;
        let totalDoors = 0;
        let totalWindows = 0;
        const rawTexts: string[] = [];

        for (const result of results) {
            // Merge units (deduplicate by unit_number)
            for (const unit of result.units || []) {
                const existingIdx = allUnits.findIndex(u => u.unit_number === unit.unit_number);
                if (existingIdx === -1) {
                    allUnits.push(unit);
                } else {
                    // Merge data into existing unit (later batches may have more complete data)
                    allUnits[existingIdx] = { ...allUnits[existingIdx], ...unit };
                }
            }

            // Merge notes (deduplicate)
            for (const note of result.general_notes || []) {
                if (!allNotes.includes(note)) {
                    allNotes.push(note);
                }
            }

            // Collect trades
            for (const trade of result.summary?.trades || []) {
                tradesSet.add(trade);
            }

            // Sum totals
            totalBaseboard += result.total_baseboard_ft || 0;
            totalDoors += result.summary?.total_doors || 0;
            totalWindows += result.summary?.total_windows || 0;

            // Collect raw text
            if (result.raw_text) {
                rawTexts.push(result.raw_text);
            }
        }

        // Recalculate baseboard from units if we have detailed room data
        const calculatedBaseboard = allUnits.reduce((sum, unit) => {
            const rooms = Array.isArray(unit.rooms) ? unit.rooms : [];
            const unitBaseboard = rooms.reduce((roomSum: number, room: any) => {
                return roomSum + (room.baseboard_length_ft || room.baseboard_ft || room.total_baseboard_ft || 0);
            }, 0);
            return sum + (unit.total_baseboard_ft || unitBaseboard);
        }, 0);

        return {
            document_id: documentId,
            units: allUnits,
            general_notes: allNotes,
            total_baseboard_ft: calculatedBaseboard || totalBaseboard,
            summary: {
                units_count: allUnits.length,
                trades: Array.from(tradesSet),
                total_doors: totalDoors,
                total_windows: totalWindows
            },
            raw_text: rawTexts.join('\n\n--- BATCH SEPARATOR ---\n\n')
        };
    }
}

/**
 * Helper to generate work orders from parsed document
 */
export function generateWorkOrdersFromParsedData(
    parsedData: ParsedDocument,
    projectId: string
): any[] {
    const workOrders: any[] = [];

    for (const unit of parsedData.units) {
        // Group upgrades by trade
        const upgradesByTrade = new Map<string, ParsedUpgrade[]>();

        for (const upgrade of unit.upgrades) {
            const trade = upgrade.trade.toLowerCase();
            if (!upgradesByTrade.has(trade)) {
                upgradesByTrade.set(trade, []);
            }
            upgradesByTrade.get(trade)!.push(upgrade);
        }

        // Create work order for each trade
        for (const [trade, upgrades] of upgradesByTrade) {
            workOrders.push({
                project_id: projectId,
                unit_number: unit.unit_number,
                trade,
                title: `${trade.charAt(0).toUpperCase() + trade.slice(1)} - Unit ${unit.unit_number}`,
                description: upgrades.map(u => u.description).join('\n'),
                tasks: upgrades.map(u => ({
                    description: u.description,
                    quantity: u.quantity,
                    unit: u.unit
                })),
                status: 'pending',
                metadata: {
                    unit_type: unit.unit_type,
                    floor: unit.floor,
                    area_sf: unit.area_sf,
                    baseboard_ft: unit.rooms.reduce((sum, r) => sum + r.baseboard_length_ft, 0)
                }
            });
        }
    }

    return workOrders;
