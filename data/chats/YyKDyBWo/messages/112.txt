import { useParams, A } from '@solidjs/router';
import { api } from '../lib/api';
import type { ProjectDocument, DocumentImage, ParsedDocumentResult } from '../lib/api';

const ProjectDocuments: Component = () => {
    const params = useParams();
    const [uploading, setUploading] = createSignal(false);
    const [extracting, setExtracting] = createSignal<string | null>(null);
    const [parsing, setParsing] = createSignal<string | null>(null);
    const [parsedResult, setParsedResult] = createSignal<ParsedDocumentResult | null>(null);
    const [selectedDocId, setSelectedDocId] = createSignal<string | null>(null);

    // Load project info
    const [project] = createResource(() =>
        params.id ? api.getProject(params.id) : Promise.reject('No project ID')
    );

    // Load documents
    const [documents, { refetch: refetchDocuments }] = createResource(() =>
        params.id ? api.getProjectDocuments(params.id) : Promise.reject('No project ID')
    );

    // Load images for selected document
    const [images, { refetch: refetchImages }] = createResource(
        selectedDocId,
        (docId) => api.getDocumentImages(docId)
    );

    const handleUpload = async (e: Event) => {
        const input = e.target as HTMLInputElement;
        const file = input.files?.[0];
        if (!file || !params.id) return;

        setUploading(true);
        try {
            await api.uploadDocument(params.id, file, file.name);
            refetchDocuments();
            input.value = '';
        } catch (error) {
            console.error('Upload failed:', error);
            alert(`Upload failed: ${error instanceof Error ? error.message : 'Unknown error'}`);
        } finally {
            setUploading(false);
        }
    };

    const extractImages = async (doc: ProjectDocument) => {
        setExtracting(doc.id);
        try {
            const result = await api.extractImagesFromDocument(doc.id);
            alert(`Successfully extracted ${result.images_extracted} images from ${doc.title}`);
            if (selectedDocId() === doc.id) {
                refetchImages();
            }
        } catch (error) {
            console.error('Image extraction failed:', error);
            alert(`Image extraction failed: ${error instanceof Error ? error.message : 'Unknown error'}`);
        } finally {
            setExtracting(null);
        }
    };

    const deleteDocument = async (doc: ProjectDocument) => {
        if (!confirm(`Delete "${doc.title}"? This will also delete all extracted images.`)) {
            return;
        }

        try {
            await api.deleteDocument(doc.id);
            refetchDocuments();
            if (selectedDocId() === doc.id) {
                setSelectedDocId(null);
            }
        } catch (error) {
            console.error('Delete failed:', error);
            alert(`Delete failed: ${error instanceof Error ? error.message : 'Unknown error'}`);
        }
    };

    const downloadDocument = async (doc: ProjectDocument) => {
        try {
            const result = await api.getDocumentDownloadUrl(doc.id);
            window.open(result.url, '_blank');
        } catch (error) {
            console.error('Download failed:', error);
            alert(`Download failed: ${error instanceof Error ? error.message : 'Unknown error'}`);
        }
    };

    const parseWithClaude = async (doc: ProjectDocument) => {
        setParsing(doc.id);
        setParsedResult(null);
        try {
            const result = await api.parseDocumentWithClaude(doc.id);
            setParsedResult(result);
            const unitsCount = result.summary?.units_count ?? result.parsed?.units?.length ?? 0;
            const baseboard = result.summary?.total_baseboard_ft ?? result.parsed?.total_baseboard_ft ?? 0;
            alert(`Successfully parsed document! Found ${unitsCount} units, ${baseboard} ft total baseboard.`);
        } catch (error) {
            console.error('Parsing failed:', error);
            alert(`Parsing failed: ${error instanceof Error ? error.message : 'Unknown error'}`);
        } finally {
            setParsing(null);
        }
    };

    const toggleImages = async (docId: string) => {
        if (selectedDocId() === docId) {
            setSelectedDocId(null);
        } else {
            setSelectedDocId(docId);
        }
    };

        if (bytes < 1024) return `${bytes} B`;
        if (bytes < 1024 * 1024) return `${(bytes / 1024).toFixed(2)} KB`;
        return `${(bytes / 1024 / 1024).toFixed(2)} MB`;
    };

    const getFileTypeIcon = (fileType: string): string => {
        if (fileType.includes('pdf')) return '📄';
        if (fileType.includes('image')) return '🖼️';
        if (fileType.includes('dwg') || fileType.includes('dxf')) return '📐';
        return '📎';

                        <span>/</span>
                        <Show when={project()} fallback={<span>...</span>}>
                            {(p) => (
                                <>
                                    <A href={`/projects/${p().id}`} class="hover:text-blue-600">
                                        {p().name}
                                    </A>
                                    <span>/</span>
                                    <span>Documents</span>
                                </>
                            )}
                        </Show>

                            file:mr-4 file:py-2 file:px-4
                            file:rounded-md file:border-0
                            file:text-sm file:font-semibold
                            file:bg-blue-50 file:text-blue-700
                            hover:file:bg-blue-100
                            disabled:opacity-50 disabled:cursor-not-allowed"
                    />
                    <Show when={uploading()}>
