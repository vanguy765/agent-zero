 167     job_id: str,
 168     temp_path: str,
 169     provider: str,
 170     job_manager: JobManager,
 171     sb_client=None,
 172     tenant_name: str = "",
 173     project_name: str = "",
 174 ) -> None:
 175     """Execute the hybrid pipeline: Docling decomposition -> Claude extraction.
 176
 177     This is the main background task for hybrid mode.
 178     Wrapped in timeout enforcement by the caller.
 179
 180     Phases:
 181     1. Docling Conversion - PDF to structured document
 182     2. Classification - Identify floorplans, tables, photos
 183     3. Image Extraction - Prepare floorplan images for Claude
 184     4. LLM Analysis - Claude analyzes each floorplan with rich prompt
 185     5. Synthesis - Cross-page correlation (Priority 5)
 186     6. Upload - Store images and results
 187     """
 188     try:
 189         # Transition to PROCESSING
 190         await job_manager.start_processing(job_id)
 191
 192         # ==================================================
 193         # PHASE 1: Docling Structural Decomposition
 194         # ==================================================
 195         await job_manager.update_progress(
 196             job_id,
 197             phase=ProcessingPhase.DOCLING_CONVERSION,
 198             current_step="converting_pdf_with_docling",
 199         )
 200
 201         from docling.document_converter import DocumentConverter, PdfFormatOption
 202         from docling.datamodel.base_models import InputFormat
 203         from docling.datamodel.pipeline_options import PdfPipelineOptions
 204         from pdf2image import convert_from_path
 205
 206         pipeline_options = PdfPipelineOptions()
 207         pipeline_options.generate_picture_images = True
 208         pipeline_options.generate_table_images = True
 209
 210         converter = DocumentConverter(
 211             format_options={
 212                 InputFormat.PDF: PdfFormatOption(
 213                     pipeline_options=pipeline_options
 214                 )
 215             }
 216         )
 217
 218         result_doc_wrapper = await asyncio.to_thread(
 219             converter.convert, temp_path
 220         )
 221         result_doc = result_doc_wrapper.document
 222
 223         # Generate full page images for reference
 224         pdf_pages = await asyncio.to_thread(convert_from_path, temp_path)
 225         total_pages = len(pdf_pages)
 226
 227         await job_manager.update_progress(
 228             job_id,
 229             total_pages=total_pages,
 230             current_step="docling_conversion_complete",
 231         )
 232
 233         logger.info(f"Job {job_id}: Docling converted {total_pages} pages")
 234
 235         # ==================================================
 236         # PHASE 2: Classification & Collection
 237         # ==================================================
 238         await job_manager.update_progress(
 239             job_id,
 240             phase=ProcessingPhase.CLASSIFICATION,
 241             current_step="classifying_pages",
 242         )
 243
 244         floorplan_candidates: List[Tuple[int, Image.Image, str]] = []  # (page, img, b64)
 245         table_data: List[Tuple[int, str]] = []  # (page, markdown)
 246         site_photos: List[Tuple[int, Image.Image]] = []
 247         diagrams: List[Tuple[int, Image.Image]] = []
 248         footer_sample: Optional[Dict[str, Any]] = None
 249
 250         # --- Collect tables from Docling's structured extraction ---
 251         for table in result_doc.tables:
 252             md_text = table.export_to_markdown()
 253             page_no = table.prov[0].page_no if table.prov else 1
 254
 255             # Check if this is a footer/title block table
 256             is_footer = sum(
 257                 1 for kw in FOOTER_KEYWORDS
 258                 if kw in md_text.lower()
 259             ) >= 3
 260
 261             if is_footer:
 262                 if not footer_sample:
 263                     footer_sample = {"markdown": md_text, "page_no": page_no}
 264                 continue
 265
 266             table_data.append((page_no, md_text))
 267
 268         await job_manager.update_progress(
 269             job_id,
 270             tables_found=len(table_data),
 271             current_step=f"found_{len(table_data)}_tables",
 272         )
 273
 274         # --- Classify pictures using improved heuristic (Priority 2) ---
 275         for pic in result_doc.pictures:
 276             if not pic.image:
 277                 continue
 278
 279             try:
 280                 pil_img = pic.image.pil_image
 281             except AttributeError:
 282                 continue
 283
 284             page_no = pic.prov[0].page_no if pic.prov else 1
 285
 286             await job_manager.update_progress(
 287                 job_id,
 288                 current_page=page_no,
 289                 current_step=f"classifying_image_page_{page_no}",
 290             )
 291
 292             classification = classify_image(pil_img)
 293
 294             if classification == "floorplan":
 295                 b64 = image_to_base64(pil_img)
 296                 floorplan_candidates.append((page_no, pil_img, b64))
 297             elif classification == "site_photo":
 298                 site_photos.append((page_no, pil_img))
 299             elif classification == "diagram":
 300                 diagrams.append((page_no, pil_img))
 301             # tables and title_blocks already handled by Docling
 302
 303         # --- Also check full page images for floorplans Docling may have missed ---
 304         docling_fp_pages = {fp[0] for fp in floorplan_candidates}
 305         for i, page_img in enumerate(pdf_pages):
 306             page_no = i + 1
 307             if page_no in docling_fp_pages:
 308                 continue  # Already found via Docling pictures
 309
 310             if is_architectural_drawing(page_img) and has_architectural_text(page_img):
 311                 b64 = image_to_base64(page_img)
 312                 floorplan_candidates.append((page_no, page_img, b64))
 313                 logger.info(f"Job {job_id}: Full-page floorplan detected on page {page_no}")
 314
 315         await job_manager.update_progress(
 316             job_id,
 317             floorplans_found=len(floorplan_candidates),
 318             current_step=f"classification_complete_{len(floorplan_candidates)}_floorplans",
 319         )
 320
 321         logger.info(
 322             f"Job {job_id}: Classification complete - "
 323             f"{len(floorplan_candidates)} floorplans, "
 324             f"{len(table_data)} tables, "
 325             f"{len(site_photos)} photos, "
 326             f"{len(diagrams)} diagrams"
 327         )
 328
 329         # ==================================================
 330         # PHASE 3: Image Extraction (prepare for Claude)
 331         # ==================================================
 332         await job_manager.update_progress(
 333             job_id,
 334             phase=ProcessingPhase.IMAGE_EXTRACTION,
 335             current_step="preparing_images_for_analysis",
 336         )
 337
 338         # Build table context string for enriching Claude prompts
 339         table_context = ""
 340         if table_data:
 341             table_context = "\n\n".join([
 342                 f"--- Table from Page {pg} ---\n{md}"
 343                 for pg, md in table_data
 344             ])
 345
 346         # ==================================================
 347         # PHASE 4: Claude Semantic Extraction
 348         # ==================================================
 349         await job_manager.update_progress(
 350             job_id,
 351             phase=ProcessingPhase.LLM_ANALYSIS,
 352             current_step="starting_claude_analysis",
 353         )
 354
 355         unit_extractions: List[Dict[str, Any]] = []
 356
 357         for i, (page_no, pil_img, b64) in enumerate(floorplan_candidates):
 358             await job_manager.update_progress(
 359                 job_id,
 360                 current_page=page_no,
 361                 current_step=f"analyzing_floorplan_{i + 1}_of_{len(floorplan_candidates)}",
 362             )
 363
 364             # Enrich prompt with table context
 365             enriched_prompt = FLOORPLAN_PROMPT_RICH
 366             if table_context:
 367                 enriched_prompt += (
 368                     "\n\nADDITIONAL CONTEXT - Door/window schedules and "
 369                     "other tables extracted from this document set. Use these "
 370                     "to enrich your extraction with schedule data (hardware, "
 371                     "materials, fire ratings, etc.):\n\n"
 372                     + table_context
 373                 )
 374
 375             try:
 376                 result_str = await run_llm_extraction(
 377                     provider, enriched_prompt, b64
 378                 )
 379
 380                 # Clean markdown code fences
 381                 result_clean = re.sub(
 382                     r"```(?:json)?\s*|\s*```", "", result_str
 383                 ).strip()
 384
 385                 unit_data = json.loads(result_clean)
 386                 unit_extractions.append({
 387                     "page_no": page_no,
 388                     "data": unit_data,
 389                     "success": True,
 390                 })
 391
 392                 logger.info(
 393                     f"Job {job_id}: Page {page_no} extracted - "
 394                     f"{len(unit_data.get('rooms', []))} rooms, "
 395                     f"{len(unit_data.get('doors', []))} doors"
 396                 )
 397
 398             except json.JSONDecodeError as e:
 399                 logger.warning(
 400                     f"Job {job_id}: JSON parse failed for page {page_no}: {e}"
 401                 )
 402                 unit_extractions.append({
 403                     "page_no": page_no,
 404                     "error": f"JSON parse error: {str(e)}",
 405                     "raw": result_str[:500] if result_str else "",
 406                     "success": False,
 407                 })
 408
 409             except Exception as e:
 410                 logger.error(
 411                     f"Job {job_id}: LLM extraction failed for page {page_no}: {e}"
 412                 )
 413                 unit_extractions.append({
 414                     "page_no": page_no,
 415                     "error": str(e),
 416                     "success": False,
 417                 })
 418
 419         # ==================================================
 420         # PHASE 5: Cross-Page Synthesis (Priority 5)
 421         # ==================================================
 422         synthesis_result = None
 423
 424         successful_extractions = [
 425             ue for ue in unit_extractions if ue.get("success")
 426         ]
 427
 428         if len(successful_extractions) > 1 or (
 429             successful_extractions and table_data
 430         ):
 431             await job_manager.update_progress(
 432                 job_id,
 433                 phase=ProcessingPhase.SYNTHESIS,
 434                 current_step="cross_page_synthesis",
 435             )
 436
 437             try:
 438                 # Build synthesis context
 439                 synthesis_context_parts = []
 440
 441                 synthesis_context_parts.append(
 442                     f"Total pages in document: {total_pages}"
 443                 )
 444                 synthesis_context_parts.append(
 445                     f"Floorplans analyzed: {len(successful_extractions)}"
 446                 )
 447
 448                 for ue in successful_extractions:
 449                     synthesis_context_parts.append(
 450                         f"\n--- Floorplan from Page {ue['page_no']} ---\n"
 451                         + json.dumps(ue["data"], indent=2)
 452                     )
 453
 454                 if table_data:
 455                     synthesis_context_parts.append(
 456                         "\n--- TABLES/SCHEDULES ---"
 457                     )
 458                     for pg, md in table_data:
 459                         synthesis_context_parts.append(
 460                             f"\n--- Table from Page {pg} ---\n{md}"
 461                         )
 462
 463                 synthesis_context = "\n".join(synthesis_context_parts)
 464
 465                 synthesis_str = await run_text_llm(
 466                     provider, SYNTHESIS_PROMPT, synthesis_context
 467                 )
 468
 469                 synthesis_clean = re.sub(
 470                     r"```(?:json)?\s*|\s*```", "", synthesis_str
 471                 ).strip()
 472
 473                 synthesis_result = json.loads(synthesis_clean)
 474                 logger.info(f"Job {job_id}: Cross-page synthesis complete")
 475
 476             except Exception as e:
 477                 logger.warning(
 478                     f"Job {job_id}: Synthesis failed (non-fatal): {e}"
 479                 )
 480                 synthesis_result = {"error": str(e)}
 481
 482         # ==================================================
 483         # PHASE 6: Assemble Output & Upload
 484         # ==================================================
 485         await job_manager.update_progress(
 486             job_id,
 487             phase=ProcessingPhase.UPLOADING,
 488             current_step="assembling_output",
 489         )
 490
 491         result = assemble_hybrid_output(
 492             unit_extractions=unit_extractions,
 493             table_data=table_data,
 494             footer_sample=footer_sample,
 495             floorplan_candidates=floorplan_candidates,
 496             site_photos=site_photos,
 497             diagrams=diagrams,
 498             pdf_pages=pdf_pages,
 499             synthesis=synthesis_result,
 500             sb_client=sb_client,
 501             tenant_name=tenant_name,
 502             project_name=project_name,
 503         )
 504
 505         # Calculate confidence (Priority 6)
 506         result["confidence"] = calculate_confidence(
 507             units=result.get("units", []),
 508             tables=result.get("tables", []),
 509             synthesis=synthesis_result,
 510             total_pages=total_pages,
 511             floorplans_found=len(floorplan_candidates),
 512         ).model_dump()
 513
 514         # Complete the job
 515         await job_manager.complete_job(job_id, result)
 516
 517         logger.info(
 518             f"Job {job_id}: Hybrid pipeline complete - "
 519             f"{len(result.get("units", []))} units extracted, "
 520             f"confidence={result["confidence"]["overall"]}"
 521         )
 522
 523     except asyncio.CancelledError:
 524         await job_manager.timeout_job(job_id)
 525         raise
 526
 527     except Exception as e:
 528         traceback.print_exc()
 529         await job_manager.fail_job(job_id, str(e))
 530
 531     finally:
 532         # Clean up temp file
 533         if os.path.exists(temp_path):
 534             try:
 535                 os.remove(temp_path)
 536             except OSError:
 537                 pass

 544 def assemble_hybrid_output(
 545     unit_extractions: List[Dict[str, Any]],
 546     table_data: List[Tuple[int, str]],
 547     footer_sample: Optional[Dict[str, Any]],
 548     floorplan_candidates: List[Tuple[int, Image.Image, str]],
 549     site_photos: List[Tuple[int, Image.Image]],
 550     diagrams: List[Tuple[int, Image.Image]],
 551     pdf_pages: list,
 552     synthesis: Optional[Dict[str, Any]] = None,
 553     sb_client=None,
 554     tenant_name: str = "",
 555     project_name: str = "",
 556 ) -> Dict[str, Any]:
 557     """Assemble hybrid pipeline output in both page-oriented and unit-oriented formats.
 558
 559     Produces BOTH formats:
 560     - pages: page-oriented (for doc-parser native consumers)
 561     - units: unit-oriented (for direct ParsedDocument compatibility with work-orders)
 562     """
 563     output: Dict[str, Any] = {
 564         "engine": "hybrid",
 565         "pages": {},
 566         "units": [],
 567         "tables": [],
 568         "footer_sample": footer_sample,
 569         "general_notes": [],
 570         "synthesis": synthesis,
 571     }
 572
 573     # --- Initialize page structure ---
 574     for i, page_img in enumerate(pdf_pages):
 575         page_no = i + 1
 576         output["pages"][page_no] = {
 577             "page_number": page_no,
 578             "floorplans": [],
 579             "tables": [],
 580             "site_photos": [],
 581             "diagrams": [],
 582             "classification": "unknown",
 583         }
 584
 585     # --- Map floorplan extractions to pages AND build units array ---
 586     for extraction in unit_extractions:
 587         page_no = extraction["page_no"]
 588         data = extraction.get("data", {})
 589
 590         if not extraction.get("success", False):
 591             # Record failed extraction in page data
 592             if page_no in output["pages"]:
 593                 output["pages"][page_no]["floorplans"].append({
 594                     "error": extraction.get("error", "unknown"),
 595                 })
 596             continue
 597
 598         # Add to page-oriented output
 599         if page_no in output["pages"]:
 600             output["pages"][page_no]["floorplans"].append(data)
 601             output["pages"][page_no]["classification"] = "floorplan"
 602
 603         # Build unit-oriented output (ParsedDocument compatible)
 604         unit = {
 605             "unit_number": data.get("unit_number", f"page_{page_no}"),
 606             "unit_type": data.get("unit_type", "unknown"),
 607             "floor": data.get("floor", 0),
 608             "bedrooms": data.get("bedrooms", 0),
 609             "bathrooms": data.get("bathrooms", 0),
 610             "half_baths": data.get("half_baths", 0),
 611             "area_sf": data.get("total_area_sf", data.get("area_sf", 0)),
 612             "rooms": data.get("rooms", []),
 613             "doors": data.get("doors", []),
 614             "windows": data.get("windows", []),
 615             "fixtures": data.get("fixtures", []),
 616             "upgrades": data.get("upgrades", []),
 617             "general_notes": data.get("general_notes", []),
 618             "source_page": page_no,
 619         }
 620         output["units"].append(unit)
 621
 622         # Collect general notes
 623         for note in data.get("general_notes", []):
 624             if note not in output["general_notes"]:
 625                 output["general_notes"].append(note)
 626
 627     # --- Add Docling tables to output ---
 628     for page_no, md in table_data:
 629         output["tables"].append({"page_no": page_no, "markdown": md})
 630         if page_no in output["pages"]:
 631             output["pages"][page_no]["tables"].append(md)
 632             if output["pages"][page_no]["classification"] == "unknown":
 633                 output["pages"][page_no]["classification"] = "table"
 634
 635     # --- Mark site photo and diagram pages ---
 636     for page_no, _ in site_photos:
 637         if page_no in output["pages"]:
 638             output["pages"][page_no]["classification"] = "site_photo"
 639
 640     for page_no, _ in diagrams:
 641         if page_no in output["pages"]:
 642             output["pages"][page_no]["classification"] = "diagram"
 643
 644     # --- Build summary ---
 645     output["summary"] = {
 646         "units_count": len(output["units"]),
 647         "pages_count": len(pdf_pages),
 648         "floorplans_count": len(floorplan_candidates),
 649         "tables_count": len(table_data),
 650         "site_photos_count": len(site_photos),
 651         "diagrams_count": len(diagrams),
 652         "total_doors": sum(
 653             len(u.get("doors", [])) for u in output["units"]
 654         ),
 655         "total_windows": sum(
 656             len(u.get("windows", [])) for u in output["units"]
 657         ),
 658         "total_baseboard_ft": sum(
 659             sum(
 660                 r.get("baseboard_length_ft", 0)
 661                 for r in u.get("rooms", [])
 662             )
 663             for u in output["units"]
 664         ),
 665         "trades": list(set(
 666             up.get("trade", "")
 667             for u in output["units"]
 668             for up in u.get("upgrades", [])
 669             if up.get("trade")
 670         )),
 671         "has_footer": footer_sample is not None,
 672         "has_synthesis": synthesis is not None and "error" not in (synthesis or {}),
 673         "extraction_errors": sum(
 674             1 for ue in unit_extractions if not ue.get("success")
 675         ),
 676     }
 677
 678     return output
