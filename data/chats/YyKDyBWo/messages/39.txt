/**
 * Submit a PDF for parsing with the specified engine.
 */
export async function submitParseJob(
  pdfBuffer: Buffer | ArrayBuffer,
  fileName: string,
  engine: 'docling' | 'hybrid',
  options: {
    provider?: string;
    tenantName?: string;
    projectName?: string;
    supabaseUrl?: string;
    supabaseKey?: string;
  } = {},
): Promise<DocParserJobResponse> {
  const endpoint = engine === 'hybrid' ? '/parse/hybrid' : '/parse/docling';

  const formData = new FormData();
  const blob = new Blob([pdfBuffer], { type: 'application/pdf' });
  formData.append('file', blob, fileName);

  if (options.provider) {
    formData.append('provider', options.provider);
  }
  if (options.tenantName) {
    formData.append('tenant_name', options.tenantName);
  }
  if (options.projectName) {
    formData.append('project_name', options.projectName);
  }
  if (options.supabaseUrl) {
    formData.append('supabase_url', options.supabaseUrl);
  }
  if (options.supabaseKey) {
    formData.append('supabase_key', options.supabaseKey);
  }

  const response = await fetch(`${DOC_PARSER_URL}${endpoint}`, {
    method: 'POST',
    body: formData,
  });

  if (!response.ok) {
    const errorText = await response.text();
    throw new Error(
      `Doc-parser submission failed (${response.status}): ${errorText}`,
    );
  }

  return response.json();
}

/**
 * Get lightweight job status (poll-safe).
 * NEVER includes result data.
 */
export async function getJobStatus(
  jobId: string,
): Promise<DocParserStatus> {
  const response = await fetch(`${DOC_PARSER_URL}/status/${jobId}`);

  if (!response.ok) {
    if (response.status === 404) {
      throw new Error(`Job ${jobId} not found`);
    }
    throw new Error(`Status check failed (${response.status})`);
  }

  return response.json();
}

/**
 * Get full extraction results (fetch-once after completed).
 * Heavy payload - only call when status confirms "completed".
 */
export async function getJobResults(
  jobId: string,
  summaryOnly: boolean = false,
): Promise<{ job_id: string; status: string; result?: DocParserResult; summary?: DocParserResultSummary }> {
  const url = summaryOnly
    ? `${DOC_PARSER_URL}/results/${jobId}?summary=true`
    : `${DOC_PARSER_URL}/results/${jobId}`;

  const response = await fetch(url);

  if (!response.ok) {
    if (response.status === 404) {
      throw new Error(`Results for job ${jobId} not found`);
    }
    throw new Error(`Results fetch failed (${response.status})`);
  }

  return response.json();
}

/**
 * Check if doc-parser service is healthy and available.
 */
export async function checkHealth(): Promise<{
  status: string;
  engines: string[];
  dependencies: Record<string, boolean>;
}> {
  try {
    const response = await fetch(`${DOC_PARSER_URL}/health`, {
      signal: AbortSignal.timeout(5000),
    });
    if (!response.ok) {
      return {
        status: 'unhealthy',
        engines: [],
        dependencies: {},
      };
    }
    return response.json();
  } catch {
    return {
      status: 'unreachable',
      engines: [],
      dependencies: {},
    };
  }
}
