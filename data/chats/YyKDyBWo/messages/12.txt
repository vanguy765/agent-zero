import Anthropic from '@anthropic-ai/sdk';

// Types for parsed PDF data
export interface ParsedUnit {
    unit_number: string;
    unit_type: string;
    floor: number;
    bedrooms: number;
    bathrooms: number;
    area_sf: number;
    rooms: ParsedRoom[];
    doors: ParsedDoor[];
    windows: ParsedWindow[];
    fixtures: ParsedFixture[];
    upgrades: ParsedUpgrade[];
}

export interface ParsedRoom {
    name: string;
    dimensions: {
        length_ft: number;
        width_ft: number;
        area_sf: number;
    };
    baseboard_length_ft: number;
    baseboard_exclusions: string[];
}

export interface ParsedDoor {
    location: string;
    type: string;
    dimensions: string;
    hardware: string;
    action: string;
    do_not_install?: boolean;
}

export interface ParsedWindow {
    location: string;
    type: string;
    dimensions: string;
    coverings: string;
}

export interface ParsedFixture {
    category: string;
    name: string;
    location: string;
    specifications: string;
}

export interface ParsedUpgrade {
    trade: string;
    description: string;
    quantity?: number;
    unit?: string;
}

export interface ParsedDocument {
    document_id: string;
    units: ParsedUnit[];
    general_notes: string[];
    total_baseboard_ft: number;
    summary: {
        units_count: number;
        trades: string[];
        total_doors: number;
        total_windows: number;
    };
    raw_text?: string;
}

    /**
     * Attempt to repair truncated JSON by closing open brackets/braces
     */
    private repairTruncatedJson(json: string): string {
        let repaired = json.trim();

        // Count open brackets and braces
        let openBraces = 0;
        let openBrackets = 0;
        let inString = false;
        let escapeNext = false;

        for (const char of repaired) {
            if (escapeNext) {
                escapeNext = false;
                continue;
            }
            if (char === '\\') {
                escapeNext = true;
                continue;
            }
            if (char === '"') {
                inString = !inString;
                continue;
            }
            if (inString) continue;

            if (char === '{') openBraces++;
            if (char === '}') openBraces--;
            if (char === '[') openBrackets++;
            if (char === ']') openBrackets--;
        }

        // If we ended inside a string, close it
        if (inString) {
            repaired += '"';
        }

        // Remove any trailing incomplete key-value pair
        repaired = repaired.replace(/,\s*"[^"]*"?\s*:?\s*[^,}\]]*$/, '');

        // Close open brackets and braces
        for (let i = 0; i < openBrackets; i++) {
            repaired += ']';
        }
        for (let i = 0; i < openBraces; i++) {
            repaired += '}';
        }

        return repaired;
    }

    /**
     * Try to extract partial units array from truncated JSON
     */
    private salvagePartialJson(text: string): any {
        // Try to find the units array
        const unitsMatch = text.match(/"units"\s*:\s*\[/);
        if (!unitsMatch) {
            return null;
        }

        // Find where units array starts
        const startIdx = text.indexOf(unitsMatch[0]) + unitsMatch[0].length - 1;
        let bracketCount = 1;
        let endIdx = startIdx + 1;
        let inString = false;
        let escapeNext = false;

        // Find complete unit objects
        const units: any[] = [];
        let currentUnitStart = startIdx + 1;

        while (endIdx < text.length && bracketCount > 0) {
            const char = text[endIdx];

            if (escapeNext) {
                escapeNext = false;
                endIdx++;
                continue;
            }
            if (char === '\\') {
                escapeNext = true;
                endIdx++;
                continue;
            }
            if (char === '"') {
                inString = !inString;
                endIdx++;
                continue;
            }
            if (inString) {
                endIdx++;
                continue;
            }

            if (char === '[') bracketCount++;
            if (char === ']') bracketCount--;
            if (char === '{' && bracketCount === 1) {
                currentUnitStart = endIdx;
            }
            if (char === '}' && bracketCount === 1) {
                // Found complete unit object
                try {
                    const unitJson = text.substring(currentUnitStart, endIdx + 1);
                    const unit = JSON.parse(unitJson);
                    units.push(unit);
                } catch (e) {
                    // Skip malformed unit
                }
            }

            endIdx++;
        }

        if (units.length === 0) {
            return null;
        }

        return {
            units,
            general_notes: [],
            total_baseboard_ft: 0,
            summary: {
                units_count: units.length,
                trades: [],
                total_doors: 0,
                total_windows: 0
            }
        };
    }

    /**
     * Parse a large document by splitting into batches
     * This avoids token limits by processing pages in smaller groups
     */
    async parseInBatches(
        images: { data: string; mediaType: 'image/png' | 'image/jpeg' | 'image/gif' | 'image/webp' }[],
        documentId: string,
        batchSize: number = 10,
        customPrompt?: string
    ): Promise<ParsedDocument> {
        // If small enough, just parse normally
        if (images.length <= batchSize) {
            return this.parseBase64Images(images, documentId, customPrompt);
        }

        console.log(`Parsing ${images.length} images in batches of ${batchSize}...`);

        // Split into batches
        const batches: typeof images[] = [];
        for (let i = 0; i < images.length; i += batchSize) {
            batches.push(images.slice(i, i + batchSize));
        }

        // Process each batch
        const batchResults: ParsedDocument[] = [];
        for (let i = 0; i < batches.length; i++) {
            console.log(`Processing batch ${i + 1}/${batches.length} (${batches[i].length} pages)...`);
            try {
                const result = await this.parseBase64Images(
                    batches[i],
                    `${documentId}_batch_${i}`,
                    customPrompt
                );
                batchResults.push(result);
                console.log(`Batch ${i + 1} complete: ${result.units.length} units found`);
            } catch (error) {
                console.error(`Batch ${i + 1} failed:`, error);
                // Continue with other batches
            }
        }

        // Merge results
        return this.mergeBatchResults(documentId, batchResults);
    }

    /**
     * Merge results from multiple batch parses into a single document
     */
    private mergeBatchResults(documentId: string, results: ParsedDocument[]): ParsedDocument {
        console.log(`Merging ${results.length} batch results...`);

        const allUnits: any[] = [];
        const allNotes: string[] = [];
        const tradesSet = new Set<string>();
        let totalBaseboard = 0;
        let totalDoors = 0;
        let totalWindows = 0;
        const rawTexts: string[] = [];

        for (const result of results) {
            // Merge units (deduplicate by unit_number)
            for (const unit of result.units || []) {
                const existingIdx = allUnits.findIndex(u => u.unit_number === unit.unit_number);
                if (existingIdx === -1) {
                    allUnits.push(unit);
                } else {
                    // Merge data into existing unit (later batches may have more complete data)
                    allUnits[existingIdx] = { ...allUnits[existingIdx], ...unit };
                }
            }

            // Merge notes (deduplicate)
            for (const note of result.general_notes || []) {
                if (!allNotes.includes(note)) {
                    allNotes.push(note);
                }
            }

            // Collect trades
            for (const trade of result.summary?.trades || []) {
                tradesSet.add(trade);
            }

            // Sum totals
            totalBaseboard += result.total_baseboard_ft || 0;
            totalDoors += result.summary?.total_doors || 0;
            totalWindows += result.summary?.total_windows || 0;

            // Collect raw text
            if (result.raw_text) {
                rawTexts.push(result.raw_text);
            }
        }

        // Recalculate baseboard from units if we have detailed room data
        const calculatedBaseboard = allUnits.reduce((sum, unit) => {
            const rooms = Array.isArray(unit.rooms) ? unit.rooms : [];
            const unitBaseboard = rooms.reduce((roomSum: number, room: any) => {
                return roomSum + (room.baseboard_length_ft || room.baseboard_ft || room.total_baseboard_ft || 0);
            }, 0);
            return sum + (unit.total_baseboard_ft || unitBaseboard);
        }, 0);

        return {
            document_id: documentId,
            units: allUnits,
            general_notes: allNotes,
            total_baseboard_ft: calculatedBaseboard || totalBaseboard,
            summary: {
                units_count: allUnits.length,
                trades: Array.from(tradesSet),
                total_doors: totalDoors,
                total_windows: totalWindows
            },
            raw_text: rawTexts.join('\n\n--- BATCH SEPARATOR ---\n\n')
        };
    }
}

/**
 * Helper to generate work orders from parsed document
 */
export function generateWorkOrdersFromParsedData(
    parsedData: ParsedDocument,
    projectId: string
): any[] {
    const workOrders: any[] = [];

    for (const unit of parsedData.units) {
        // Group upgrades by trade
        const upgradesByTrade = new Map<string, ParsedUpgrade[]>();

        for (const upgrade of unit.upgrades) {
            const trade = upgrade.trade.toLowerCase();
            if (!upgradesByTrade.has(trade)) {
                upgradesByTrade.set(trade, []);
            }
            upgradesByTrade.get(trade)!.push(upgrade);
        }

        // Create work order for each trade
        for (const [trade, upgrades] of upgradesByTrade) {
            workOrders.push({
                project_id: projectId,
                unit_number: unit.unit_number,
                trade,
                title: `${trade.charAt(0).toUpperCase() + trade.slice(1)} - Unit ${unit.unit_number}`,
                description: upgrades.map(u => u.description).join('\n'),
                tasks: upgrades.map(u => ({
                    description: u.description,
                    quantity: u.quantity,
                    unit: u.unit
                })),
                status: 'pending',
                metadata: {
                    unit_type: unit.unit_type,
                    floor: unit.floor,
                    area_sf: unit.area_sf,
                    baseboard_ft: unit.rooms.reduce((sum, r) => sum + r.baseboard_length_ft, 0)
                }
            });
        }
    }

    return workOrders;
}
