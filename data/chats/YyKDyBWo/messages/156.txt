import { createClient } from '@supabase/supabase-js';
import { SUPABASE_URL, SUPABASE_ANON_KEY } from '@repo/config';
import { logger } from '@repo/logger';

// Initialize Supabase Client (for verifying tokens)
// Note: In a real app, you might use the Admin key or just the JWT verification capabilities.
// For simple token verification, the standard client works if we just pass the access_token.
const supabase = createClient(SUPABASE_URL, SUPABASE_ANON_KEY);

export const authMiddleware = createMiddleware(async (c, next) => {
  const authHeader = c.req.header('Authorization');

  if (!authHeader) {
    logger.warn('Missing Authorization header');
    return c.json({ error: 'Unauthorized', message: 'Missing Authorization header' }, 401);
  }

  const token = authHeader.replace('Bearer ', '');
  const { data: { user }, error } = await supabase.auth.getUser(token);

  if (error || !user) {
    logger.error({ err: error }, 'Invalid token');
    return c.json({ error: 'Unauthorized', message: 'Invalid token' }, 401);
  }

  // Attach user to context
  c.set('user', user);
  
  await next();
});
