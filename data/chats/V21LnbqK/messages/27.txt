async def upload_image_to_storage(
    pil_image: Image.Image,
    sb_client,
    tenant_name: str,
    project_name: str,
    filename: str,
    page_no: int = 1,
    image_type: str = "doc-parser_extraction",
) -> Optional[str]:
    """Upload image to Supabase Storage and return public URL."""
    if not sb_client:
        return None
    try:
        buffered = BytesIO()
        pil_image.save(buffered, format="PNG")
        image_bytes = buffered.getvalue()
        path = f"{tenant_name}/{project_name}/doc-parser/{filename}"
        upload_resp = await asyncio.to_thread(
            lambda: sb_client.storage
            .from_("project-documents")
            .upload(path, image_bytes, {"contentType": "image/png", "upsert": "true"})
        )
        if isinstance(upload_resp, dict) and upload_resp.get("error"):
            raise Exception(f"Storage upload error: {upload_resp['error']}")
        url_response = path  # Return storage path; frontend constructs authenticated URL
        doc_id = project_name if _UUID_RE.match(project_name or '') else None
        try:
            await asyncio.to_thread(
                lambda: sb_client.table("document_images").insert({
                    "document_id": doc_id,
                    "image_url": path,
                    "page_number": page_no,
                    "image_type": image_type,
                    "metadata": {
                        "type": "png", "source": "doc-parser",
                        "width": pil_image.width, "height": pil_image.height
                    }
                }).execute()
            )
        except Exception as db_err:
            logger.warning(f"DB record insert failed for {filename}: {db_err}")
        return url_response
    except Exception as e:
        logger.warning(f"Image upload failed for {filename}: {e}")
        return None


async def upload_table_to_storage(
    markdown_content: str,
    sb_client,
    tenant_name: str,
    project_name: str,
    filename: str,
    page_no: int = 1,
    table_type: str = "table",
) -> Optional[str]:
    """Upload table markdown to Supabase Storage."""
    if not sb_client:
        return None
    try:
        content_bytes = markdown_content.encode("utf-8")
        path = f"{tenant_name}/{project_name}/doc-parser/{filename}"
        upload_resp = await asyncio.to_thread(
            lambda: sb_client.storage
            .from_("project-documents")
            .upload(path, content_bytes, {"contentType": "text/markdown", "upsert": "true"})
        )
        # supabase-py 2.x raises StorageApiError on failure, but some versions return
        # an error dict instead — check explicitly so the path is never returned on failure.
        if isinstance(upload_resp, dict) and upload_resp.get("error"):
            raise Exception(f"Storage upload error: {upload_resp['error']}")
        url_response = path  # Return storage path; frontend constructs authenticated URL
        doc_id = project_name if _UUID_RE.match(project_name or '') else None
        try:
            await asyncio.to_thread(
                lambda: sb_client.table("document_images").insert({
                    "document_id": doc_id,
                    "image_url": path,
                    "page_number": page_no,
                    "image_type": table_type,
                    "metadata": {"type": "markdown", "source": "doc-parser"},
                }).execute()
            )
        except Exception as db_err:
            logger.warning(f"DB record insert failed for {filename}: {db_err}")
        logger.info(f"Table markdown uploaded: {path}")
        return url_response
    except Exception as e:
        logger.warning(f"Table upload failed for {filename}: {e}")
        return None
