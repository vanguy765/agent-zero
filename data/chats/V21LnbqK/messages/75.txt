=== dev-start.sh ===
#!/bin/bash
# Start all container-side dev services
# Windows-side services (Supabase Kong :54321, Studio :54323) must be started
# separately on the host with: supabase start
cd "$(dirname "$0")/.."

echo "=== Starting Dev Stack ==="
echo "(Windows Supabase must already be running on host)"
echo ""

# ── 1. Fallback infrastructure (only needed if Windows Supabase is down) ──────
echo "[infra] Checking fallback PostgreSQL..."
if pg_isready -h 127.0.0.1 -p 54322 -U postgres -q 2>/dev/null; then
  echo "  ✅ PostgreSQL already running on :54322"
else
  pg_ctlcluster 18 main start 2>/dev/null || \
  su -c "/usr/lib/postgresql/18/bin/pg_ctl start -D /tmp/pgdata" postgres 2>/dev/null || \
  echo "  ⚠️  PostgreSQL not started — fallback unavailable"
fi

echo "[infra] Checking PostgREST proxy (fallback :54330)..."
if curl -sf --connect-timeout 2 http://localhost:54330/rest/v1/ > /dev/null 2>&1; then
  echo "  ✅ PostgREST proxy already running on :54330"
else
  nohup node postgrest-proxy.js > /tmp/postgrest-proxy.log 2>&1 &
  sleep 1
  curl -sf --connect-timeout 2 http://localhost:54330/rest/v1/ > /dev/null 2>&1 && \
    echo "  ✅ PostgREST proxy started on :54330" || \
    echo "  ⚠️  PostgREST proxy failed — check /tmp/postgrest-proxy.log"
fi

echo ""

# ── helper: start a pnpm app if not already running ───────────────────────────
start_app() {
  local name=$1   # pnpm filter name
  local port=$2   # health-check port
  local log=$3    # log file
  if curl -sf --connect-timeout 2 http://localhost:${port}/ > /dev/null 2>&1; then
    echo "  ✅ ${name} already running on :${port}"
  else
    echo "  Starting ${name} on :${port}..."
    nohup pnpm --filter "${name}" dev > "${log}" 2>&1 &
    sleep 4
    curl -sf --connect-timeout 3 http://localhost:${port}/ > /dev/null 2>&1 && \
      echo "  ✅ ${name} started" || \
      echo "  ⚠️  ${name} still starting — check ${log}"
  fi
}

# ── 2. API ────────────────────────────────────────────────────────────────────
echo "[1/7] API (Hono) :3000"
if curl -sf --connect-timeout 3 http://localhost:3000/health > /dev/null 2>&1; then
  echo "  ✅ API already running on :3000"
else
  echo "  Starting API..."
  nohup pnpm --filter api dev > /tmp/api.log 2>&1 &
  sleep 5
  curl -sf --connect-timeout 3 http://localhost:3000/health > /dev/null 2>&1 && \
    echo "  ✅ API started" || \
    echo "  ⚠️  API still starting — check /tmp/api.log"
fi

# ── 3. Frontend apps ──────────────────────────────────────────────────────────
echo "[2/7] Web Dashboard :3001"
start_app "web-dashboard"  3001 /tmp/web-dashboard.log

echo "[3/7] Site Companion :3002"
start_app "site-companion" 3002 /tmp/site-companion.log

echo "[4/7] Material Admin :3003"
start_app "material-admin" 3003 /tmp/material-admin.log

echo "[5/7] Work Orders :3004"
start_app "work-orders"    3004 /tmp/work-orders.log

echo "[6/7] Tenant Portal :3005"
start_app "tenant-portal"  3005 /tmp/tenant-portal.log

# ── 4. Doc Parser ─────────────────────────────────────────────────────────────
echo "[7/7] Doc Parser :5000"
if curl -sf --connect-timeout 3 http://localhost:5000/health > /dev/null 2>&1 || \
   curl -sf --connect-timeout 3 http://localhost:5000/ > /dev/null 2>&1; then
  echo "  ✅ Doc Parser already running on :5000"
else
  echo "  Starting Doc Parser..."
  nohup bash -c 'cd doc-parser && source ../venv/bin/activate && python server.py' \
    > /tmp/doc-parser.log 2>&1 &
  sleep 4
  curl -sf --connect-timeout 3 http://localhost:5000/ > /dev/null 2>&1 && \
    echo "  ✅ Doc Parser started" || \
    echo "  ⚠️  Doc Parser still starting — check /tmp/doc-parser.log"
fi

echo ""
echo "=== Stack Status ==="
