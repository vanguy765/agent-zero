    43	/**
    44	 * Infer a page category from the raw page data returned by doc-parser.
    45	 */
    46	function inferPageCategory(pd: any): string {
    47	  if (pd?.has_floorplan || pd?.is_floorplan)                          return 'floorplan';
    48	  if (pd?.has_table || (pd?.table_count ?? 0) > 0)                   return 'table';
    49	  if (pd?.is_site_photo)                                              return 'site_photo';
    50	  if (pd?.is_diagram)                                                 return 'diagram';
    51	  if (pd?.has_text && (pd?.has_floorplan || pd?.has_table))          return 'mixed';
    52	  return 'text';
    53	}

    57	 *
    58	 * Downloads the PDF from Supabase storage, submits it to doc-parser,
    59	 * polls until complete, then upserts document_pages + ocr_results.
    60	 *
    61	 * Returns a summary object ready to send as a JSON response.
    62	 */
    63	async function executePipeline(
    64	  documentId: string,
    65	  engine: 'docling' | 'hybrid',
    66	  options: {
    67	    pollIntervalMs?: number;
    68	    timeoutMs?: number;
    69	    tenantName?: string;
    70	    projectName?: string;
    71	  } = {},
    72	): Promise<{
    73	  document_id: string;
    74	  job_id: string;
    75	  engine: string;
    76	  status: string;
    77	  pages_persisted: number;
    78	  ocr_tables_persisted: number;
    79	  summary: any;
    80	  poll_url: string;
    81	  results_url: string;
    82	}> {
    83	  const pollIntervalMs = options.pollIntervalMs ?? 3000;
    84	  const timeoutMs      = options.timeoutMs      ?? 600_000;
    85
    86	  const serviceClient = createClient(SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY);
    87

    89	  const { data: document, error: fetchError } = await serviceClient
    90	    .from('project_documents')
    91	    .select('*')
    92	    .eq('id', documentId)
    93	    .single();

    97

    99	  const { data: fileData, error: downloadError } = await serviceClient.storage
   100	    .from('project-documents')
   101	    .download(document.file_url);
   102
   103	  if (downloadError || !fileData) {
   104	    throw Object.assign(
   105	      new Error(`Storage download failed: ${downloadError?.message ?? 'unknown'}`),
   106	      { status: 500 },
   107	    );
   108	  }
   109
   110	  const pdfBuffer = await fileData.arrayBuffer();
   111
   112	  // 3. Submit to doc-parser
   113	  const jobResponse: DocParserJobResponse = await submitParseJob(
   114	    pdfBuffer,
   115	    document.title || `document-${documentId}.pdf`,
   116	    engine,
   117	    {
   118	      supabaseUrl:  process.env.SUPABASE_URL,
   119	      supabaseKey:  process.env.SUPABASE_SERVICE_KEY,
   120	      projectName:  options.projectName || documentId,
   121	      tenantName:   options.tenantName,

   130	  let status: DocParserStatus;

   137	      );
   138	    }

   142	      `[pipeline] Job ${jobId} status=${status.status}` +
   143	      ` (${status.progress?.percent_complete ?? 0}%)`,

   146	    if (status.status === 'completed') break;
   147	    if (status.status === 'failed' || status.status === 'timeout') {
   148	      throw Object.assign(
   149	        new Error(`Doc-parser job ${status.status}: ${status.error ?? 'no details'}`),
   150	        { status: 502, job_id: jobId },
   151	      );
   152	    }

   154	    await new Promise((resolve) => setTimeout(resolve, pollIntervalMs));
   155	  }
   156

   158	  const rawResults = await getJobResults(jobId, false);
   159	  const result = rawResults.result;
   160	  if (!result) {
   161	    throw Object.assign(
   162	      new Error('Job completed but result payload is empty'),
   163	      { status: 502, job_id: jobId },
   164	    );
   165	  }

   183	          category:      inferPageCategory(pd),
   184	          raw_text:      pd?.raw_text ?? pd?.text ?? null,
   185	          metadata:      pd ?? null,
   186	        },
   187	        { onConflict: 'document_id,page_number', ignoreDuplicates: false },
   188	      )
   189	      .select('id')
   190	      .single();
   191
   192	    if (pageErr) {
   193	      console.error(`[pipeline] document_pages upsert failed for page ${pageNumber}:`, pageErr.message);
   194	    } else if (pageRow) {
   195	      pageIdMap[pageNumber] = pageRow.id;
   196	    }
   197	  }
   198
   199	  console.log(`[pipeline] Persisted ${Object.keys(pageIdMap).length}/${pageEntries.length} pages`);
   200
   201	  // 7. Persist ocr_results
   202	  const tables: any[] = result.tables || [];
   203	  let ocrCount = 0;
   204
   205	  for (let tableIndex = 0; tableIndex < tables.length; tableIndex++) {
   206	    const tbl = tables[tableIndex] as any;
   207	    const pageNumber: number | null = tbl?.page_number ?? tbl?.source_page ?? null;
   208	    const pageId = pageNumber !== null ? (pageIdMap[pageNumber] ?? null) : null;
   209
   210	    const { error: ocrErr } = await serviceClient
   211	      .from('ocr_results')
   212	      .insert({
   213	        document_id:     documentId,
   214	        page_id:         pageId,
   215	        table_index:     tableIndex,
   216	        markdown:        tbl?.markdown ?? tbl?.content ?? null,
   217	        structured_data: tbl?.structured_data ?? tbl?.data ?? tbl ?? null,
   218	        confidence:      tbl?.confidence ?? null,
   219	      });
   220
   221	    if (ocrErr) {
   222	      console.error(`[pipeline] ocr_results insert failed for table ${tableIndex}:`, ocrErr.message);
   223	    } else {
   224	      ocrCount++;
   225	    }
   226	  }
   227
   228	  console.log(`[pipeline] Persisted ${ocrCount}/${tables.length} OCR table rows`);
   229
   230	  return {
   231	    document_id:          documentId,
   232	    job_id:               jobId,
   233	    engine,
   234	    status:               'completed',
   235	    pages_persisted:      Object.keys(pageIdMap).length,
   236	    ocr_tables_persisted: ocrCount,
   237	    summary:              result.summary ?? null,
   238	    poll_url:             `/api/doc-parser/status/${jobId}`,
   239	    results_url:          `/api/doc-parser/results/${jobId}`,
   240	  };
   241	}

   457	  const { data: fileData, error: downloadError } = await serviceClient.storage

   459	    .download(document.file_url);
   460
   461	  if (downloadError || !fileData) {

   520	    }
   521
   522	    try {
   523	      status = await getJobStatus(jobId);
   524	    } catch (err) {
   525	      const msg = err instanceof Error ? err.message : 'Unknown error';
