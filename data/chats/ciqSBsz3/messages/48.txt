class ApiClient {
  private baseUrl: string;

  constructor(baseUrl: string) {
    this.baseUrl = baseUrl;
  }

  private async request<T>(endpoint: string, options?: RequestInit): Promise<T> {
    const { data: { session } } = await supabase.auth.getSession();
    const response = await fetch(`${this.baseUrl}${endpoint}`, {
      ...options,
      headers: {
        'Content-Type': 'application/json',
        ...(session?.access_token ? { Authorization: `Bearer ${session.access_token}` } : {}),
        ...options?.headers,
      },
    });

    if (!response.ok) {
      const error = await response.text();
      throw new Error(`API Error: ${response.status} - ${error}`);
    }

    return response.json();
  }

  // Units
  async getUnits(): Promise<Unit[]> {
    return this.request<Unit[]>('/units');
  }

  async getUnit(id: string): Promise<Unit> {
    return this.request<Unit>(`/units/${id}`);
  }

  async updateUnit(id: string, data: Partial<Unit>): Promise<Unit> {
    return this.request<Unit>(`/units/${id}`, {
      method: 'PATCH',
      body: JSON.stringify(data),
    });
  }

  // Templates
  async getTemplates(scope?: 'app' | 'company' | 'project'): Promise<WorkOrderTemplate[]> {
    const query = scope ? `?scope=${scope}` : '';
    return this.request<WorkOrderTemplate[]>(`/templates${query}`);
  }

  async getTemplate(id: string): Promise<WorkOrderTemplate> {
    return this.request<WorkOrderTemplate>(`/templates/${id}`);
  }

  async createTemplate(data: Partial<WorkOrderTemplate>): Promise<WorkOrderTemplate> {
    return this.request<WorkOrderTemplate>('/templates', {
      method: 'POST',
      body: JSON.stringify(data),
    });
  }

  async updateTemplate(id: string, data: Partial<WorkOrderTemplate>): Promise<WorkOrderTemplate> {
    return this.request<WorkOrderTemplate>(`/templates/${id}`, {
      method: 'PATCH',
      body: JSON.stringify(data),
    });
  }

  async deleteTemplate(id: string): Promise<{ success: boolean }> {
    return this.request(`/templates/${id}`, {
      method: 'DELETE',
    });
  }

  // Template Inheritance / Assignment
  async getAppTemplates(): Promise<WorkOrderTemplate[]> {
    return this.request<WorkOrderTemplate[]>('/templates/app');
  }

  async getCompanyTemplates(): Promise<WorkOrderTemplate[]> {
    return this.request<WorkOrderTemplate[]>('/templates/company');
  }

  async getProjectTemplates(projectId: string): Promise<WorkOrderTemplate[]> {
    return this.request<WorkOrderTemplate[]>(`/templates/project/${projectId}`);
  }

  async assignTemplateToCompany(appTemplateId: string, organizationId: string): Promise<WorkOrderTemplate> {
    return this.request<WorkOrderTemplate>('/templates/assign-to-company', {
      method: 'POST',
      body: JSON.stringify({ app_template_id: appTemplateId, organization_id: organizationId }),
    });
  }

  async assignTemplateToProject(companyTemplateId: string, projectId: string, organizationId: string): Promise<WorkOrderTemplate> {
    return this.request<WorkOrderTemplate>('/templates/assign-to-project', {
      method: 'POST',
      body: JSON.stringify({ company_template_id: companyTemplateId, project_id: projectId, organization_id: organizationId }),
    });
  }

  async unassignTemplateFromProject(templateId: string): Promise<{ success: boolean }> {
    return this.request(`/templates/${templateId}/unassign`, {
      method: 'DELETE',
    });
  }

  // Template Visibility Preferences
  async getHiddenTemplateIds(): Promise<string[]> {
    return this.request<string[]>('/templates/prefs/hidden');
  }

  async setTemplateHidden(templateId: string, isHidden: boolean): Promise<{ success: boolean }> {
    return this.request(`/templates/prefs/${templateId}/hidden`, {
      method: 'PUT',
      body: JSON.stringify({ is_hidden: isHidden }),
    });
  }

  // Work Orders
  async getWorkOrders(): Promise<WorkOrder[]> {
    return this.request<WorkOrder[]>('/work-orders');
  }

  async getWorkOrder(id: string): Promise<WorkOrder> {
    return this.request<WorkOrder>(`/work-orders/${id}`);
  }

  async createWorkOrder(data: { unit_id: string; template_id: string }): Promise<WorkOrder> {
    return this.request<WorkOrder>('/work-orders', {

  async getProjects(): Promise<WOProject[]> {
    return this.request<WOProject[]>('/projects');
  }

  async getProject(id: string): Promise<WOProject> {
    return this.request<WOProject>(`/projects/${id}`);
  }

  async createProject(data: Partial<WOProject>): Promise<WOProject> {
    return this.request<WOProject>('/projects', {

  async getTemplateLibraries(params?: {
    project_type?: string;
    scope?: 'app' | 'company';
  }): Promise<TemplateLibrary[]> {
    const query = params ? new URLSearchParams(params as any).toString() : '';
    return this.request<TemplateLibrary[]>(
      `/template-libraries${query ? `?${query}` : ''}`
    );
  }

  async getTemplateLibrary(id: string): Promise<TemplateLibrary> {
    return this.request<TemplateLibrary>(`/template-libraries/${id}`);
  }

  async getDefaultLibrary(projectType: string): Promise<TemplateLibrary> {
    return this.request<TemplateLibrary>(
      `/template-libraries/default/${projectType}`
    );
  }

  // Documents
  async getProjectDocuments(projectId: string): Promise<ProjectDocument[]> {
    return this.request<ProjectDocument[]>(`/projects/${projectId}/documents`);
  }

  async uploadDocument(projectId: string, file: File, title?: string, description?: string): Promise<ProjectDocument> {
    const formData = new FormData();
    formData.append('file', file);
    if (title) formData.append('title', title);
    if (description) formData.append('description', description);

    const response = await fetch(`${this.baseUrl}/projects/${projectId}/documents`, {
      method: 'POST',
      body: formData,
    });

    if (!response.ok) {
      const error = await response.text();
      throw new Error(`API Error: ${response.status} - ${error}`);
    }

    return response.json();
  }

  async getDocument(documentId: string): Promise<ProjectDocument> {
    return this.request<ProjectDocument>(`/documents/${documentId}`);
  }

  async updateDocument(documentId: string, data: Partial<ProjectDocument>): Promise<ProjectDocument> {
    return this.request<ProjectDocument>(`/documents/${documentId}`, {

    });
  }

  async deleteDocument(documentId: string): Promise<{ success: boolean }> {
    return this.request(`/documents/${documentId}`, {
      method: 'DELETE',
    });
  }

  async getDocumentDownloadUrl(documentId: string): Promise<{ url: string; filename: string; expires_at: string }> {
    return this.request(`/documents/${documentId}/download`);
  }

  // Image Extraction
  async extractImagesFromDocument(documentId: string): Promise<{ success: boolean; images_extracted: number; images: DocumentImage[] }> {
    return this.request(`/documents/${documentId}/extract-images`, {
      method: 'POST',
    });
  }

  async getDocumentImages(documentId: string): Promise<DocumentImage[]> {
    return this.request<DocumentImage[]>(`/documents/${documentId}/images`);
  }

  async getImage(imageId: string): Promise<DocumentImage> {
    return this.request<DocumentImage>(`/images/${imageId}`);
  }

  async updateImage(imageId: string, data: Partial<DocumentImage>): Promise<DocumentImage> {
    return this.request<DocumentImage>(`/images/${imageId}`, {
      method: 'PATCH',

  }

  async deleteImage(imageId: string): Promise<{ success: boolean }> {
    return this.request(`/images/${imageId}`, {
      method: 'DELETE',
    });
  }

  // Text Extraction
  async extractTextFromDocument(documentId: string): Promise<{ success: boolean; page_count: number; text: string; pages: string[] }> {
    return this.request(`/documents/${documentId}/extract-text`, {
      method: 'POST',
    });
  }

  // Claude AI Parsing
  async parseDocumentWithClaude(documentId: string, customPrompt?: string): Promise<ParsedDocumentResult> {
    return this.request(`/documents/${documentId}/parse`, {
      method: 'POST',
      body: customPrompt ? JSON.stringify({ customPrompt }) : undefined,
    });
  }

  async generateWorkOrdersFromDocument(documentId: string): Promise<{ success: boolean; work_orders_count: number; work_orders: any[] }> {
    return this.request(`/documents/${documentId}/generate-work-orders`, {
      method: 'POST',
    });
  }

  // ============================================================
  // Doc-Parser Integration (P1-P7)
  // ============================================================

  /** Submit a document for parsing via doc-parser service */
  async submitDocParserJob(
    documentId: string,
    engine: 'claude-direct' | 'hybrid' = 'hybrid'
  ): Promise<{ job_id: string; status: string; engine: string }> {
    return this.request(`/doc-parser/parse`, {
      method: 'POST',
      body: JSON.stringify({ document_id: documentId, engine }),
    });
  }

  /** Poll job status (lightweight endpoint) */
  async getDocParserStatus(jobId: string): Promise<{
    job_id: string;
    status: 'queued' | 'processing' | 'completed' | 'failed' | 'timeout';
    engine: string;
    file_name: string;
    progress: {
      current_phase: string;
      phases: string[];
      current_page: number;
      total_pages: number;
      current_step: string;
      floorplans_found: number;
      tables_found: number;
      percent_complete: number;
    };
    error?: string;
    created_at: string;
    updated_at: string;
  }> {
    return this.request(`/doc-parser/status/${jobId}`);
  }

  /** Fetch full results (heavy endpoint, call once after completed) */
  async getDocParserResults(jobId: string, summary?: boolean): Promise<any> {
    const query = summary ? '?summary=true' : '';
    return this.request(`/doc-parser/results/${jobId}${query}`);
  }

  /** List all doc-parser jobs */
  async listDocParserJobs(status?: string): Promise<any[]> {
    const query = status ? `?status=${status}` : '';
    return this.request(`/doc-parser/jobs${query}`);
  }

  /** Delete a doc-parser job */
  async deleteDocParserJob(jobId: string): Promise<{ status: string }> {
    return this.request(`/doc-parser/jobs/${jobId}`, {
      method: 'DELETE',
    });
  }

  /** Check doc-parser service health */
  async getDocParserHealth(): Promise<{
    status: string;
    version: string;
    engines: string[];
  }> {
    return this.request('/doc-parser/health');
  }

}
