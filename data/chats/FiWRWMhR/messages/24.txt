    90	def main(argv: Optional[List[str]] = None) -> int:
    91	    """Main entry point for the CLI.
    92
    93	    Args:
    94	        argv: Command-line arguments (defaults to sys.argv[1:]).
    95
    96	    Returns:
    97	        Exit code: 0=clean, 1=violations found, 2=error.
    98	    """
    99	    parser = build_parser()
   100	    args = parser.parse_args(argv)
   101
   102	    # Configure logging
   103	    log_level = logging.DEBUG if args.verbose else logging.WARNING
   104	    logging.basicConfig(
   105	        level=log_level,
   106	        format="%(name)s: %(levelname)s: %(message)s",
   107	        stream=sys.stderr,
   108	    )
   109
   110	    try:
   111	        # Load config
   112	        config = load_config(
   113	            config_path=args.config,
   114	            start_path=args.path[0] if args.path else ".",
   115	        )
   116
   117	        # Apply CLI overrides
   118	        rules_filter = None
   119	        if args.rules:
   120	            rules_filter = [r.strip() for r in args.rules.split(",")]
   121
   122	        exclude_list = None
   123	        if args.exclude:
   124	            exclude_list = [e.strip() for e in args.exclude.split(",")]
   125
   126	        config = merge_cli_overrides(
   127	            config,
   128	            format=args.format,
   129	            severity=args.severity,
   130	            rules_filter=rules_filter,
   131	            exclude=exclude_list,
   132	            fix_hint=args.fix_hint,
   133	            no_color=args.no_color,
   134	            stats=args.stats,
   135	        )
   136
   137	        # Discover files
   138	        filepaths = discover_files(
   139	            paths=args.path,
   140	            extensions=config.extensions,
   141	            exclude=config.exclude,
   142	        )
   143	        files_scanned = len(filepaths)
   144
   145	        if files_scanned == 0:
   146	            logger.warning("No files found to lint")
   147	            print("No files found to lint.", file=sys.stderr)
   148	            return 0
   149
   150	        # Run engine
   151	        engine = LintEngine(config)
   152	        results = engine.lint_files(filepaths)
   153
   154	        # Report
   155	        if config.format == "json":
   156	            reporter = JsonReporter(
   157	                fix_hint=config.fix_hint,
   158	            )
   159	        else:
   160	            reporter = TerminalReporter(
   161	                no_color=config.no_color,
   162	                fix_hint=config.fix_hint,
   163	                stats=config.stats,
   164	            )
   165
   166	        reporter.report(results, files_scanned=files_scanned)
   167
   168	        # Exit code
   169	        total_violations = sum(len(v) for v in results.values())
   170	        return 1 if total_violations > 0 else 0
   171
   172	    except KeyboardInterrupt:
   173	        print("\nInterrupted.", file=sys.stderr)
   174	        return 2
   175	    except Exception as exc:
   176	        logger.error("Fatal error: %s", exc, exc_info=True)
   177	        print(f"Error: {exc}", file=sys.stderr)
   178	        return 2

   182	    sys.exit(main())
