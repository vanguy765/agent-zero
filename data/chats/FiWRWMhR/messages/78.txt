"""BLW001 — function-needs-guard

Functions with parameters should contain at least one guard call.
"""

import re
from typing import List

from bulwark_lint.rules.base import BaseRule, Severity, Violation
from bulwark_lint.languages import FunctionInfo

# Guard call patterns — match any function call containing "guard" in its name
_GUARD_PATTERN = re.compile(
    r'\bguard(?:_?[a-zA-Z]*)\s*\(',
    re.IGNORECASE,
)

# Python-only self/cls parameters
_PYTHON_SELF_ONLY = re.compile(r'^\s*(self|cls)\s*$')
_PYTHON_STRIP_SELF = re.compile(r'^\s*(self|cls)\s*,?\s*')


def _parse_param_names(params: str, language: str) -> List[str]:
    """Extract parameter names from raw parameter string."""
    if not params.strip():
        return []

    cleaned = params

    # Strip Python self/cls
    if language == "python":
        cleaned = _PYTHON_STRIP_SELF.sub('', cleaned)

    if not cleaned.strip():
        return []

    names: List[str] = []
    for part in cleaned.split(','):
        part = part.strip()
        if not part:
            continue

        # Handle type annotations
        # Python: name: type = default
        # TS/JS: name: type
        # Go: name type
        if language == "python":
            name = part.split(':')[0].split('=')[0].strip()
        elif language in ("typescript", "javascript"):
            # Remove optional marker and type
            name = part.split(':')[0].split('=')[0].strip()
            name = name.lstrip('.')
            # Handle destructuring — skip complex patterns
            if '{' in name or '[' in name:
                name = 'destructured'
        elif language == "go":
            # Go: name type or name, name type
            tokens = part.split()
            name = tokens[0] if tokens else part
        else:
            name = part.split(':')[0].split('=')[0].strip()

        # Remove * and ** prefixes (Python)
        name = name.lstrip('*')
        # Remove ? suffix (TS optional)
        name = name.rstrip('?')

        if name and name not in ('self', 'cls'):
            names.append(name)

    return names


def _generate_fix_hint(func: FunctionInfo, param_names: List[str]) -> str:
    """Generate a suggested guard call based on function signature."""
    hints: List[str] = []

    for pname in param_names[:3]:  # Limit to first 3 params for readability
        if func.language == "python":
            hints.append(f'guard_type({pname}, expected_type, "{func.name}", "{pname}")')
        elif func.language in ("typescript", "javascript"):
            hints.append(f'guardType({pname}, "type", "{func.name}", "{pname}")')
        elif func.language == "go":
            hints.append(f'guard({pname} != nil, "{func.name}", "{pname} must not be nil")')

    if len(param_names) > 3:
        hints.append('...')

    return 'Consider adding: ' + '; '.join(hints)


class GuardPresentRule(BaseRule):
    """Functions with parameters should contain at least one guard call."""

    rule_id = "BLW001"
    rule_name = "function-needs-guard"
    description = "Functions with parameters should contain at least one guard call."
    severity = Severity.WARNING

    def check(self, func: FunctionInfo) -> List[Violation]:
        violations: List[Violation] = []

        # Skip: no parameters
        if not func.params.strip():
            return violations

        # Skip: Python self/cls only
        if func.language == "python" and _PYTHON_SELF_ONLY.match(func.params):
            return violations

        # Parse actual parameter names (excluding self/cls)
        param_names = _parse_param_names(func.params, func.language)
        if not param_names:
            return violations

        # Skip: very short functions (< 3 lines) — likely simple getters/wrappers
        body_lines = [l for l in func.body.split('\n') if l.strip()]
        if len(body_lines) < 3:
            return violations

        # Skip: functions that ARE guard/verify/trace implementations
        name_lower = func.name.lower()
        if (name_lower.startswith('guard') or
                name_lower.startswith('verify') or
                name_lower == 'trace' or
                name_lower.startswith('trace_')):
            return violations

        # Skip: private helpers with no real params (Python)
        if func.language == "python" and func.name.startswith('_'):
            real_params = _PYTHON_STRIP_SELF.sub('', func.params).strip()
            if not real_params:
                return violations

        # Check for guard calls in function body
        if _GUARD_PATTERN.search(func.body):
            return violations

        # No guard found — violation
        fix_hint = _generate_fix_hint(func, param_names)

        violations.append(self._make_violation(
            message=f"{func.name}() has parameters but no guard calls",
            filepath=func.filepath,
            line=func.start_line,
            function_name=func.name,
            fix_hint=fix_hint,
        ))

        return violations

_MUTATION_PATTERNS: List[Tuple[re.Pattern, str]] = [

    (re.compile(r'\bdb\.(insert|update|delete|execute|run)\s*\(', re.IGNORECASE), "database mutation"),

    (re.compile(r'\bfs\.(write|append|mkdir|rm|unlink)', re.IGNORECASE), "filesystem mutation"),

    (re.compile(r'\.(publish|send|emit)\s*\(', re.IGNORECASE), "message/event emission"),

]

# Read-only patterns — if the function ONLY has these, skip it
_READ_ONLY_PATTERNS = [

def _detect_mutations(body: str) -> List[str]:
    """Detect side-effect operations in function body. Returns list of descriptions."""
    found: List[str] = []
    for pattern, desc in _MUTATION_PATTERNS:
        if pattern.search(body):
            if desc not in found:
                found.append(desc)
    return found


def _is_read_only(body: str) -> bool:
    """Check if function body contains only read operations."""
    has_read = any(p.search(body) for p in _READ_ONLY_PATTERNS)
    has_mutation = any(p.search(body) for p, _ in _MUTATION_PATTERNS)
    return has_read and not has_mutation

        "Prisma mutation": f'verify(result !== null, "{func.name}", "Prisma operation returned null")',
        "file write": f'verify_file_written(filepath, "{func.name}", "file write failed")',
        "filesystem mutation": f'verify_file_written(path, "{func.name}", "filesystem operation failed")',
        "HTTP POST": f'verify_response_ok(response, "{func.name}", "POST request failed")',
        "HTTP PUT": f'verify_response_ok(response, "{func.name}", "PUT request failed")',
        "HTTP PATCH": f'verify_response_ok(response, "{func.name}", "PATCH request failed")',
        "HTTP DELETE": f'verify_response_ok(response, "{func.name}", "DELETE request failed")',
        "HTTP mutation": f'verify_response_ok(response, "{func.name}", "HTTP mutation failed")',
        "Axios mutation": f'verify_response_ok(response, "{func.name}", "Axios request failed")',
        "message/event emission": f'verify(sent === true, "{func.name}", "message send failed")',
        "notification send": f'verify(sent === true, "{func.name}", "notification send failed")',

"""BLW003 — trace-at-boundary

Functions at phase boundaries should have trace calls.
"""

import re
from typing import List

from bulwark_lint.rules.base import BaseRule, Severity, Violation
from bulwark_lint.languages import FunctionInfo

# ---------------------------------------------------------------------------
# Boundary indicators
# ---------------------------------------------------------------------------

# Boundary keywords to search for in function names (case-insensitive)
_BOUNDARY_KEYWORDS = {
    "fetch", "parse", "transform", "process", "convert", "load", "extract",
    "handle", "middleware", "serialize", "deserialize", "decode", "encode",
    "normalize", "validate", "sanitize", "aggregate", "compile", "render",
    "translate", "ingest", "digest", "pipeline", "dispatch", "route", "resolve",
}

# Accessor patterns — skip these
_ACCESSOR_PREFIXES = ('get_', 'get', 'is_', 'is', 'has_', 'has', 'should_', 'can_')

# Trace call patterns
_TRACE_PATTERN = re.compile(
    r'\btrace(?:_?[a-zA-Z]*)\s*\(',
    re.IGNORECASE,
)

# Exclude false positives: traceback, stacktrace
_TRACE_FALSE_POSITIVES = re.compile(
    r'\b(traceback|stacktrace|stack_trace|traceError|backtrace)\b',
    re.IGNORECASE,
)

# Decorator-based trace (Python)
_TRACE_DECORATOR = re.compile(r'@\s*[Tt]raced?\b')


def _split_name_parts(name: str) -> List[str]:
    """Split a function name into lowercase parts.

    Handles both snake_case and camelCase:
      - "parseResponse" -> ["parse", "response"]
      - "fetch_remote_users" -> ["fetch", "remote", "users"]
      - "handleCreateUser" -> ["handle", "create", "user"]
    """
    parts: List[str] = []
    for segment in name.split("_"):
        # Split camelCase: insert boundary before uppercase letters
        tokens = re.sub(r'([A-Z])', r'_\1', segment).split('_')
        parts.extend(t.lower() for t in tokens if t)
    return parts


def _has_boundary_name(func_name: str) -> bool:
    """Check if function name contains a boundary keyword."""
    parts = _split_name_parts(func_name)
    for part in parts:
        if part in _BOUNDARY_KEYWORDS:
            return True
    return False


def _is_accessor(func_name: str) -> bool:
    """Check if function name suggests a simple accessor."""
    lower = func_name.lower()
    for prefix in _ACCESSOR_PREFIXES:
        if lower.startswith(prefix) and len(lower) > len(prefix):
            return True
    return False


def _is_boundary_function(func: FunctionInfo) -> bool:
    """Heuristic: does this function appear to be at a phase boundary?

    A function is considered a boundary if its name contains a boundary keyword.
    Being async alone is NOT sufficient — many async functions are simple CRUD
    operations that don't represent phase boundaries.
    """
    return _has_boundary_name(func.name)


def _has_trace(func: FunctionInfo, source_before_func: str = "") -> bool:
    """Check if function has trace calls (excluding false positives)."""
    # Check for trace decorator above function (Python)
    if _TRACE_DECORATOR.search(source_before_func):
        return True

    # Check body for trace calls
    matches = _TRACE_PATTERN.findall(func.body)
    if not matches:
        return False

    # Filter out false positives
    body_without_fp = _TRACE_FALSE_POSITIVES.sub('', func.body)
    return bool(_TRACE_PATTERN.search(body_without_fp))


def _generate_fix_hint(func: FunctionInfo) -> str:
    """Generate a suggested trace call."""
    if func.language == "python":
        return f'Consider adding: result = trace("after_{func.name}", data)'
    elif func.language in ("typescript", "javascript"):
        cap = func.name[0].upper() + func.name[1:] if func.name else "Func"
        return f'Consider adding: const result = trace("after{cap}", data);'
    elif func.language == "go":
        return f'Consider adding: result = trace("after_{func.name}", data)'
    return f'Consider adding a trace() call in {func.name}()'


class TraceAtBoundaryRule(BaseRule):
    """Functions at phase boundaries should have trace calls."""

    rule_id = "BLW003"
    rule_name = "trace-at-boundary"
    description = "Functions at phase boundaries should have trace calls for observability."
    severity = Severity.INFO

    def check(self, func: FunctionInfo) -> List[Violation]:
        violations: List[Violation] = []

        # Skip accessors
        if _is_accessor(func.name):
            return violations

        # Skip very short functions (< 5 lines)
        body_lines = [l for l in func.body.split('\n') if l.strip()]
        if len(body_lines) < 5:
            return violations

        # Check if this looks like a boundary function
        if not _is_boundary_function(func):
            return violations

        # Check for existing trace calls
        if _has_trace(func):
            return violations

        # Boundary function without trace — violation
        fix_hint = _generate_fix_hint(func)

        violations.append(self._make_violation(
            message=f"{func.name}() appears to be a phase boundary but has no trace call",
            filepath=func.filepath,
            line=func.start_line,
            function_name=func.name,
            fix_hint=fix_hint,
        ))

        return violations

"""BLW004 — no-bare-assert

Code should use guard()/GuardError instead of bare assert or raw throw/raise for validation.
"""

import re
from pathlib import Path
from typing import List

from bulwark_lint.rules.base import BaseRule, Severity, Violation
from bulwark_lint.languages import FunctionInfo

# ---------------------------------------------------------------------------
# Test file detection
# ---------------------------------------------------------------------------

_TEST_FILE_PATTERNS = re.compile(
    r'(test_|_test\.|_test$|\.test\.|_spec\.|spec_|\.spec\.)',
    re.IGNORECASE,
)

_TEST_FUNC_PATTERNS = re.compile(
    r'^(test_|test[A-Z]|it_|describe_|spec_)',
)


def _is_test_file(filepath: str) -> bool:
    """Check if file is a test file."""
    name = Path(filepath).name
    return bool(_TEST_FILE_PATTERNS.search(name))


def _is_test_function(func_name: str) -> bool:
    """Check if function is a test function."""
    return bool(_TEST_FUNC_PATTERNS.match(func_name))


# ---------------------------------------------------------------------------
# Anti-patterns
# ---------------------------------------------------------------------------

# Python: bare assert at start of line (not in string)
_PY_BARE_ASSERT = re.compile(r'^\s*assert\s+', re.MULTILINE)

# Python: raise ValueError/TypeError for validation
_PY_RAISE_VALIDATION = re.compile(
    r'^\s*raise\s+(ValueError|TypeError|AssertionError)\s*\(',
    re.MULTILINE,
)

# TS/JS: throw new Error for what looks like validation
_TS_THROW_ERROR = re.compile(
    r'^\s*throw\s+new\s+Error\s*\(',
    re.MULTILINE,
)

# TS/JS: throw new TypeError/RangeError for validation
_TS_THROW_VALIDATION = re.compile(
    r'^\s*throw\s+new\s+(TypeError|RangeError|ReferenceError)\s*\(',
    re.MULTILINE,
)


def _find_violations_in_body(body: str, language: str) -> List[dict]:
    """Find bare assert/throw patterns in function body. Returns list of {pattern, line_offset}."""
    results: List[dict] = []
    lines = body.split('\n')

    for i, line in enumerate(lines):
        stripped = line.strip()

        # Skip comments
        if stripped.startswith('#') or stripped.startswith('//'):
            continue

        # Skip strings (rough heuristic: line is mostly a string)
        if stripped.startswith(('"', "'", '`')):
            continue

        if language == "python":
            if _PY_BARE_ASSERT.match(line):
                results.append({
                    "pattern": "bare assert",
                    "line_offset": i,
                    "text": stripped,
                })
            elif _PY_RAISE_VALIDATION.match(line):
                results.append({
                    "pattern": "raise for validation",
                    "line_offset": i,
                    "text": stripped,
                })

        elif language in ("typescript", "javascript"):
            if _TS_THROW_ERROR.match(line):
                results.append({
                    "pattern": "throw new Error",
                    "line_offset": i,
                    "text": stripped,
                })
            elif _TS_THROW_VALIDATION.match(line):
                results.append({
                    "pattern": "throw validation error",
                    "line_offset": i,
                    "text": stripped,
                })

    return results


def _generate_fix_hint(pattern: str, language: str, func_name: str) -> str:
    """Generate fix suggestion."""
    if language == "python":
        if "assert" in pattern:
            return f'Replace assert with: guard(condition, "{func_name}", "description")'
        else:
            return f'Replace raise with: guard(condition, "{func_name}", "description") which raises GuardError'
    else:
        return f'Replace throw with: guard(condition, "{func_name}", "description") which throws GuardError'


class NoBareAssertRule(BaseRule):
    """Code should use guard() instead of bare assert/throw for validation."""

    rule_id = "BLW004"
    rule_name = "no-bare-assert"
    description = "Use guard()/GuardError instead of bare assert or raw throw/raise for validation."
    severity = Severity.WARNING

    def check(self, func: FunctionInfo) -> List[Violation]:
        violations: List[Violation] = []

        # Skip test files
        if _is_test_file(func.filepath):
            return violations

        # Skip test functions
        if _is_test_function(func.name):
            return violations

        # Find anti-patterns
        found = _find_violations_in_body(func.body, func.language)

        for item in found:
            # Calculate actual line number
            # We need to find the line offset within the body relative to start_line
            actual_line = func.start_line + item["line_offset"] + 1  # +1 for def line itself

            fix_hint = _generate_fix_hint(item["pattern"], func.language, func.name)

            violations.append(self._make_violation(
                message=f"{func.name}() uses {item['pattern']} for validation instead of guard()",
                filepath=func.filepath,
                line=actual_line,
                function_name=func.name,
                fix_hint=fix_hint,
            ))

        return violations

    def check_file(self, filepath: str, content: str, language: str) -> List[Violation]:
        """Check entire file for GuardError/VerifyError presence."""
        violations: List[Violation] = []

        # Skip Bulwark reference implementations
        if _is_bulwark_reference(filepath):
            return violations

        uses_guard = bool(_GUARD_USAGE.search(content))
        uses_verify = bool(_VERIFY_USAGE.search(content))

        # Skip if file doesn't use any guard/verify calls
        if not uses_guard and not uses_verify:
            return violations

        # Check for GuardError
        if uses_guard and not _has_guard_error(content):
            violations.append(self._make_violation(
                message="File uses guard() calls but does not define or import GuardError",
                filepath=filepath,
                line=1,
                function_name="(file-level)",
                fix_hint=self._guard_error_hint(language),

        if uses_verify and not _has_verify_error(content):
            violations.append(self._make_violation(
                message="File uses verify() calls but does not define or import VerifyError",
                filepath=filepath,
                line=1,
                function_name="(file-level)",
                fix_hint=self._verify_error_hint(language),
            ))

        return violations
