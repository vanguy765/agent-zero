export async function handleCreateUser(req: Request, env: Env): Promise<Response> {
    guardType(req, "object", "handleCreateUser", "req");
    guardType(env, "object", "handleCreateUser", "env");

    const body = await req.json();
    const name = body.name;
    const email = body.email;

    guardNotEmpty(name, "handleCreateUser", "name");
    guardType(name, "string", "handleCreateUser", "name");
    guardNotEmpty(email, "handleCreateUser", "email");
    guard(email.includes("@"), "handleCreateUser", "email must contain @");

    const result = await db.insert("users", { name, email });
    verify(result.id !== undefined, "handleCreateUser", "INSERT must return id");

    const traced = trace("afterCreateUser", { id: result.id });
    return new Response(JSON.stringify(traced), { status: 201 });
}

export async function syncUserToExternalApi(userId: string, apiKey: string): Promise<void> {
    guardType(userId, "string", "syncUserToExternalApi", "userId");
    guardNotEmpty(userId, "syncUserToExternalApi", "userId");
    guardType(apiKey, "string", "syncUserToExternalApi", "apiKey");

    const user = await db.query("SELECT * FROM users WHERE id = ?", [userId]);

    const response = await fetch("https://api.external.com/users", {
        method: "POST",
        headers: { "Authorization": `Bearer ${apiKey}` },
        body: JSON.stringify(user),
    });

    verifyResponseOk(response, "syncUserToExternalApi", "POST to external API");
}
