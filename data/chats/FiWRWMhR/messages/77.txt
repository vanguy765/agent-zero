class LintEngine:
    """Core linting engine.

    Coordinates file discovery, function extraction, rule execution,
    and violation filtering.
    """

    def __init__(self, config: LintConfig):
        self.config = config
        self.rules = self._load_rules()
        self.min_severity = _SEVERITY_MAP.get(config.severity, Severity.WARNING)

    def _load_rules(self) -> List[BaseRule]:
        """Load and configure enabled rules based on config."""
        all_rules = get_all_rules()
        enabled: List[BaseRule] = []

        for rule in all_rules:
            if not self.config.is_rule_enabled(rule.rule_id):
                logger.debug("Rule %s disabled by config", rule.rule_id)
                continue

            # Apply severity override from config
            severity_override = self.config.get_rule_severity(rule.rule_id)
            if severity_override and severity_override in _SEVERITY_MAP:
                rule.severity = _SEVERITY_MAP[severity_override]

            enabled.append(rule)

        logger.info("Loaded %d rules: %s", len(enabled),
                     ", ".join(r.rule_id for r in enabled))
        return enabled

    def lint_file(self, filepath: str) -> List[Violation]:
        """Lint a single file and return violations.

        Args:
            filepath: Absolute path to the file.

        Returns:
            List of Violation objects found in the file.
        """
        violations: List[Violation] = []

        # Read file
        try:
            with open(filepath, "r", encoding="utf-8", errors="replace") as f:
                content = f.read()
        except OSError as exc:
            logger.error("Cannot read %s: %s", filepath, exc)
            return violations

        # Detect language
        language = detect_language(filepath)
        if language is None:
            logger.debug("Unsupported file type: %s", filepath)
            return violations

        # Extract functions
        functions = extract_functions(content, filepath, language)
        logger.debug("Extracted %d functions from %s", len(functions), filepath)

        # Separate function-level and file-level rules
        func_rules: List[BaseRule] = []
        file_rules: List[BaseRule] = []
        for rule in self.rules:
            if rule.file_level:
                file_rules.append(rule)
            else:
                func_rules.append(rule)

        # Run function-level rules
        for func in functions:
            for rule in func_rules:
                try:
                    found = rule.check(func)
                    violations.extend(found)
                except Exception as exc:
                    logger.error("Rule %s failed on %s:%s — %s",
                                 rule.rule_id, filepath, func.name, exc)

        # Run file-level rules
        for rule in file_rules:
            try:
                found = rule.check_file(filepath, content, language)
                violations.extend(found)
            except Exception as exc:
                logger.error("File rule %s failed on %s — %s",
                             rule.rule_id, filepath, exc)

        # Filter by minimum severity
        violations = [
            v for v in violations
            if v.severity >= self.min_severity
        ]

        return violations

    def lint_files(self, filepaths: List[str]) -> Dict[str, List[Violation]]:
        """Lint multiple files.

        Args:
            filepaths: List of absolute file paths.

        Returns:
            Dict mapping filepath to list of violations.

        results: Dict[str, List[Violation]] = {}

        for filepath in filepaths:
            violations = self.lint_file(filepath)
            if violations:

        return self.lint_files(filepaths)

    def files_scanned(self) -> int:
        """Return count of files that would be scanned (for stats)."""
        # This is set after lint_paths runs
        return getattr(self, '_files_scanned', 0)

@dataclass
class FunctionInfo:
    """Extracted function metadata and body."""
    name: str
    start_line: int   # 1-indexed
    end_line: int      # 1-indexed
    params: str        # raw parameter string
    body: str          # raw function body
    language: str      # 'python', 'typescript', 'javascript', 'go'
    filepath: str      # absolute path


# ---------------------------------------------------------------------------
# Python function extraction
# ---------------------------------------------------------------------------

# Matches: def name(params):  and  async def name(params):
_PY_FUNC_RE = re.compile(
    r'^([ \t]*)(async\s+)?def\s+([a-zA-Z_][a-zA-Z0-9_]*)\s*\(([^)]*)\)\s*(?:->\s*[^:]+)?\s*:',
    re.MULTILINE,
)


def _extract_python_functions(source: str, filepath: str) -> List[FunctionInfo]:
    """Extract functions from Python source using indentation-based body detection."""
    lines = source.split('\n')
    functions: List[FunctionInfo] = []

    for match in _PY_FUNC_RE.finditer(source):
        indent = match.group(1)
        func_name = match.group(3)
        params = match.group(4).strip()

        # Find the line number of the def statement (1-indexed)
        start_offset = match.start()
        start_line = source[:start_offset].count('\n') + 1

        # Determine indentation level of the def line
        def_indent_len = len(indent.replace('\t', '    '))

        # Collect body lines: everything indented deeper than the def line
        body_lines: List[str] = []
        end_line = start_line
        i = start_line  # 0-indexed into lines array = start_line (since start_line is 1-indexed)

        while i < len(lines):
            line = lines[i]
            stripped = line.strip()

            if not stripped:  # blank line — include in body, keep going
                body_lines.append(line)
                end_line = i + 1
                i += 1
                continue

            line_indent_len = len(line) - len(line.lstrip())
            line_indent_len = len(line[:len(line) - len(line.lstrip())].replace('\t', '    '))

            if line_indent_len > def_indent_len:
                body_lines.append(line)
                end_line = i + 1
                i += 1
            else:
                break

        # Strip trailing blank lines from body
        while body_lines and not body_lines[-1].strip():
            body_lines.pop()
            end_line -= 1

        if not body_lines:
            end_line = start_line

        body = '\n'.join(body_lines)

        functions.append(FunctionInfo(
            name=func_name,
            start_line=start_line,
            end_line=end_line,
            params=params,
            body=body,
            language="python",
            filepath=filepath,
        ))

    return functions

# Pattern 1: function declarations
#   [export [default]] [async] function name(params) [: ReturnType] {
_TS_FUNC_DECL_RE = re.compile(
    r'^[ \t]*(?:export\s+(?:default\s+)?)?(?:async\s+)?function\s+'
    r'([a-zA-Z_$][a-zA-Z0-9_$]*)\s*(?:<[^>]*>)?\s*\(([^)]*)\)\s*(?::\s*[^{]+?)?\s*\{',
    re.MULTILINE,
)

# Pattern 2: arrow functions / const assignments
#   [export] const name = [async] (params) [: ReturnType] => {
_TS_ARROW_RE = re.compile(
    r'^[ \t]*(?:export\s+)?(?:const|let|var)\s+'
    r'([a-zA-Z_$][a-zA-Z0-9_$]*)\s*(?::\s*[^=]+?)?\s*=\s*(?:async\s+)?'
    r'(?:\([^)]*\)|[a-zA-Z_$][a-zA-Z0-9_$]*)\s*(?::\s*[^=]+?)?\s*=>\s*\{',
    re.MULTILINE,
)

# Pattern 2b: extract params from arrow functions more carefully
_TS_ARROW_PARAMS_RE = re.compile(
    r'=\s*(?:async\s+)?\(([^)]*)\)\s*(?::\s*[^=]+?)?\s*=>\s*\{',
)

# Pattern 3: class methods
#   [async] methodName(params) [: ReturnType] {
_TS_METHOD_RE = re.compile(
    r'^[ \t]+(?:(?:public|private|protected|static|readonly|abstract|override)\s+)*'
    r'(?:async\s+)?([a-zA-Z_$][a-zA-Z0-9_$]*)\s*(?:<[^>]*>)?\s*\(([^)]*)\)\s*(?::\s*[^{]+?)?\s*\{',
    re.MULTILINE,
)


def _find_matching_brace(source: str, open_pos: int) -> int:
    """Find the position of the closing brace matching the opening brace at open_pos."""
    depth = 0
    i = open_pos
    in_string = None  # None, "'", '"', '`'
    escape_next = False

    while i < len(source):
        ch = source[i]

        if escape_next:
            escape_next = False
            i += 1
            continue

        if ch == '\\':
            escape_next = True
            i += 1
            continue

        if in_string:
            if ch == in_string:
                # Check for template literal end vs interpolation
                in_string = None
            i += 1
            continue

        if ch in ('"', "'", '`'):
            in_string = ch
            i += 1
            continue

        # Skip single-line comments
        if ch == '/' and i + 1 < len(source):
            if source[i + 1] == '/':
                # Skip to end of line
                nl = source.find('\n', i)
                if nl == -1:
                    break
                i = nl + 1
                continue
            elif source[i + 1] == '*':
                # Skip block comment
                end = source.find('*/', i + 2)
                if end == -1:
                    break
                i = end + 2
                continue

        if ch == '{':
            depth += 1
        elif ch == '}':
            depth -= 1
            if depth == 0:
                return i

        i += 1

    return len(source) - 1  # fallback: end of file


def _extract_ts_js_functions(source: str, filepath: str, language: str) -> List[FunctionInfo]:
    """Extract functions from TypeScript/JavaScript source using brace counting."""
    functions: List[FunctionInfo] = []
    seen_ranges: List[Tuple[int, int]] = []  # avoid duplicates

    def _add_function(name: str, params: str, match_start: int, brace_search_start: int) -> None:
        # Find the opening brace
        brace_pos = source.find('{', brace_search_start)
        if brace_pos == -1:
            return

        # Check for overlap with already-extracted functions
        for s, e in seen_ranges:
            if s <= match_start <= e:
                return

        close_pos = _find_matching_brace(source, brace_pos)
        body = source[brace_pos + 1:close_pos]

        start_line = source[:match_start].count('\n') + 1
        end_line = source[:close_pos].count('\n') + 1

        seen_ranges.append((match_start, close_pos))

        functions.append(FunctionInfo(
            name=name,
            start_line=start_line,
            end_line=end_line,
            params=params.strip(),
            body=body,
            language=language,
            filepath=filepath,
        ))

    # Function declarations
    for m in _TS_FUNC_DECL_RE.finditer(source):
        _add_function(m.group(1), m.group(2), m.start(), m.end() - 1)

    # Arrow functions
    for m in _TS_ARROW_RE.finditer(source):
        name = m.group(1)
        # Extract params from the arrow part
        params = ""
        pm = _TS_ARROW_PARAMS_RE.search(m.group(0))
        if pm:
            params = pm.group(1)
        _add_function(name, params, m.start(), m.end() - 1)

    # Class methods
    for m in _TS_METHOD_RE.finditer(source):
        name = m.group(1)
        # Skip keywords that look like method names
        if name in ('if', 'else', 'for', 'while', 'switch', 'catch', 'constructor',
                     'return', 'throw', 'new', 'delete', 'typeof', 'import', 'export'):
            if name != 'constructor':  # constructor is a valid method
                continue
        _add_function(name, m.group(2), m.start(), m.end() - 1)

    # Sort by start line
    functions.sort(key=lambda f: f.start_line)
    return functions

_GO_FUNC_RE = re.compile(

def _extract_go_functions(source: str, filepath: str) -> List[FunctionInfo]:
    """Extract functions from Go source using brace counting."""
    functions: List[FunctionInfo] = []
    seen_ranges: List[Tuple[int, int]] = []

    for pattern in (_GO_FUNC_RE, _GO_METHOD_RE):
        for m in pattern.finditer(source):
            name = m.group(1)
            params = m.group(2).strip()
            match_start = m.start()

            # Check overlap
            overlap = False
            for s, e in seen_ranges:
                if s <= match_start <= e:
                    overlap = True
                    break
            if overlap:
                continue

            brace_pos = source.find('{', m.end() - 1)
            if brace_pos == -1:
                continue

            close_pos = _find_matching_brace(source, brace_pos)
            body = source[brace_pos + 1:close_pos]

            start_line = source[:match_start].count('\n') + 1
            end_line = source[:close_pos].count('\n') + 1

            seen_ranges.append((match_start, close_pos))

            functions.append(FunctionInfo(
                name=name,
                start_line=start_line,
                end_line=end_line,
                params=params,
                body=body,
                language="go",
                filepath=filepath,
            ))

    functions.sort(key=lambda f: f.start_line)
    return functions

def extract_functions(source: str, filepath: str, language: Optional[str] = None) -> List[FunctionInfo]:
    """Extract all functions from source code.

    Args:
        source: Raw source code string.
        filepath: Absolute path to the file (used in FunctionInfo).
        language: Override language detection. If None, detected from filepath.

    Returns:
        List of FunctionInfo objects.
    """
    if language is None:
        language = detect_language(filepath)
    if language is None:
        return []

    if language == "python":
        return _extract_python_functions(source, filepath)
    elif language in ("typescript", "javascript"):
        return _extract_ts_js_functions(source, filepath, language)
    elif language == "go":
        return _extract_go_functions(source, filepath)
    else:
        return []

class Severity(Enum):
    """Violation severity levels."""
    ERROR = "error"
    WARNING = "warning"
    INFO = "info"

    @property
    def weight(self) -> int:
        """Numeric weight for comparison (higher = more severe)."""
        return {"error": 3, "warning": 2, "info": 1}[self.value]

    def __ge__(self, other: "Severity") -> bool:
        return self.weight >= other.weight

    def __gt__(self, other: "Severity") -> bool:
        return self.weight > other.weight

    def __le__(self, other: "Severity") -> bool:
        return self.weight <= other.weight

    def __lt__(self, other: "Severity") -> bool:
        return self.weight < other.weight


@dataclass
class Violation:
    """A single linter violation found in source code."""
    rule_id: str
    rule_name: str
    message: str
    filepath: str
    line: int
    function_name: str
    severity: Severity = Severity.WARNING
    fix_hint: Optional[str] = None

    def to_dict(self) -> dict:
        """Serialize to dictionary for JSON output."""
        result = {
            "rule_id": self.rule_id,
            "rule_name": self.rule_name,
            "severity": self.severity.value,
            "message": self.message,
            "filepath": self.filepath,
            "line": self.line,
            "function_name": self.function_name,
        }
        if self.fix_hint:
            result["fix_hint"] = self.fix_hint
        return result
