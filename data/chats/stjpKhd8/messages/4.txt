CREATE TABLE IF NOT EXISTS public.trades (
    id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
    code text NOT NULL UNIQUE,
    name text NOT NULL,
    description text,
    icon text,
    color text,

    is_active boolean NOT NULL DEFAULT true,
    scope text NOT NULL DEFAULT 'system',

-- 1O. Trade name sync trigger
-- =============================================================================
CREATE OR REPLACE FUNCTION public.sync_trade_name()
RETURNS TRIGGER AS $$
BEGIN
    IF NEW.trade_id IS NOT NULL AND (TG_OP = 'INSERT' OR OLD.trade_id IS DISTINCT FROM NEW.trade_id) THEN
        SELECT name INTO NEW.trade_name FROM public.trades WHERE id = NEW.trade_id;
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trg_sync_trade_name_templates ON public.work_order_templates;
CREATE TRIGGER trg_sync_trade_name_templates
    BEFORE INSERT OR UPDATE OF trade_id ON public.work_order_templates
    FOR EACH ROW EXECUTE FUNCTION public.sync_trade_name();

DROP TRIGGER IF EXISTS trg_sync_trade_name_work_orders ON public.work_orders;
CREATE TRIGGER trg_sync_trade_name_work_orders
    BEFORE INSERT OR UPDATE OF trade_id ON public.work_orders
    FOR EACH ROW EXECUTE FUNCTION public.sync_trade_name();

-- Trade name + code sync trigger for sops
CREATE OR REPLACE FUNCTION public.sync_trade_name_and_code()
RETURNS TRIGGER AS $$
BEGIN
    IF NEW.trade_id IS NOT NULL AND (TG_OP = 'INSERT' OR OLD.trade_id IS DISTINCT FROM NEW.trade_id) THEN
        SELECT name, code INTO NEW.trade_name, NEW.trade_code FROM public.trades WHERE id = NEW.trade_id;
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trg_sync_trade_name_code_sops ON public.sops;
CREATE TRIGGER trg_sync_trade_name_code_sops
    BEFORE INSERT OR UPDATE OF trade_id ON public.sops
    FOR EACH ROW EXECUTE FUNCTION public.sync_trade_name_and_code();
