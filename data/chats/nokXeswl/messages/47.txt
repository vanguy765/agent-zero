import type { Component } from 'solid-js';
import { Router, Route } from '@solidjs/router';
import Layout from './components/Layout';
import Dashboard from './pages/Dashboard';
import Units from './pages/Units';
import Templates from './pages/Templates';
import TemplateDetail from './pages/TemplateDetail';
import TemplateEditor from './pages/TemplateEditor';
import CompanyTemplates from './pages/CompanyTemplates';
import ProjectTemplates from './pages/ProjectTemplates';
import WorkOrderDetail from './pages/WorkOrderDetail';
import Projects from './pages/Projects';
import ProjectDetail from './pages/ProjectDetail';
import NewProject from './pages/NewProject';
import ProjectDocuments from './pages/ProjectDocuments';
import './index.css';

const App: Component = () => {
  return (
    <Router root={Layout}>
      <Route path="/" component={Dashboard} />
      <Route path="/units" component={Units} />
      <Route path="/templates" component={Templates} />
      <Route path="/templates/company" component={CompanyTemplates} />
      <Route path="/templates/new" component={TemplateEditor} />
      <Route path="/templates/:id/edit" component={TemplateEditor} />
      <Route path="/templates/:id" component={TemplateDetail} />
      <Route path="/work-orders/:id" component={WorkOrderDetail} />
      <Route path="/projects" component={Projects} />
      <Route path="/projects/new" component={NewProject} />
      <Route path="/projects/:id" component={ProjectDetail} />
      <Route path="/projects/:id/documents" component={ProjectDocuments} />
      <Route path="/projects/:projectId/templates" component={ProjectTemplates} />
    </Router>
  );
};

export default App;
