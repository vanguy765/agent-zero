-- 4. Trigger: freeze project templates on activation
--    When a project transitions to 'active', copy the
--    definition from the parent company template into
--    each unfrozen project template.
-- -----------------------------------------------
CREATE OR REPLACE FUNCTION public.freeze_project_templates()
RETURNS TRIGGER AS $$
BEGIN
  -- Only fire when status changes TO 'active'
  IF NEW.status = 'active' AND (OLD.status IS DISTINCT FROM 'active') THEN
    UPDATE public.work_order_templates pt
    SET
      definition = ct.definition,
      is_frozen = true,
      updated_at = now()
    FROM public.work_order_templates ct
    WHERE pt.project_id = NEW.id
      AND pt.scope = 'project'
      AND pt.parent_template_id = ct.id
      AND pt.is_frozen = false
      AND pt.definition IS NULL;
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;
