onboardRoute.get('/trades', async (c) => {
  const { data, error } = await adminClient
    .from('work_order_templates')
    .select('trade_name')
    .eq('scope', 'app')
    .eq('is_active', true)
    .order('trade_name');

  if (error) return c.json({ error: error.message }, 500);

  // Deduplicate trade names
  const trades = [...new Set((data || []).map((r: { trade_name: string }) => r.trade_name))].sort();
  return c.json(trades);
});
