// GET PROJECT TEMPLATES (with parent definition resolution for lazy-copy)
workOrdersRoute.get('/templates/project/:projectId', async (c) => {
    const supabase = getScopedClient(c);
    const projectId = c.req.param('projectId');

    const { data, error } = await supabase
        .from('work_order_templates')
        .select('*')
        .eq('scope', 'project')
        .eq('project_id', projectId)
        .eq('is_active', true)
        .order('trade_name', { ascending: true });

    if (error) {
        return c.json({ error: error.message }, 500);
    }

    // For unfrozen templates with null definition, resolve from parent
    const resolved = await Promise.all(
        (data || []).map(async (template: any) => {
            if (template.definition === null && template.parent_template_id) {
                const { data: parent } = await supabase
                    .from('work_order_templates')
                    .select('definition')
                    .eq('id', template.parent_template_id)
                    .single();
                return { ...template, definition: parent?.definition || null, _inherited: true };
            }
            return { ...template, _inherited: false };
        })
    );

    return c.json(resolved);
});
