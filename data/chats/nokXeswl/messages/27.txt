import { createSignal, createResource, For, Show } from 'solid-js';
import type { Component } from 'solid-js';

const API = 'http://localhost:3000';

interface FormState {
  company_name: string;
  company_email: string;
  phone: string;
  address_line1: string;
  address_line2: string;
  city: string;
  state: string;
  zip: string;
  country: string;
  contact_name: string;
  contact_title: string;
  website: string;
  admin_email: string;
  admin_password: string;
  admin_password_confirm: string;
  industry: string;
  description: string;
}

const EMPTY: FormState = {
  company_name: '',
  company_email: '',
  phone: '',
  address_line1: '',
  address_line2: '',
  city: '',
  state: '',
  zip: '',
  country: 'US',
  contact_name: '',
  contact_title: '',
  website: '',
  admin_email: '',
  admin_password: '',
  admin_password_confirm: '',
  industry: '',
  description: '',
};

interface OnboardingFormPageProps {
  /** If provided, the form shows a "Back to login" button */
  onNavigate?: (path: string) => void;
}

/** Fetch available trades from the API */
const fetchTrades = async (): Promise<string[]> => {
  const res = await fetch(`${API}/onboard/trades`);
  if (!res.ok) return [];
  return res.json();
};

export const OnboardingFormPage: Component<OnboardingFormPageProps> = (props) => {
  const [form, setForm] = createSignal<FormState>({ ...EMPTY });
  const [submitting, setSubmitting] = createSignal(false);
  const [error, setError] = createSignal<string | null>(null);
  const [success, setSuccess] = createSignal<{ orgName: string; adminEmail: string } | null>(null);

  // Fetch trade names for the industry dropdown
  const [trades] = createResource(fetchTrades);

  const set = (field: keyof FormState, value: string) => {
    setForm((prev) => ({ ...prev, [field]: value }));
    setError(null);
  };

  const f = (field: keyof FormState) => form()[field];

  const validate = (): string | null => {
    if (!f('company_name').trim()) return 'Company name is required.';
    if (!f('company_email').trim()) return 'Company email is required.';
    if (!f('contact_name').trim()) return 'Contact name is required.';
    if (!f('admin_email').trim()) return 'Admin login email is required.';
    if (!f('admin_password')) return 'Password is required.';
    if (f('admin_password').length < 8) return 'Password must be at least 8 characters.';
    if (f('admin_password') !== f('admin_password_confirm')) return 'Passwords do not match.';
    if (!f('industry')) return 'Please select an industry.';
    return null;
  };

  const handleSubmit = async (e: Event) => {
    e.preventDefault();
    const err = validate();
    if (err) { setError(err); return; }

    setSubmitting(true);
    setError(null);

    const { admin_password_confirm, ...payload } = form();

    try {
      const res = await fetch(`${API}/onboard`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload),
      });
      const json = await res.json();
      if (!res.ok) {
        setError(json.error || 'Submission failed. Please try again.');
      } else {
        setSuccess({
          orgName: json.organization.name,
          adminEmail: json.admin.email,
        });
      }
    } catch {
      setError('Network error \u2014 please check your connection and try again.');
    } finally {
      setSubmitting(false);
    }
  };

  // \u2500\u2500 Success screen \u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500
  return (
    <div class="min-h-screen bg-gray-50 flex items-center justify-center p-4">
      <div class="w-full max-w-2xl">
        {/* Header */}
        <div class="text-center mb-8">
          <h1 class="text-3xl font-bold text-gray-900">Get Started</h1>
          <p class="mt-2 text-gray-500">
            Register your company and create your admin account.
          </p>
        </div>

        <Show
          when={!success()}
          fallback={
            <div class="card text-center">
              <div class="text-5xl mb-4">\u2713</div>
              <h2 class="text-2xl font-bold text-gray-900 mb-2">You're all set!</h2>
              <p class="text-gray-600 mb-1">
                <strong>{success()!.orgName}</strong> has been registered.
              </p>
              <p class="text-gray-500 text-sm mb-6">
                Sign in with <strong>{success()!.adminEmail}</strong> to access the platform.
              </p>
              <a
                href="http://localhost:3004/"
                class="btn btn-primary inline-block"
              >
                Login to Work Orders
              </a>
            </div>
          }
        >
          <form onSubmit={handleSubmit} class="space-y-6">
            {/* Company Info */}
            <div class="card">
              <h2 class="text-lg font-semibold text-gray-900 mb-4">Company Information</h2>
              <div class="grid grid-cols-1 sm:grid-cols-2 gap-4">
                <div class="sm:col-span-2">
                  <label class="block text-sm font-medium text-gray-700 mb-1">
                    Company Name <span class="text-red-500">*</span>
                  </label>
                  <input
                    class="input"
                    type="text"
                    placeholder="Acme Construction LLC"
                    value={f('company_name')}
                    onInput={(e) => set('company_name', e.currentTarget.value)}
                    required
                  />
                </div>

                <div>
                  <label class="block text-sm font-medium text-gray-700 mb-1">
                    Company Email <span class="text-red-500">*</span>
                  </label>
                  <input
                    class="input"
                    type="email"
                    placeholder="info@acme.com"
                    value={f('company_email')}
                    onInput={(e) => set('company_email', e.currentTarget.value)}
                    required
                  />
                </div>

                <div>
                  <label class="block text-sm font-medium text-gray-700 mb-1">Phone</label>
                  <input
                    class="input"
                    type="tel"
                    placeholder="+1 (555) 000-0000"
                    value={f('phone')}
                    onInput={(e) => set('phone', e.currentTarget.value)}
                  />
                </div>

                <div>
                  <label class="block text-sm font-medium text-gray-700 mb-1">Website</label>
                  <input
                    class="input"
                    type="url"
                    placeholder="https://acme.com"
                    value={f('website')}
                    onInput={(e) => set('website', e.currentTarget.value)}
                  />
                </div>

                {/* Industry Select */}
                <div>
                  <label class="block text-sm font-medium text-gray-700 mb-1">
                    Industry <span class="text-red-500">*</span>
                  </label>
                  <select
                    class="input"
                    value={f('industry')}
                    onChange={(e) => set('industry', e.currentTarget.value)}
                    required
                  >
                    <option value="">Select an industry...</option>
                    <option value="General Contractor">General Contractor (all trades)</option>
                    <Show when={!trades.loading && trades()}>
                      <For each={trades()!}>
                        {(trade) => <option value={trade}>{trade}</option>}
                      </For>
                    </Show>
                  </select>
                </div>

                {/* Description */}
                <div class="sm:col-span-2">
                  <label class="block text-sm font-medium text-gray-700 mb-1">
                    Company Description
                  </label>
                  <textarea
                    class="input"
                    rows={3}
                    placeholder="Briefly describe your company and the services you provide..."
                    value={f('description')}
                    onInput={(e) => set('description', e.currentTarget.value)}
                  />
                </div>
              </div>
            </div>

            {/* Address */}
            <div class="card">
              <h2 class="text-lg font-semibold text-gray-900 mb-4">Business Address</h2>
              <div class="grid grid-cols-1 sm:grid-cols-2 gap-4">
                <div class="sm:col-span-2">
                  <label class="block text-sm font-medium text-gray-700 mb-1">Address Line 1</label>
                  <input
                    class="input"
                    placeholder="123 Main St"
                    value={f('address_line1')}
                    onInput={(e) => set('address_line1', e.currentTarget.value)}
                  />
                </div>
                <div class="sm:col-span-2">
                  <label class="block text-sm font-medium text-gray-700 mb-1">Address Line 2</label>
                  <input
                    class="input"
                    placeholder="Suite 400"
                    value={f('address_line2')}
                    onInput={(e) => set('address_line2', e.currentTarget.value)}
                  />
                </div>
                <div>
                  <label class="block text-sm font-medium text-gray-700 mb-1">City</label>
                  <input
                    class="input"
                    placeholder="Denver"
                    value={f('city')}
                    onInput={(e) => set('city', e.currentTarget.value)}
                  />
                </div>
                <div>
                  <label class="block text-sm font-medium text-gray-700 mb-1">State / Province</label>
                  <input
                    class="input"
                    placeholder="CO"
                    value={f('state')}
                    onInput={(e) => set('state', e.currentTarget.value)}
                  />
                </div>
                <div>
                  <label class="block text-sm font-medium text-gray-700 mb-1">ZIP / Postal Code</label>
                  <input
                    class="input"
                    placeholder="80202"
                    value={f('zip')}
                    onInput={(e) => set('zip', e.currentTarget.value)}
                  />
                </div>
                <div>
                  <label class="block text-sm font-medium text-gray-700 mb-1">Country</label>
                  <input
                    class="input"
                    placeholder="US"
                    value={f('country')}
                    onInput={(e) => set('country', e.currentTarget.value)}
                  />
                </div>
              </div>
            </div>

            {/* Primary Contact */}
            <div class="card">
              <h2 class="text-lg font-semibold text-gray-900 mb-4">Primary Contact</h2>
              <div class="grid grid-cols-1 sm:grid-cols-2 gap-4">
                <div>
                  <label class="block text-sm font-medium text-gray-700 mb-1">
                    Full Name <span class="text-red-500">*</span>
                  </label>
                  <input
                    class="input"
                    placeholder="Jane Smith"
                    value={f('contact_name')}
                    onInput={(e) => set('contact_name', e.currentTarget.value)}
                    required
                  />
                </div>
                <div>
                  <label class="block text-sm font-medium text-gray-700 mb-1">Title</label>
                  <input
                    class="input"
                    placeholder="Project Manager"
                    value={f('contact_title')}
                    onInput={(e) => set('contact_title', e.currentTarget.value)}
                  />
                </div>
              </div>
            </div>

            {/* Admin Account */}
            <div class="card">
              <h2 class="text-lg font-semibold text-gray-900 mb-1">Admin Login Credentials</h2>
              <p class="text-sm text-gray-500 mb-4">
                Use these credentials to sign in to the platform and manage your projects.
              </p>
              <div class="grid grid-cols-1 sm:grid-cols-2 gap-4">
                <div class="sm:col-span-2">
                  <label class="block text-sm font-medium text-gray-700 mb-1">
                    Admin Email <span class="text-red-500">*</span>
                  </label>
                  <input
                    class="input"
                    type="email"
                    placeholder="you@acme.com"
                    value={f('admin_email')}
                    onInput={(e) => set('admin_email', e.currentTarget.value)}
                    required
                  />
                </div>
                <div>
                  <label class="block text-sm font-medium text-gray-700 mb-1">
                    Password <span class="text-red-500">*</span>
                  </label>
                  <input
                    class="input"
                    type="password"
                    placeholder="Min 8 characters"
                    value={f('admin_password')}
                    onInput={(e) => set('admin_password', e.currentTarget.value)}
                    required
                  />
                </div>
                <div>
                  <label class="block text-sm font-medium text-gray-700 mb-1">
                    Confirm Password <span class="text-red-500">*</span>
                  </label>
                  <input
                    class="input"
                    type="password"
                    placeholder="Repeat password"
                    value={f('admin_password_confirm')}
                    onInput={(e) => set('admin_password_confirm', e.currentTarget.value)}
                    required
                  />
                </div>
              </div>
            </div>

            {/* Error */}
            <Show when={error()}>
              <div class="px-4 py-3 rounded-lg bg-red-50 border border-red-200 text-red-700 text-sm">
                {error()}
              </div>
            </Show>

            {/* Submit */}
            <div class="flex items-center justify-between">
              <Show when={props.onNavigate}>
                <button
                  type="button"
                  class="text-sm text-gray-500 hover:underline"
                  onClick={() => props.onNavigate!('/login')}
                >
                  Already have an account? Sign in
                </button>
              </Show>
              <button
                type="submit"
                class="btn btn-primary ml-auto"
                disabled={submitting()}
              >
                {submitting() ? 'Submitting\u2026' : 'Create Account'}
              </button>
            </div>
          </form>
        </Show>
      </div>
    </div>
  );
};
